007    package au.org.intersect.exsite9.service;
008    
009    import java.util.List;
010    
011    import au.org.intersect.exsite9.domain.Group;
012    import au.org.intersect.exsite9.domain.MetadataAttributeValue;
013    import au.org.intersect.exsite9.domain.MetadataCategory;
014    import au.org.intersect.exsite9.domain.MetadataValue;
015    import au.org.intersect.exsite9.domain.ResearchFileSortField;
016    import au.org.intersect.exsite9.domain.SortFieldDirection;
017    import au.org.intersect.exsite9.dto.HierarchyMoveDTO;
018    
019    /**
020     * A Service for performing actions to Groups.
021     */
022    public interface IGroupService
023    {
024        /**
025         * Creates a new group.
026         * @param groupName The name of the new group to created.
027         * @param parentGroup The parent of this new group.
028         * @return The newly created Group.
029         */
030        Group createNewGroup(final String groupName);
031    
032        /**
033         * Deletes a group. Moves all the child groups and files to the provided group's parent.
034         * @param groupToDelete The group to delete.
035         */
036        void deleteGroup(final Group groupToDelete);
037    
038        /**
039         * Checks if it is safe to delete a group. Should be called before {@link IGroupService#deleteGroup(Group)}.
040         * @param groupToDelete The group to check.
041         * @return {@code null} if the provided group can be safely deleted, or an error message indicating why it cannot be safely deleted.
042         */
043        String deleteGroupCheck(final Group groupToDelete);
044        
045        /**
046         * Renames a group.
047         * @param groupToBeRenamed The group to renamed.
048         * @param NewName The new group name.
049         */
050        void renameGroup(final Group groupToBeRenamed, final String NewName);
051    
052        /**
053         * Adds a group to another group.
054         * @param parentGroup The group that will contain the child group.
055         * @param childGroup The group that will be added.
056         */
057        void addChildGroup(final Group parentGroup, final Group childGroup);
058        
059        /**
060         * Moves selected group members to a new group
061         * @param moveList The list of items to move
062         * @return {@code null} if successful, or a reason why the move was not successful.
063         */
064        String performHierarchyMove(final List<HierarchyMoveDTO> moveList);
065    
066        /**
067         * Associates a metadata category and value to a Group.
068         * @param group The group to associate metadata to.
069         * @param metadataCategory The category of metadata to associate.
070         * @param metadataVale The value of metadata to associate.
071         */
072        void associateMetadata(final Group group, final MetadataCategory metadataCategory, final MetadataValue metadataValue, final MetadataAttributeValue metadataAttributeValue);
073    
074        /**
075         * Diassociates a metadata category and value from a Group.
076         * @param group The group to disassociate metadata from.
077         * @param metadataCategory The category of metadata to disassociate.
078         * @param metadataValue The value of metadata to disassociate.
079         */
080        void disassociateMetadata(final Group group, final MetadataCategory metadataCategory, final MetadataValue metadataValue);
081    
082        void disassociateMetadataAttributeValue(final MetadataCategory metadataCategory, final MetadataAttributeValue metadataVale);
083    
084        void disassociateMultipleMetadataValues(final Group group, final MetadataCategory metadataCategory, final List<MetadataValue> metadataValues);
085    
086        /**
087         * Determines which groups are associated with a metadata category/value pair.
088         * @param metadataCategory The metadata category.
089         * @param metadataVale The metadata value.
090         * @return A list of group associated with a metadata category/value pair.
091         */
092        List<Group> getGroupsWithAssociatedMetadata(final MetadataCategory metadataCategory, final MetadataValue metadataValue);
093    
094        List<Group> getGroupsWithAssociatedMetadataAttribute(final MetadataCategory metadataCategory, final MetadataAttributeValue metadataAttributeValue);
095    
096        /**
097         * Obtains a group by its ID.
098         * @param groupID The ID of the group to get.
099         * @return The group, or {@code null} if it could not be found.
100         */
101        Group findGroupByID(final Long groupID);
102    
103        /**
104         * Configures a group's research files to be sorted by a particular field.
105         * @param group The group to configure the sort on.
106         * @param sortField The field to sort by.
107         * @param sortDirection The direction to sort by.
108         */
109        void sortResearchFilesInGroup(final Group group, final ResearchFileSortField sortField, final SortFieldDirection sortDirection);
110    }
