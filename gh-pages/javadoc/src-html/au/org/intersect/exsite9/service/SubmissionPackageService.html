007    package au.org.intersect.exsite9.service;
008    
009    import java.io.File;
010    import java.util.List;
011    
012    import javax.persistence.EntityManager;
013    import javax.persistence.EntityManagerFactory;
014    
015    import org.eclipse.jface.operation.IRunnableWithProgress;
016    
017    import au.org.intersect.exsite9.dao.GroupDAO;
018    import au.org.intersect.exsite9.dao.ProjectDAO;
019    import au.org.intersect.exsite9.dao.SubmissionPackageDAO;
020    import au.org.intersect.exsite9.dao.factory.ProjectDAOFactory;
021    import au.org.intersect.exsite9.dao.factory.SubmissionPackageDAOFactory;
022    import au.org.intersect.exsite9.domain.Group;
023    import au.org.intersect.exsite9.domain.Project;
024    import au.org.intersect.exsite9.domain.ResearchFile;
025    import au.org.intersect.exsite9.domain.SubmissionPackage;
026    import au.org.intersect.exsite9.xml.SIPXMLBuilder;
027    import au.org.intersect.exsite9.zip.SIPZIPBuilderRunnable;
028    
029    /**
030     * A service for manipulating {@link SubmissionPackage}s
031     */
032    public final class SubmissionPackageService implements ISubmissionPackageService
033    {
034        private final EntityManagerFactory entityManagerFactory;
035        private final SubmissionPackageDAOFactory submissionPackageDAOFactory;
036        private final ProjectDAOFactory projectDAOFactory;
037    
038        public SubmissionPackageService(final EntityManagerFactory entityManagerFactory,
039                                        final SubmissionPackageDAOFactory submissionPackageDAOFactory,
040                                        final ProjectDAOFactory projectDAOFactory)
041        {
042            this.entityManagerFactory = entityManagerFactory;
043            this.submissionPackageDAOFactory = submissionPackageDAOFactory;
044            this.projectDAOFactory = projectDAOFactory;
045        }
046    
047        /**
048         * @{inheritDoc}
049         */
050        @Override
051        public SubmissionPackage createSubmissionPackage(final Project project, final String name, final String description, final List<ResearchFile> researchFiles)
052        {
053            final SubmissionPackage submissionPackage = new SubmissionPackage();
054            submissionPackage.setName(name);
055            submissionPackage.setDescription(description);
056            submissionPackage.getResearchFiles().addAll(researchFiles);
057    
058            final EntityManager em = this.entityManagerFactory.createEntityManager();
059    
060            try
061            {
062                final SubmissionPackageDAO submissionPackageDAO = this.submissionPackageDAOFactory.createInstance(em);
063                submissionPackageDAO.createSubmissionPackage(submissionPackage);
064    
065                project.getSubmissionPackages().add(submissionPackage);
066                final ProjectDAO projectDAO = this.projectDAOFactory.createInstance(em);
067                projectDAO.updateProject(project);
068    
069                return submissionPackage;
070            }
071            finally
072            {
073                em.close();
074            }
075        }
076    
077        /**
078         * @{inheritDoc}
079         */
080        @Override
081        public SubmissionPackage updateSubmissionPackage(final SubmissionPackage submissionPackage, final String name, final String description, final List<ResearchFile> researchFiles)
082        {
083            submissionPackage.setName(name);
084            submissionPackage.setDescription(description);
085            submissionPackage.getResearchFiles().clear();
086            submissionPackage.getResearchFiles().addAll(researchFiles);
087    
088            final EntityManager em = this.entityManagerFactory.createEntityManager();
089    
090            try
091            {
092                final SubmissionPackageDAO submissionPackageDAO = this.submissionPackageDAOFactory.createInstance(em);
093                submissionPackageDAO.updateSubmissionPackage(submissionPackage);
094                return submissionPackage;
095            }
096            finally
097            {
098                em.close();
099            }
100        }
101    
102        /**
103         * @{inheritDoc}
104         */
105        @Override
106        public SubmissionPackage updateSubmissionPackage(final SubmissionPackage submissionPackage, final List<ResearchFile> researchFiles)
107        {
108            return updateSubmissionPackage(submissionPackage, submissionPackage.getName(), submissionPackage.getDescription(), researchFiles);
109        }
110    
111        /**
112         * @{inheritDoc}
113         */
114        @Override
115        public SubmissionPackage findSubmissionPackageById(final Long id)
116        {
117            final EntityManager em = this.entityManagerFactory.createEntityManager();
118            try
119            {
120                final SubmissionPackageDAO submissionPackageDAO = this.submissionPackageDAOFactory.createInstance(em);
121                return submissionPackageDAO.findSubmissionPackageById(id);
122            }
123            finally
124            {
125                em.close();
126            }
127        }
128    
129        /**
130         * @{inheritDoc}
131         */
132        @Override
133        public void deleteSubmissionPackage(final SubmissionPackage submissionPackage)
134        {
135            final EntityManager em = this.entityManagerFactory.createEntityManager();
136            try
137            {
138                final ProjectDAO projectDAO = this.projectDAOFactory.createInstance(em);
139    
140                final Project project = projectDAO.findProjectWithSubmissionPackage(submissionPackage);
141                project.getSubmissionPackages().remove(submissionPackage);
142                projectDAO.updateProject(project);
143    
144                final SubmissionPackageDAO submissionPackageDAO = this.submissionPackageDAOFactory.createInstance(em);
145                submissionPackageDAO.deleteSubmissionPackage(submissionPackage);
146            }
147            finally
148            {
149                em.close();
150            }
151        }
152    
153        /**
154         * @{inheritDoc}
155         */
156        @Override
157        public String buildXMLForSubmissionPackage(Project project, SubmissionPackage submissionPackage)
158        {
159            final EntityManager em = this.entityManagerFactory.createEntityManager();
160            try
161            {
162                return SIPXMLBuilder.buildXML(project, 
163                                              GroupDAO.getGroupsContainingSelectedFiles(submissionPackage.getResearchFiles()), 
164                                              submissionPackage, false);
165            }
166            finally
167            {
168                em.close();
169            }
170        }
171    
172        /**
173         * @{inheritDoc}
174         */
175        @Override
176        public IRunnableWithProgress buildZIPForSubmissionPackage(final Project project, final SubmissionPackage submissionPackage, final File fileToWrite)
177        {
178            final EntityManager em = this.entityManagerFactory.createEntityManager();
179    
180            try
181            {
182                final List<Group> selectedGroups = GroupDAO.getGroupsContainingSelectedFiles(submissionPackage.getResearchFiles());
183                return new SIPZIPBuilderRunnable(project, selectedGroups, submissionPackage, fileToWrite);
184            }
185            finally
186            {
187                em.close();
188            }
189    
190        }
191    
192        /**
193         * @{inheritDoc}
194         */
195        @Override
196        public List<SubmissionPackage> findSubmissionPackagesWithResearchFile(final ResearchFile researchFile)
197        {
198            final EntityManager em = this.entityManagerFactory.createEntityManager();
199            try
200            {
201                final SubmissionPackageDAO submissionPackageDAO = submissionPackageDAOFactory.createInstance(em);
202                return submissionPackageDAO.findSubmissionPackagesWithResearchFile(researchFile);
203            }
204            finally
205            {
206                em.close();
207            }
208        }
209    }
