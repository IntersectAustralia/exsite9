196        @Override
197        public Project removeFoldersFromProject(Project project, List<String> deletedFolderList)
198        {
199            EntityManager em = entityManagerFactory.createEntityManager();
200            try
201            {
202                final ProjectDAO projectDAO = this.projectDAOFactory.createInstance(em);
203                final FolderDAO folderDAO = this.folderDAOFactory.createInstance(em);
204                final ResearchFileDAO researchFileDAO = this.researchFileDAOFactory.createInstance(em);
205                final MetadataAssociationDAO metadataAssociationDAO = this.metaAssociationDAOFactory.createInstance(em);
206                final SubmissionPackageDAO submissionPackageDAO = this.submissionPackageDAOFactory.createInstance(em);
207    
208                project = projectDAO.findById(project.getId());
209    
210                Iterator<Folder> folderIter = project.getFolders().iterator();
211                while (folderIter.hasNext())
212                {
213                    Folder folder = folderIter.next();
214                    if (deletedFolderList.contains(folder.getFolder().getAbsolutePath()))
215                    {
216                        LOG.info("Removing folder id= " + folder.getId());
217    
218                        em.getTransaction().begin();
219    
220                        Iterator<ResearchFile> fileIter = folder.getFiles().iterator();
221                        while (fileIter.hasNext())
222                        {
223                            ResearchFile researchFile = fileIter.next();
224                            Group parentGroup = researchFile.getParentGroup();
225                            parentGroup.getResearchFiles().remove(researchFile);
226    
227                            Iterator<MetadataAssociation> maIter = researchFile.getMetadataAssociations().iterator();
228                            while (maIter.hasNext())
229                            {
230                                MetadataAssociation metadataAssociation = maIter.next();
231                                metadataAssociationDAO.removeMetadataAssociation(metadataAssociation);
232                                maIter.remove();
233                            }
234    
235                            // Remove the research file from any submission packages it is part of.
236                            final List<SubmissionPackage> submissionPackages = submissionPackageDAO
237                                    .findSubmissionPackagesWithResearchFile(researchFile);
238                            for (final SubmissionPackage submissionPackage : submissionPackages)
239                            {
240                                submissionPackage.getResearchFiles().remove(researchFile);
241                                submissionPackageDAO.updateSubmissionPackage(submissionPackage);
242                            }
243    
244                            fileIter.remove();
245                            researchFileDAO.removeResearchFile(researchFile);
246                        }
247                        folderIter.remove();
248                        folderDAO.removeFolder(folder);
249                        em.getTransaction().commit();
250                    }
251                }
252                return project;
253            }
254            finally
255            {
256                em.close();
257            }
258        }

260        @Override
261        public void updateFolderPath(long folderId, File newFileForFolder)
262        {
263            EntityManager em = entityManagerFactory.createEntityManager();
264            try
265            {
266                final FolderDAO folderDAO = this.folderDAOFactory.createInstance(em);
267                final ResearchFileDAO researchFileDAO = this.researchFileDAOFactory.createInstance(em);
268    
269                Folder folder = folderDAO.findById(folderId);
270                
271                String newFolderPath = newFileForFolder.getAbsolutePath();
272    
273                // modify the path in each of the research files
274                List<ResearchFile> researchFiles = folder.getFiles();
275    
276                for (ResearchFile researchFile : researchFiles)
277                {
278                    String originalPath = researchFile.getFile().getAbsolutePath();
279                    
280                    if(! originalPath.startsWith(newFolderPath))
281                    {
282                        String newPath = originalPath.replace(folder.getFolder().getAbsolutePath(),newFolderPath);
283                        File replacementFile = new File(newPath);
284                        researchFile.setFile(replacementFile);
285                        researchFileDAO.updateResearchFile(researchFile);
286                    }
287                }
288    
289                folder.setFolder(newFileForFolder);
290                folderDAO.updateFolder(folder);
291            }
292            finally
293            {
294                em.close();
295            }
296        }

304        @Override
305        public void removeResearchFileFromSystem(long fileToBeRemovedId)
306        {
307            EntityManager em = entityManagerFactory.createEntityManager();
308            try
309            {
310                final ResearchFileDAO researchFileDAO = this.researchFileDAOFactory.createInstance(em);
311                final MetadataAssociationDAO metadataAssociationDAO = this.metaAssociationDAOFactory.createInstance(em);
312                final SubmissionPackageDAO submissionPackageDAO = this.submissionPackageDAOFactory.createInstance(em);
313                final FolderDAO folderDAO = this.folderDAOFactory.createInstance(em);
314    
315                ResearchFile fileToBeRemoved = researchFileDAO.findById(fileToBeRemovedId);
316                
317                em.getTransaction().begin();
318    
319                Group parentGroup = fileToBeRemoved.getParentGroup();
320                parentGroup.getResearchFiles().remove(fileToBeRemoved);
321    
322                Iterator<MetadataAssociation> maIter = fileToBeRemoved.getMetadataAssociations().iterator();
323                while (maIter.hasNext())
324                {
325                    MetadataAssociation metadataAssociation = maIter.next();
326                    metadataAssociationDAO.removeMetadataAssociation(metadataAssociation);
327                    maIter.remove();
328                }
329    
330                // Remove the research file from any submission packages it is part of.
331                final List<SubmissionPackage> submissionPackages = submissionPackageDAO
332                        .findSubmissionPackagesWithResearchFile(fileToBeRemoved);
333                for (final SubmissionPackage submissionPackage : submissionPackages)
334                {
335                    submissionPackage.getResearchFiles().remove(fileToBeRemoved);
336                    submissionPackageDAO.updateSubmissionPackage(submissionPackage);
337                }
338                
339                Folder folder = researchFileDAO.getParentFolder(fileToBeRemoved);
340                folder.getFiles().remove(fileToBeRemoved);
341                folderDAO.updateFolder(folder);
342    
343                researchFileDAO.removeResearchFile(fileToBeRemoved);
344                em.getTransaction().commit();
345            }
346            finally
347            {
348                em.close();
349            }
350        }
