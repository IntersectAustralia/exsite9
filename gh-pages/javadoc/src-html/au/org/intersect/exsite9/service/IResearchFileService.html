007    package au.org.intersect.exsite9.service;
008    
009    import java.util.List;
010    
011    import au.org.intersect.exsite9.domain.Folder;
012    import au.org.intersect.exsite9.domain.MetadataAttributeValue;
013    import au.org.intersect.exsite9.domain.MetadataCategory;
014    import au.org.intersect.exsite9.domain.MetadataValue;
015    import au.org.intersect.exsite9.domain.Project;
016    import au.org.intersect.exsite9.domain.ResearchFile;
017    
018    /**
019     * A Service to provide actions that may be performed with Research Files.
020     */
021    public interface IResearchFileService 
022    {
023        /**
024         * Identifies new files for a project.
025         * @param project The project to identify new files for.
026         */
027            public void identifyNewFilesForProject(Project project);
028            
029            /**
030             * Identifies new files for a project in a folder.
031             * @param project The project to identify new files for.
032             * @param folder The folder to identify new files in.
033             */
034            public void identifyNewFilesForFolder(Project project, Folder folder);
035            
036            /**
037             * Associates metadata to a research file.
038             * @param file The file.
039             * @param metadataCategory The metadata category to associate.
040             * @param metadataValue The metadata value to associate.
041             */
042            public void associateMetadata(final ResearchFile file, final MetadataCategory metadataCategory, final MetadataValue metadataValue, final MetadataAttributeValue metadataAttributeValue);
043    
044            /**
045             * Disassociates metadata from a research file. 
046             * @param file The file.
047             * @param metadataCategory The metadata category to disassociate.
048             * @param metadataValue The metadata value to disassociate.
049             */
050            public void disassociateMetadata(final ResearchFile file, final MetadataCategory metadataCategory, final MetadataValue metadataValue);
051    
052            public void disassociateMetadataAttributeValue(final MetadataCategory metadataCategory, final MetadataAttributeValue metadataAttributeValue);
053    
054            /**
055             * Dissassociates multiple metadata values from a research file.
056             * @param file The file.
057         * @param metadataCategory The metadata category to disassociate.
058         * @param metadataValue The metadata values to disassociate.
059             */
060            public void disassociateMultipleMetadataValues(final ResearchFile file, final MetadataCategory metadataCategory, final List<MetadataValue> metadataValues);
061    
062            /**
063             * Obtains research files with a metadata association.
064             * @param metadataCategory The metadata category of the association.
065             * @param metadataValue The metadata value of the assocation.
066             * @return The research files with the provided assocation.
067             */
068            public List<ResearchFile> getResearchFilesWithAssociatedMetadata(final MetadataCategory metadataCategory, final MetadataValue metadataValue);
069    
070            public List<ResearchFile> getResearchFilesWithAssociatedMetadataAttribute(MetadataCategory metadataCategory, MetadataAttributeValue toDelete);
071    
072            /**
073             * Obtains a research file by it's ID.
074             * @param id The ID of the research file to find.
075             * @return The research file.
076             */
077            public ResearchFile findResearchFileByID(final Long id);
078    
079            /**
080             * Updates a research file.
081             * @param selectionObject The research file to update.
082             */
083        public void updateResearchFile(ResearchFile selectionObject);
084    
085        /**
086         * Consolidates a sub folder into its parent folder.
087         * @param project The project to perform the consolidation for.
088         * @param parentFolder The parent folder.
089         * @param subFolder The sub folder.
090         */
091        public void consolidateSubFolderIntoParentFolder(final Project project, final Folder parentFolder, final Folder subFolder);
092        
093        /**
094         * Import an existing folder structure into the project creating new Groups for folders and new ResearchFiles for files.
095         * The created group will be a child of the Project.
096         * @param project
097         * @param folder
098         */
099        public void importFolderStructureForProject(final Project project, Folder folder);
100    
101        public void changeAFilesParentFolder(ResearchFile researchFile, long newFolderId);
102    }
