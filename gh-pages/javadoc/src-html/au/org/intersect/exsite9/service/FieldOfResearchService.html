007    package au.org.intersect.exsite9.service;
008    
009    import java.util.List;
010    
011    import javax.persistence.EntityManager;
012    import javax.persistence.EntityManagerFactory;
013    
014    import au.org.intersect.exsite9.dao.FieldOfResearchDAO;
015    import au.org.intersect.exsite9.dao.factory.FieldOfResearchDAOFactory;
016    import au.org.intersect.exsite9.domain.FieldOfResearch;
017    
018    public final class FieldOfResearchService implements IFieldOfResearchService
019    {
020        private final EntityManagerFactory emf;
021        private final FieldOfResearchDAOFactory fieldOfResearchDAOFactory;
022    
023        public FieldOfResearchService(final EntityManagerFactory entityManagerFactory, final FieldOfResearchDAOFactory fieldOfResearchDAOFactory)
024        {
025            this.emf = entityManagerFactory;
026            this.fieldOfResearchDAOFactory = fieldOfResearchDAOFactory;
027        }
028    
029        /**
030         * @{inheritDoc}
031         */
032        @Override
033        public List<FieldOfResearch> getAll()
034        {
035            final EntityManager em = this.emf.createEntityManager();
036            try
037            {
038                final FieldOfResearchDAO fieldOfResearchDAO = fieldOfResearchDAOFactory.createInstance(em);
039                return fieldOfResearchDAO.getAll();
040            }
041            finally
042            {
043                em.close();
044            }
045        }
046    }
