083        private void addFilesFromFolder(Project project, Folder folder)
084        {
085            EntityManager em = entityManagerFactory.createEntityManager();
086            try
087            {
088                if (!folder.getFolder().exists())
089                {
090                    LOG.info("Cannot identify new files in folder '" + folder.getFolder().getAbsolutePath() + "'. It does not exist.");
091                    return;
092                }
093    
094                final List<File> newFileList = FolderUtils.getAllFilesInFolder(folder);
095                final ProjectDAO projectDAO = projectDAOFactory.createInstance(em);
096                final ResearchFileDAO researchFileDAO = researchFileDAOFactory.createInstance(em);
097                final FolderDAO folderDAO = folderDAOFactory.createInstance(em);
098                for (final File file : newFileList)
099                {
100                    // If there is already a research file in the database, do not insert another.
101                    if(researchFileDAO.findByPath(project, file) == null)
102                    {
103                        final ResearchFile researchFile = new ResearchFile(file);
104                        researchFile.setProject(project);
105                        researchFile.setParentGroup(project.getNewFilesNode());
106                        researchFileDAO.createResearchFile(researchFile);
107                        folder.getFiles().add(researchFile);
108                        folderDAO.updateFolder(folder);
109                        project.getNewFilesNode().getResearchFiles().add(researchFile);
110                    }
111                }
112                projectDAO.updateProject(project);
113            }
114            finally
115            {
116                em.close();
117            }
118        }

120        @Override
121        public void associateMetadata(final ResearchFile file, final MetadataCategory metadataCategory, final MetadataValue metadataValue, final MetadataAttributeValue metadataAttributeValue)
122        {
123    
124            LOG.info("Assosciating metadata with file. " + file + " " + metadataCategory + " " + metadataValue);
125    
126            final EntityManager em = this.entityManagerFactory.createEntityManager();
127            try
128            {
129                final List<MetadataAssociation> existingAssociations = file.getMetadataAssociations();
130    
131                final MetadataAssociationDAO metadataAssociationDAO = this.metadataAssociationDAOFactory.createInstance(em);
132        
133                boolean addedAssociation = false;
134    
135                if (metadataCategory.getType() == MetadataCategoryType.CONTROLLED_VOCABULARY)
136                {
137                    for (final MetadataAssociation existingAssociation : existingAssociations)
138                    {
139                        if (existingAssociation.getMetadataCategory().equals(metadataCategory))
140                        {
141                            if (existingAssociation.getMetadataValues().contains(metadataValue))
142                            {
143                                // nothing to do!
144                                return;
145                            }
146        
147                            if (metadataCategory.getType() == MetadataCategoryType.FREETEXT)
148                            {
149                                existingAssociation.getMetadataValues().clear();
150                            }
151                            existingAssociation.getMetadataValues().add(metadataValue);
152                            existingAssociation.setMetadataAttributeValue(metadataAttributeValue);
153                            metadataAssociationDAO.updateMetadataAssociation(existingAssociation);
154                            addedAssociation = true;
155                        }
156                    }
157                }
158    
159                if (metadataCategory.getType() == MetadataCategoryType.FREETEXT || !addedAssociation)
160                {
161                    final MetadataAssociation metadataAssociation = new MetadataAssociation(metadataCategory);
162                    metadataAssociation.getMetadataValues().add(metadataValue);
163                    metadataAssociation.setMetadataAttributeValue(metadataAttributeValue);
164                    metadataAssociationDAO.createMetadataAssociation(metadataAssociation);
165                    file.getMetadataAssociations().add(metadataAssociation);
166                }
167                
168                final ResearchFileDAO researchFileDAO = this.researchFileDAOFactory.createInstance(em);
169                researchFileDAO.updateResearchFile(file);
170            }
171            finally
172            {
173                em.close();
174            }
175        
176        }

178        @Override
179        public void disassociateMetadata(final ResearchFile file, final MetadataCategory metadataCategory, final MetadataValue metadataValue)
180        {
181            LOG.info("Disassosciating metadata from file. " + file + " " + metadataCategory + " " + metadataValue);
182    
183            final List<MetadataAssociation> existingAssociations = file.getMetadataAssociations();
184    
185            for (final MetadataAssociation existingAssociation : existingAssociations)
186            {
187                if (existingAssociation.getMetadataCategory().equals(metadataCategory))
188                {
189                    if (existingAssociation.getMetadataValues().remove(metadataValue))
190                    {
191                        final EntityManager em = this.entityManagerFactory.createEntityManager();
192                        try
193                        {
194                            final MetadataAssociationDAO metadataAssociationDAO = this.metadataAssociationDAOFactory.createInstance(em);
195                            metadataAssociationDAO.updateMetadataAssociation(existingAssociation);
196    
197                            if (existingAssociation.getMetadataValues().isEmpty())
198                            {
199                                existingAssociations.remove(existingAssociation);
200    
201                                final ResearchFileDAO researchFileDAO = this.researchFileDAOFactory.createInstance(em);
202                                researchFileDAO.updateResearchFile(file);
203    
204                                metadataAssociationDAO.removeMetadataAssociation(existingAssociation);
205                            }
206                            return;
207                        }
208                        finally
209                        {
210                            em.close();
211                        }
212                    }
213                }
214            }
215        }

217        @Override
218        public void disassociateMetadataAttributeValue(final MetadataCategory metadataCategory, final MetadataAttributeValue metadataAttributeValue)
219        {
220            final List<ResearchFile> files = getResearchFilesWithAssociatedMetadataAttribute(metadataCategory, metadataAttributeValue);
221            for (final ResearchFile file : files)
222            {
223                final List<MetadataAssociation> associations = file.getMetadataAssociations();
224                for (final MetadataAssociation association : associations)
225                {
226                    final MetadataAttributeValue currentMetadataAttributeValue = association.getMetadataAttributeValue();
227                    if (currentMetadataAttributeValue == null)
228                    {
229                        continue;
230                    }
231    
232                    if (association.getMetadataCategory().equals(metadataCategory) && currentMetadataAttributeValue.equals(metadataAttributeValue))
233                    {
234                        association.setMetadataAttributeValue(null);
235                        final EntityManager em = this.entityManagerFactory.createEntityManager();
236                        try
237                        {
238                            final MetadataAssociationDAO metadataAssociationDAO = this.metadataAssociationDAOFactory.createInstance(em);
239                            metadataAssociationDAO.updateMetadataAssociation(association);
240                        }
241                        finally
242                        {
243                            em.close();
244                        }
245                    }
246                }
247            }
248        }

250        @Override
251        public List<ResearchFile> getResearchFilesWithAssociatedMetadata(MetadataCategory metadataCategory, MetadataValue metadataValue)
252        {
253            final EntityManager em = entityManagerFactory.createEntityManager();
254            try
255            {
256                final String queryJQL = "SELECT f FROM ResearchFile f JOIN f.metadataAssociations a WHERE a.metadataCategory = :category AND :value MEMBER OF a.metadataValues";
257                final TypedQuery<ResearchFile> query = em.createQuery(queryJQL, ResearchFile.class);
258                query.setParameter("category", metadataCategory);
259                query.setParameter("value", metadataValue);
260                return query.getResultList();
261            }
262            finally
263            {
264                em.close();
265            }
266        }

326        public void consolidateSubFolderIntoParentFolder(final Project project, final Folder parentFolder, final Folder subFolder)
327        {
328            final EntityManager em = entityManagerFactory.createEntityManager();
329            try
330            {
331                final ProjectDAO projectDAO = projectDAOFactory.createInstance(em);
332                final FolderDAO folderDAO = folderDAOFactory.createInstance(em);
333                
334                parentFolder.getFiles().addAll(subFolder.getFiles());
335                subFolder.getFiles().clear();
336    
337                folderDAO.updateFolder(parentFolder);
338                folderDAO.updateFolder(subFolder);
339    
340                project.getFolders().remove(subFolder);
341                projectDAO.updateProject(project);
342                folderDAO.removeFolder(subFolder);
343            }
344            finally
345            {
346                em.close();
347            }

350        @Override
351        public void importFolderStructureForProject(Project project, Folder folder)
352        {
353            final EntityManager em = entityManagerFactory.createEntityManager();
354            
355            try
356            {
357                final ProjectDAO projectDAO = projectDAOFactory.createInstance(em);
358                final FolderDAO folderDAO = folderDAOFactory.createInstance(em);
359                
360                em.getTransaction().begin();
361                
362                List<ResearchFile> folderResearchFiles = new ArrayList<ResearchFile>();
363                importFolder(em, project, folderResearchFiles, project.getRootNode(), folder.getFolder());
364                
365                folder.getFiles().addAll(folderResearchFiles);
366                folderDAO.createFolder(folder);
367                project.getFolders().add(folder);
368                projectDAO.updateProject(project);
369                
370                em.getTransaction().commit();
371            }
372            catch(Exception e)
373            {
374                em.getTransaction().rollback();
375            }
376            finally
377            {
378                em.close();
379            }
380        }

382        private void importFolder(EntityManager em, Project project, List<ResearchFile> folderResearchFiles, Group parentGroup, File folder)
383        {
384            LOG.debug("Import Folder: " + folder.getName());
385            
386            final ResearchFileDAO researchFileDAO = researchFileDAOFactory.createInstance(em);
387            final GroupDAO groupDAO = groupDAOFactory.createInstance(em);
388            
389            // Create group for folder
390            Group newGroup = null;
391            
392            for(Group group : parentGroup.getGroups())
393            {
394                if(group.getName().equalsIgnoreCase(folder.getName()))
395                {
396                    newGroup = groupDAO.findById(group.getId());
397                    break;
398                }
399            }
400            
401            if(newGroup == null)
402            {
403                newGroup = new Group(folder.getName());
404                newGroup.setParentGroup(parentGroup);
405                newGroup.setResearchFileSortDirection(parentGroup.getResearchFileSortDirection());
406                newGroup.setResearchFileSortField(parentGroup.getResearchFileSortField());
407                parentGroup.getGroups().add(newGroup);
408                groupDAO.createGroup(newGroup);
409                groupDAO.updateGroup(parentGroup);
410            }
411                
412            // Create Research files for files
413            List<File> folderList = new ArrayList<File>(0);
414            for(File file : folder.listFiles())
415            {
416                if(file.isDirectory())
417                {
418                    folderList.add(file);
419                }
420                else
421                {
422                    boolean createFile = false;
423                    
424                    if(researchFileDAO.findByPath(project, file) == null){
425                        createFile = true;
426                    }
427                    
428                    if (createFile)
429                    {
430                        ResearchFile researchFile = new ResearchFile(file);
431                        researchFile.setProject(project);
432                        researchFile.setParentGroup(newGroup);
433                        researchFileDAO.createResearchFile(researchFile);
434                        folderResearchFiles.add(researchFile);
435                        newGroup.getResearchFiles().add(researchFile);
436                    }
437                }
438            }
439    
440            // Recurse through the folders
441            if(! folderList.isEmpty())
442            {
443                for(File newFolder : folderList)
444                {
445                    importFolder(em, project, folderResearchFiles, newGroup, newFolder);
446                }
447            }
448            
449            groupDAO.updateGroup(newGroup);
450        }

452        @Override
453        public void changeAFilesParentFolder(ResearchFile researchFile, long newFolderId)
454        {
455            EntityManager em = entityManagerFactory.createEntityManager();
456            try
457            {
458                final ResearchFileDAO researchFileDAO = this.researchFileDAOFactory.createInstance(em);
459                final FolderDAO folderDAO = this.folderDAOFactory.createInstance(em);
460                
461                Folder folder = researchFileDAO.getParentFolder(researchFile);
462                folder.getFiles().remove(researchFile);
463                Folder newFolder = folderDAO.findById(newFolderId);
464                newFolder.getFiles().add(researchFile);
465                folderDAO.updateFolder(folder);
466                folderDAO.updateFolder(newFolder);
467    
468            }
469            finally
470            {
471                em.close();
472            }
473        }
