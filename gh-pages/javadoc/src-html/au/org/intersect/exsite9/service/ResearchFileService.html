074        private void addFilesFromFolder(Project project, Folder folder)
075        {
076            EntityManager em = entityManagerFactory.createEntityManager();
077            try
078            {
079                if (!folder.getFolder().exists())
080                {
081                    LOG.info("Cannot identify new files in folder '" + folder.getFolder().getAbsolutePath() + "'. It does not exist.");
082                    return;
083                }
084    
085                final List<File> newFileList = FolderUtils.getAllFilesInFolder(folder);
086                final ProjectDAO projectDAO = projectDAOFactory.createInstance(em);
087                final ResearchFileDAO researchFileDAO = researchFileDAOFactory.createInstance(em);
088                final FolderDAO folderDAO = folderDAOFactory.createInstance(em);
089                for (final File file : newFileList)
090                {
091                    // If there is already a research file in the database, do not insert another.
092                    if(researchFileDAO.findByPath(project, file) == null)
093                    {
094                        final ResearchFile researchFile = new ResearchFile(file);
095                        researchFile.setProject(project);
096                        researchFile.setParentGroup(project.getNewFilesNode());
097                        researchFileDAO.createResearchFile(researchFile);
098                        folder.getFiles().add(researchFile);
099                        folderDAO.updateFolder(folder);
100                        project.getNewFilesNode().getResearchFiles().add(researchFile);
101                    }
102                }
103                projectDAO.updateProject(project);
104            }
105            finally
106            {
107                em.close();
108            }
109        }

111        @Override
112        public void associateMetadata(final ResearchFile file, final MetadataCategory metadataCategory, final MetadataValue metadataValue, final MetadataAttributeValue metadataAttributeValue)
113        {
114    
115            LOG.info("Assosciating metadata with file. " + file + " " + metadataCategory + " " + metadataValue);
116    
117            final EntityManager em = this.entityManagerFactory.createEntityManager();
118            try
119            {
120                final List<MetadataAssociation> existingAssociations = file.getMetadataAssociations();
121    
122                final MetadataAssociationDAO metadataAssociationDAO = this.metadataAssociationDAOFactory.createInstance(em);
123        
124                boolean addedAssociation = false;
125                for (final MetadataAssociation existingAssociation : existingAssociations)
126                {
127                    if (existingAssociation.getMetadataCategory().equals(metadataCategory))
128                    {
129                        if (existingAssociation.getMetadataValues().contains(metadataValue))
130                        {
131                            // nothing to do!
132                            return;
133                        }
134    
135                        if (metadataCategory.getType() == MetadataCategoryType.FREETEXT)
136                        {
137                            existingAssociation.getMetadataValues().clear();
138                        }
139                        existingAssociation.getMetadataValues().add(metadataValue);
140                        metadataAssociationDAO.updateMetadataAssociation(existingAssociation);
141                        addedAssociation = true;
142                    }
143                }
144    
145                if (!addedAssociation)
146                {
147                    final MetadataAssociation metadataAssociation = new MetadataAssociation(metadataCategory);
148                    metadataAssociation.getMetadataValues().add(metadataValue);
149                    metadataAssociation.setMetadataAttributeValue(metadataAttributeValue);
150                    metadataAssociationDAO.createMetadataAssociation(metadataAssociation);
151                    file.getMetadataAssociations().add(metadataAssociation);
152                }
153                
154                final ResearchFileDAO researchFileDAO = this.researchFileDAOFactory.createInstance(em);
155                researchFileDAO.updateResearchFile(file);
156            }
157            finally
158            {
159                em.close();
160            }
161        
162        }

164        @Override
165        public void disassociateMetadata(final ResearchFile file, final MetadataCategory metadataCategory, final MetadataValue metadataValue)
166        {
167            LOG.info("Disassosciating metadata from file. " + file + " " + metadataCategory + " " + metadataValue);
168    
169            final List<MetadataAssociation> existingAssociations = file.getMetadataAssociations();
170    
171            for (final MetadataAssociation existingAssociation : existingAssociations)
172            {
173                if (existingAssociation.getMetadataCategory().equals(metadataCategory))
174                {
175                    if (existingAssociation.getMetadataValues().remove(metadataValue))
176                    {
177                        final EntityManager em = this.entityManagerFactory.createEntityManager();
178                        try
179                        {
180                            final MetadataAssociationDAO metadataAssociationDAO = this.metadataAssociationDAOFactory.createInstance(em);
181                            metadataAssociationDAO.updateMetadataAssociation(existingAssociation);
182    
183                            if (existingAssociation.getMetadataValues().isEmpty())
184                            {
185                                existingAssociations.remove(existingAssociation);
186    
187                                final ResearchFileDAO researchFileDAO = this.researchFileDAOFactory.createInstance(em);
188                                researchFileDAO.updateResearchFile(file);
189    
190                                metadataAssociationDAO.removeMetadataAssociation(existingAssociation);
191                            }
192                            return;
193                        }
194                        finally
195                        {
196                            em.close();
197                        }
198                    }
199                }
200            }
201        }

203        @Override
204        public void disassociateMetadataAttributeValue(final MetadataCategory metadataCategory, final MetadataAttributeValue metadataAttributeValue)
205        {
206            final List<ResearchFile> files = getResearchFilesWithAssociatedMetadataAttribute(metadataCategory, metadataAttributeValue);
207            for (final ResearchFile file : files)
208            {
209                final List<MetadataAssociation> associations = file.getMetadataAssociations();
210                for (final MetadataAssociation association : associations)
211                {
212                    if (association.getMetadataCategory().equals(metadataCategory) && association.getMetadataAttributeValue().equals(metadataAttributeValue))
213                    {
214                        association.setMetadataAttributeValue(null);
215                        final EntityManager em = this.entityManagerFactory.createEntityManager();
216                        try
217                        {
218                            final MetadataAssociationDAO metadataAssociationDAO = this.metadataAssociationDAOFactory.createInstance(em);
219                            metadataAssociationDAO.updateMetadataAssociation(association);
220                        }
221                        finally
222                        {
223                            em.close();
224                        }
225                    }
226                }
227            }
228        }

230        @Override
231        public List<ResearchFile> getResearchFilesWithAssociatedMetadata(MetadataCategory metadataCategory, MetadataValue metadataValue)
232        {
233            final EntityManager em = entityManagerFactory.createEntityManager();
234            try
235            {
236                final String queryJQL = "SELECT f FROM ResearchFile f JOIN f.metadataAssociations a WHERE a.metadataCategory = :category AND :value MEMBER OF a.metadataValues";
237                final TypedQuery<ResearchFile> query = em.createQuery(queryJQL, ResearchFile.class);
238                query.setParameter("category", metadataCategory);
239                query.setParameter("value", metadataValue);
240                return query.getResultList();
241            }
242            finally
243            {
244                em.close();
245            }
246        }

305        @Override
306        public void consolidateSubFolderIntoParentFolder(final Project project, final Folder parentFolder, final Folder subFolder)
307        {
308            final EntityManager em = entityManagerFactory.createEntityManager();
309            try
310            {
311                final ProjectDAO projectDAO = projectDAOFactory.createInstance(em);
312                final FolderDAO folderDAO = folderDAOFactory.createInstance(em);
313                
314                parentFolder.getFiles().addAll(subFolder.getFiles());
315                subFolder.getFiles().clear();
316    
317                folderDAO.updateFolder(parentFolder);
318                folderDAO.updateFolder(subFolder);
319    
320                project.getFolders().remove(subFolder);
321                projectDAO.updateProject(project);
322                folderDAO.removeFolder(subFolder);
323            }
324            finally
325            {
326                em.close();
327            }
328        }

330        @Override
331        public void importFolderStructureForProject(Project project, Folder folder)
332        {
333            final EntityManager em = entityManagerFactory.createEntityManager();
334            
335            try
336            {
337                final ProjectDAO projectDAO = projectDAOFactory.createInstance(em);
338                final FolderDAO folderDAO = folderDAOFactory.createInstance(em);
339                
340                em.getTransaction().begin();
341                
342                List<ResearchFile> folderResearchFiles = new ArrayList<ResearchFile>();
343                importFolder(em, project, folderResearchFiles, project.getRootNode(), folder.getFolder());
344                
345                folder.getFiles().addAll(folderResearchFiles);
346                folderDAO.createFolder(folder);
347                project.getFolders().add(folder);
348                projectDAO.updateProject(project);
349                
350                em.getTransaction().commit();
351            }
352            catch(Exception e)
353            {
354                em.getTransaction().rollback();
355            }
356            finally
357            {
358                em.close();
359            }
360        }

362        private void importFolder(EntityManager em, Project project, List<ResearchFile> folderResearchFiles, Group parentGroup, File folder)
363        {
364            LOG.debug("Import Folder: " + folder.getName());
365            
366            final ResearchFileDAO researchFileDAO = researchFileDAOFactory.createInstance(em);
367            final GroupDAO groupDAO = groupDAOFactory.createInstance(em);
368            
369            // Create group for folder
370            Group newGroup = null;
371            
372            for(Group group : parentGroup.getGroups())
373            {
374                if(group.getName().equalsIgnoreCase(folder.getName()))
375                {
376                    newGroup = groupDAO.findById(group.getId());
377                    break;
378                }
379            }
380            
381            if(newGroup == null)
382            {
383                newGroup = new Group(folder.getName());
384                newGroup.setParentGroup(parentGroup);
385                newGroup.setResearchFileSortDirection(parentGroup.getResearchFileSortDirection());
386                newGroup.setResearchFileSortField(parentGroup.getResearchFileSortField());
387                parentGroup.getGroups().add(newGroup);
388                groupDAO.createGroup(newGroup);
389                groupDAO.updateGroup(parentGroup);
390            }
391                
392            // Create Research files for files
393            List<File> folderList = new ArrayList<File>(0);
394            for(File file : folder.listFiles())
395            {
396                if(file.isDirectory())
397                {
398                    folderList.add(file);
399                }
400                else
401                {
402                    boolean createFile = false;
403                    
404                    if(researchFileDAO.findByPath(project, file) == null){
405                        createFile = true;
406                    }
407                    
408                    if (createFile)
409                    {
410                        ResearchFile researchFile = new ResearchFile(file);
411                        researchFile.setProject(project);
412                        researchFile.setParentGroup(newGroup);
413                        researchFileDAO.createResearchFile(researchFile);
414                        folderResearchFiles.add(researchFile);
415                        newGroup.getResearchFiles().add(researchFile);
416                    }
417                }
418            }
419    
420            // Recurse through the folders
421            if(! folderList.isEmpty())
422            {
423                for(File newFolder : folderList)
424                {
425                    importFolder(em, project, folderResearchFiles, newGroup, newFolder);
426                }
427            }
428            
429            groupDAO.updateGroup(newGroup);
430        }

432        @Override
433        public void changeAFilesParentFolder(ResearchFile researchFile, long newFolderId)
434        {
435            EntityManager em = entityManagerFactory.createEntityManager();
436            try
437            {
438                final ResearchFileDAO researchFileDAO = this.researchFileDAOFactory.createInstance(em);
439                final FolderDAO folderDAO = this.folderDAOFactory.createInstance(em);
440                
441                Folder folder = researchFileDAO.getParentFolder(researchFile);
442                folder.getFiles().remove(researchFile);
443                Folder newFolder = folderDAO.findById(newFolderId);
444                newFolder.getFiles().add(researchFile);
445                folderDAO.updateFolder(folder);
446                folderDAO.updateFolder(newFolder);
447    
448            }
449            finally
450            {
451                em.close();
452            }
453        }
