017    /**
018     * For performing operations with {@link MetadataCategory}
019     */
020    public interface IMetadataCategoryService
021    {

034         * @param type The type of the metadata category.
035         * @param use The use of the metadata category.
036         * @param imported {@code true} if this metadata category was imported.
037         * @param values The allowable values for this metadata category.
038         * @param metadataAttribute The attribute for this metadata category. May be {@code null}.
039         * @return The newly created metadata category.
040         */
041        MetadataCategory createNewMetadataCategory(final String name, final String description, final MetadataCategoryType type, final MetadataCategoryUse use,
042            final boolean inextensible, final boolean imported, final List<MetadataValue> values, final MetadataAttribute metadataAttribute);
