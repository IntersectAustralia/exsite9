007    package au.org.intersect.exsite9.service;
008    
009    import java.io.File;
010    import java.util.List;
011    
012    import au.org.intersect.exsite9.domain.Folder;
013    import au.org.intersect.exsite9.domain.Project;
014    import au.org.intersect.exsite9.domain.Schema;
015    import au.org.intersect.exsite9.dto.ProjectFieldsDTO;
016    
017    public interface IProjectService
018    {
019        /**
020         * Creates a project.
021         * @param projectFields the various fields that can be inputed when creating a project.
022         * @param schema the schema to use for the project.
023         * @return The newly created project.
024         */
025        Project createProject(final ProjectFieldsDTO projectFields, final Schema schema);
026    
027        /**
028         * Maps a folder to a project.
029         * @param project The project.
030         * @param folder The folder to map to the project.
031         */
032        void mapFolderToProject(final Project project, final Folder folder);
033        
034        /**
035         * Remove folders from a project. This will also remove all files added to the project
036         * because they were from a removed folder.
037         * @param project The current project
038         * @param deletedFolderList The list of folders to be removed
039         */
040        public Project removeFoldersFromProject(final Project project, final List<String> deletedFolderList);
041        
042        /**
043         * finds all projects in the database.
044         * @return list of projects
045         */
046        public List<Project> getAllProjects();
047        
048        /**
049         * Edit a project.
050         * @param projectFields the various fields that can be inputed when editing a project.
051         * @param id of project to update 
052         * @return The updated project.
053         */
054        Project editProject(final ProjectFieldsDTO projectFields, final Long id);
055    
056        /**
057         * Edit a project
058         * @param schema The new schema to assign to the project.
059         * @param id of the project to update.
060         * @return The updated project.
061         */
062        Project editProject(final Schema schema, final Long id);
063    
064        
065        Project findProjectById(Long id);
066        
067        /**
068         * Updates the path of a folder within the given project.
069         * @param folderId
070         * @param newFileForFolder
071         */
072        void updateFolderPath(long folderId, File newFileForFolder);
073        
074        /**
075         * Builds the xml for the project's metadata schema
076         * @param project The project
077         * @return the xml
078         */
079        String buildMetadataSchemaXML(final Project project);
080    
081         /** Updates any submission packages that a specified research file is in and deletes the file from the system
082         * @param fileToBeReplacedId
083         */
084        void removeResearchFileFromSystem(long fileToBeRemovedId);
085    }
