007    package au.org.intersect.exsite9.service.factory;
008    
009    import javax.persistence.EntityManagerFactory;
010    
011    import org.eclipse.ui.PlatformUI;
012    import org.eclipse.ui.services.AbstractServiceFactory;
013    import org.eclipse.ui.services.IServiceLocator;
014    
015    import au.org.intersect.exsite9.dao.factory.GroupDAOFactory;
016    import au.org.intersect.exsite9.dao.factory.MetadataAssociationDAOFactory;
017    import au.org.intersect.exsite9.dao.factory.ResearchFileDAOFactory;
018    import au.org.intersect.exsite9.service.GroupService;
019    
020    /**
021     * 
022     */
023    public final class GroupServiceFactory extends AbstractServiceFactory
024    {
025        
026        public GroupServiceFactory()
027        {
028        }
029    
030        /**
031         * @{inheritDoc}
032         */
033        @Override
034        @SuppressWarnings("rawtypes")
035        public Object create(final Class serviceInterface, final IServiceLocator parentLocator, final IServiceLocator locator)
036        {
037            final EntityManagerFactory emf = (EntityManagerFactory) PlatformUI.getWorkbench().getService(EntityManagerFactory.class);
038            final GroupDAOFactory groupDAOFactory = new GroupDAOFactory();
039            final MetadataAssociationDAOFactory metadataAssociationDAOFactory = new MetadataAssociationDAOFactory();
040            final ResearchFileDAOFactory researchFileDAOFactory = new ResearchFileDAOFactory();
041    
042            return new GroupService(emf, groupDAOFactory, metadataAssociationDAOFactory, researchFileDAOFactory);
043        }
044    }
