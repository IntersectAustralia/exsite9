007    package au.org.intersect.exsite9.service;
008    
009    import java.io.File;
010    import java.io.IOException;
011    
012    import javax.xml.parsers.ParserConfigurationException;
013    
014    import org.xml.sax.SAXException;
015    
016    import au.org.intersect.exsite9.domain.MetadataCategory;
017    import au.org.intersect.exsite9.domain.Schema;
018    import au.org.intersect.exsite9.exception.InvalidSchemaException;
019    
020    /**
021     * Provides services for manipulating Metadata {@link Schema}s
022     */
023    public interface ISchemaService
024    {
025        /**
026         * Creates a new Local Schema
027         * @param schemaName The name of the schema.
028         * @param schemaDescription The description of the schema.
029         * @param schemaNamespaceURL The namespace URL of the schema.
030         * @return The newly created schema.
031         */
032        public Schema createLocalSchema(final String schemaName, final String schemaDescription, final String schemaNamespaceURL);
033    
034        /**
035         * Persists an Imported Schema
036         * @param schema The schema to persist.
037         */
038        public void createImportedSchema(final Schema schema);
039    
040        /**
041         * Updates the details of the provided schema.
042         * @param schema The schema to update.
043         * @param schemaName The new name of the schema.
044         * @param schemaDescription The new description of the schema.
045         * @param schemaNamespaceURL The new namespace URL of the schema.
046         */
047        public void updateSchema(final Schema schema, final String schemaName, final String schemaDescription, final String schemaNamespaceURL);
048    
049        /**
050         * Removes a schema, its association to a project, and its association to files/groups.
051         * @param schema The schema to delete.
052         */
053        public void removeSchema(final Schema schema);
054    
055        /**
056         * Adds a metadata category to a schema.
057         * @param schema The schema.
058         * @param metadataCategory The metadata category to add.
059         */
060        public void addMetadataCategoryToSchema(final Schema schema, final MetadataCategory metadataCategory);
061    
062        /**
063         * Removes a metadata category from a schema.
064         * @param schema The schema.
065         * @param metadataCategory The metadata category to remove.
066         */
067        public void removeMetadataCategoryFromSchema(final Schema schema, final MetadataCategory metadataCategory);
068    
069        /**
070         * Obtains the default directory on the file system that contains the pre-defined Schemas.
071         * @return The default directory on the file system that contains the pre-defined Schemas. May be {@code null} if there is no default schema directory.
072         */
073        public File getDefaultSchemaDirectory();
074    
075        /**
076         * Obtains the default Schema.
077         * @return The default Schema to be used with new projects.
078         */
079        public File getDefaultSchema();
080    
081        /**
082         * Parses the provides schema XML file to a schema object. <em>Does not persist the Schema to the database.</em>
083         * @param xmlFile The XML File to import.
084         * @return The imported Schema.
085         * @throws SAXException
086         * @throws IOException
087         * @throws ParserConfigurationException
088         * @throws InvalidSchemaException 
089         */
090        public Schema parseSchema(final File xmlFile) throws SAXException, IOException, ParserConfigurationException, InvalidSchemaException;
091    }
