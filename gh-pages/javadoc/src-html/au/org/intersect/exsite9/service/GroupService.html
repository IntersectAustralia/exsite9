084        @Override
085        public String deleteGroupCheck(final Group groupToDelete)
086        {
087            final Group parent = groupToDelete.getParentGroup();
088            final List<Group> siblingGroups = parent.getGroups();
089            final List<ResearchFile> parentResearchFiles = parent.getResearchFiles();
090    
091            final StringBuilder sb = new StringBuilder();
092    
093            // All children of the group to be deleted will become children of the parent. Ensure the deletion will not cause duplicate group names.
094            for (final Group group : groupToDelete.getGroups())
095            {
096                final String groupName = group.getName();
097                for (final Group sibling : siblingGroups)
098                {
099                    if (sibling.getName().equalsIgnoreCase(groupName))
100                    {
101                        sb.append(NEW_LINE + "Duplicate group " + groupName + " under " + parent.getName());
102                    }
103                }
104            }
105    
106            // All research files of the group to be deleted will become children of the parent. Ensure the deletion will not cause duplicate file names.
107            for (final ResearchFile researchFile : groupToDelete.getResearchFiles())
108            {
109                final String researchFileName = researchFile.getFile().getName();
110                for (final ResearchFile parentResearchFile : parentResearchFiles)
111                {
112                    if (parentResearchFile.getFile().getName().equalsIgnoreCase(researchFileName))
113                    {
114                        sb.append(NEW_LINE + "Duplicate file named " + researchFileName + " under group " + parent.getName());
115                    }
116                }
117            }
118    
119            if (sb.length() > 0)
120            {
121                return "Unable to delete group " + groupToDelete.getName() + "." + sb.toString();
122            }
123            return null;
124        }

128         */
129        @Override
130        public void deleteGroup(final Group groupToDelete)
131        {
132            final EntityManager em = entityManagerFactory.createEntityManager();
133            try
134            {
135                final GroupDAO groupDAO = groupDAOFactory.createInstance(em);
136                final MetadataAssociationDAO metadataDAO = metadataAssociationDAOFactory.createInstance(em);
137                final ResearchFileDAO researchFileDAO = researchFileDAOFactory.createInstance(em);
138    
139                final Group parentGroup = groupToDelete.getParentGroup();
140    
141                parentGroup.getGroups().remove(groupToDelete);
142    
143                final List<Group> childGroups = groupToDelete.getGroups();
144                for (final Group childGroup : childGroups)
145                {
146                    childGroup.setParentGroup(parentGroup);
147                    groupDAO.updateGroup(childGroup);
148                }
149                parentGroup.getGroups().addAll(childGroups);
150    
151                final List<ResearchFile> childResearchFiles = groupToDelete.getResearchFiles();
152                for (final ResearchFile childResearchFile : childResearchFiles)
153                {
154                    childResearchFile.setParentGroup(parentGroup);
155                    researchFileDAO.updateResearchFile(childResearchFile);
156                }
157                parentGroup.getResearchFiles().addAll(childResearchFiles);
158    
159                groupToDelete.getGroups().clear();
160                groupToDelete.setParentGroup(null);
161                childResearchFiles.clear();
162    
163                groupDAO.updateGroup(parentGroup);
164                groupDAO.deleteGroup(groupToDelete);
165    
166                // Remove metadata associations
167                for (final MetadataAssociation metadataAssociation : groupToDelete.getMetadataAssociations())
168                {
169                    metadataDAO.removeMetadataAssociation(metadataAssociation);
170                }
171            }
172            finally
173            {
174                em.close();
175            }
176        }

202         * {@inheritDoc}
203         */
204        @Override
205        public String performHierarchyMove(final List<HierarchyMoveDTO> moveList)
206        {
207            final EntityManager em = entityManagerFactory.createEntityManager();
208            final GroupDAO groupDAO = groupDAOFactory.createInstance(em);
209            final ResearchFileDAO researchFileDAO = researchFileDAOFactory.createInstance(em);
210    
211            final StringBuilder sb = new StringBuilder();
212    
213            try
214            {
215                for (final HierarchyMoveDTO moveDTO : moveList)
216                {
217                    final Group oldParent;
218                    if (moveDTO.getOldParent() instanceof Project)
219                    {
220                        oldParent = ((Project)moveDTO.getOldParent()).getRootNode();
221                    }
222                    else
223                    {
224                        oldParent = (Group) moveDTO.getOldParent();
225                    }
226                    
227                    final Group newParent;
228                    if(moveDTO.getNewParent() instanceof Project)
229                    {
230                        newParent = ((Project)moveDTO.getNewParent()).getRootNode();
231                    }
232                    else
233                    {
234                        newParent = (Group) moveDTO.getNewParent();
235                    }
236                    
237                    final Object childObj = moveDTO.getChild();
238                    
239                    boolean movedOK = false;
240                    
241                    if (childObj instanceof Group)
242                    {
243                        final Group group = (Group) childObj;
244                        boolean toContinue = false;
245    
246                        // Ensure there are no groups with the same name as the one being dropped into the destination.
247                        for (final Group siblingGroup : newParent.getGroups())
248                        {
249                            if (siblingGroup.getName().equalsIgnoreCase(group.getName()))
250                            {
251                                sb.append("A group named " + group.getName() + " already exists in the destination group.").append(NEW_LINE);
252                                toContinue = true;
253                                break;
254                            }
255                        }
256    
257                        if (toContinue)
258                        {
259                            continue;
260                        }
261    
262                        movedOK = moveGroupToNewGroup((Group) childObj,oldParent,newParent);
263                        groupDAO.updateGroup(group);
264                    }
265                    else if (childObj instanceof ResearchFile)
266                    {
267                        final ResearchFile researchFile = (ResearchFile) childObj;
268                        boolean toContinue = false;
269    
270                        // Ensure there are no research files with the same name as the one being dropped into the destination.
271                        // It is OK to have research files with the same name in the NewFilesGroup or the ExcludedFilesGroup - since they never make it into SIPs.
272                        if (!(newParent instanceof NewFilesGroup || newParent instanceof ExcludedFilesGroup))
273                        {
274                            for (final ResearchFile siblingResearchFile : newParent.getResearchFiles())
275                            {
276                                if (siblingResearchFile.getFile().getName().equalsIgnoreCase(researchFile.getFile().getName()))
277                                {
278                                    sb.append("A file named " + researchFile.getFile().getName() + " already exists in the destination group.").append(NEW_LINE);
279                                    toContinue = true;
280                                    break;
281                                }
282                            }
283                        }
284    
285                        if (toContinue)
286                        {
287                            continue;
288                        }
289    
290                        movedOK = moveFileToNewGroup(researchFile, oldParent, newParent);
291                        researchFileDAO.updateResearchFile(researchFile);
292                    }
293                    else
294                    {
295                        // TODO: Log this?
296                        continue;
297                    }
298                    
299                    if (movedOK)
300                    {
301                        em.getTransaction().begin();
302                        groupDAO.updateGroup(oldParent);
303                        groupDAO.updateGroup(newParent);
304                        em.getTransaction().commit();
305                    }
306                }
307            }
308            finally
309            {
310                em.close();
311            }
312    
313            if (sb.length() == 0)
314            {
315                return null;
316            }
317            return sb.toString();
318        }

320        @Override
321        public void associateMetadata(final Group group, final MetadataCategory metadataCategory, final MetadataValue metadataValue, final MetadataAttributeValue metadataAttribtueValue)
322        {
323            LOG.info("Assosciating metadata with group. " + group + " " + metadataCategory + " " + metadataValue);
324    
325            final EntityManager em = this.entityManagerFactory.createEntityManager();
326            try
327            {
328                final List<MetadataAssociation> existingAssociations = group.getMetadataAssociations();
329    
330                final MetadataAssociationDAO metadataAssociationDAO = this.metadataAssociationDAOFactory.createInstance(em);
331        
332                boolean addedAssociation = false;
333                for (final MetadataAssociation existingAssociation : existingAssociations)
334                {
335                    if (existingAssociation.getMetadataCategory().equals(metadataCategory))
336                    {
337                        if (existingAssociation.getMetadataValues().contains(metadataValue))
338                        {
339                            // nothing to do!
340                            return;
341                        }
342    
343                        if (metadataCategory.getType() == MetadataCategoryType.FREETEXT)
344                        {
345                            // In a free-text metadata category type, there is only ever one value.
346                            existingAssociation.getMetadataValues().clear();
347                        }
348                        existingAssociation.getMetadataValues().add(metadataValue);
349                        metadataAssociationDAO.updateMetadataAssociation(existingAssociation);
350                        addedAssociation = true;
351                    }
352                }
353    
354                if (!addedAssociation)
355                {
356                    final MetadataAssociation metadataAssociation = new MetadataAssociation(metadataCategory);
357                    metadataAssociation.getMetadataValues().add(metadataValue);
358                    metadataAssociation.setMetadataAttributeValue(metadataAttribtueValue);
359                    metadataAssociationDAO.createMetadataAssociation(metadataAssociation);
360                    group.getMetadataAssociations().add(metadataAssociation);
361                }
362    
363                final GroupDAO groupDAO = this.groupDAOFactory.createInstance(em);
364                groupDAO.updateGroup(group);
365    
366            }
367            finally
368            {
369                em.close();
370            }
371        }

373        @Override
374        public void disassociateMetadata(final Group group, final MetadataCategory metadataCategory, final MetadataValue metadataValue)
375        {
376            LOG.info("Disassosciating metadata from group. " + group + " " + metadataCategory + " " + metadataValue);
377    
378            final List<MetadataAssociation> existingAssociations = group.getMetadataAssociations();
379    
380            for (final MetadataAssociation existingAssociation : existingAssociations)
381            {
382                if (existingAssociation.getMetadataCategory().equals(metadataCategory))
383                {
384                    if (existingAssociation.getMetadataValues().remove(metadataValue))
385                    {
386                        final EntityManager em = this.entityManagerFactory.createEntityManager();
387                        try
388                        {
389                            final MetadataAssociationDAO metadataAssociationDAO = this.metadataAssociationDAOFactory.createInstance(em);
390                            metadataAssociationDAO.updateMetadataAssociation(existingAssociation);
391    
392                            if (existingAssociation.getMetadataValues().isEmpty())
393                            {
394                                existingAssociations.remove(existingAssociation);
395    
396                                final GroupDAO groupDAO = this.groupDAOFactory.createInstance(em);
397                                groupDAO.updateGroup(group);
398    
399                                metadataAssociationDAO.removeMetadataAssociation(existingAssociation);
400                            }
401                            return;
402                        }
403                        finally
404                        {
405                            em.close();
406                        }
407                    }
408                }
409            }
410        }

412        @Override
413        public void disassociateMetadataAttributeValue(final MetadataCategory metadataCategory, final MetadataAttributeValue metadataAttributeValue)
414        {
415            final List<Group> groups = getGroupsWithAssociatedMetadataAttribute(metadataCategory, metadataAttributeValue);
416            for (final Group group : groups)
417            {
418                final List<MetadataAssociation> associations = group.getMetadataAssociations();
419                for (final MetadataAssociation association : associations)
420                {
421                    if (association.getMetadataCategory().equals(metadataCategory) && association.getMetadataAttributeValue().equals(metadataAttributeValue))
422                    {
423                        association.setMetadataAttributeValue(null);
424                        final EntityManager em = this.entityManagerFactory.createEntityManager();
425                        try
426                        {
427                            final MetadataAssociationDAO metadataAssociationDAO = this.metadataAssociationDAOFactory.createInstance(em);
428                            metadataAssociationDAO.updateMetadataAssociation(association);
429                        }
430                        finally
431                        {
432                            em.close();
433                        }
434                    }
435                }
436            }
437        }

439        private boolean moveGroupToNewGroup(Group child, Group oldParent, Group newParent)
440        {
441            if(oldParent.getGroups().remove(child))
442            {
443                if(newParent.getGroups().add(child))
444                {
445                    child.setParentGroup(newParent);
446                    return true;
447                }
448            }
449            return false;
450        }
451        
452        private boolean moveFileToNewGroup(ResearchFile child, Group oldParent, Group newParent)
453        {
454            if(oldParent.getResearchFiles().remove(child))
455            {
456                if(newParent.getResearchFiles().add(child))
457                {
458                    child.setParentGroup(newParent);
459                    return true;
460                }
461            }
462            return false;
463        }

535        @Override
536        public void sortResearchFilesInGroup(final Group group, final ResearchFileSortField sortField, final SortFieldDirection sortDirection)
537        {
538            final EntityManager em = entityManagerFactory.createEntityManager();
539            final GroupDAO groupDAO = new GroupDAO(em);
540            try
541            {
542                configureReserachFileSortParams(groupDAO, group, sortField, sortDirection);
543            }
544            finally
545            {
546                em.close();
547            }
548        }
549    
550        private static void configureReserachFileSortParams(final GroupDAO groupDAO, final Group group, final ResearchFileSortField sortField, final SortFieldDirection sortDirection)
551        {
552            group.setResearchFileSortDirection(sortDirection);
553            group.setResearchFileSortField(sortField);
554            for (final Group childGroup : group.getGroups())
555            {
556                // Recursion
557                configureReserachFileSortParams(groupDAO, childGroup, sortField, sortDirection);
558            }
559            groupDAO.updateGroup(group);
560        }
