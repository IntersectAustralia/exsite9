007    package au.org.intersect.exsite9.service.factory;
008    
009    import java.io.File;
010    
011    import javax.persistence.EntityManagerFactory;
012    
013    import org.eclipse.core.runtime.Platform;
014    import org.eclipse.ui.PlatformUI;
015    import org.eclipse.ui.services.AbstractServiceFactory;
016    import org.eclipse.ui.services.IServiceLocator;
017    
018    import au.org.intersect.exsite9.dao.factory.MetadataAttributeDAOFactory;
019    import au.org.intersect.exsite9.dao.factory.MetadataCategoryDAOFactory;
020    import au.org.intersect.exsite9.dao.factory.SchemaDAOFactory;
021    import au.org.intersect.exsite9.service.SchemaService;
022    
023    /**
024     * Responsible for creating instances of {@link SchemaService}
025     */
026    public final class SchemaServiceFactory extends AbstractServiceFactory
027    {
028    
029        /**
030         * 
031         */
032        public SchemaServiceFactory()
033        {
034        }
035    
036        /**
037         * @{inheritDoc}
038         */
039        @Override
040        @SuppressWarnings("rawtypes")
041        public Object create(final Class serviceInterface, final IServiceLocator parentLocator, final IServiceLocator locator)
042        {
043            // Lookup the default schema directory (which is under the workspace directory).
044            final String workspaceDir = Platform.getInstallLocation().getURL().getPath();
045            final File schemaDir = new File(workspaceDir, "schemas");
046    
047            final File schemaDirToUse;
048            final File defaultSchemaToUse;
049            if (schemaDir.exists() && schemaDir.isDirectory())
050            {
051                schemaDirToUse = schemaDir;
052                final File defaultSchema = new File(schemaDirToUse, "DefaultSchema.xml");
053                if (defaultSchema.exists() && defaultSchema.canRead())
054                {
055                    defaultSchemaToUse = defaultSchema;
056                }
057                else
058                {
059                    defaultSchemaToUse = null;
060                }
061            }
062            else
063            {
064                schemaDirToUse = null;
065                defaultSchemaToUse = null;
066            }
067    
068            final File configurationDir = new File(workspaceDir, "configuration");
069            final File metadataSchemaSchema = new File(configurationDir, "ExSite9MetadataSchema.rng");
070    
071            final EntityManagerFactory emf = (EntityManagerFactory) PlatformUI.getWorkbench().getService(EntityManagerFactory.class);
072            final SchemaDAOFactory schemaDAOFactory = new SchemaDAOFactory();
073            final MetadataCategoryDAOFactory metadataCategoryDAOFactory = new MetadataCategoryDAOFactory();
074            final MetadataAttributeDAOFactory metadataAttributeDAOFactory = new MetadataAttributeDAOFactory();
075            final SchemaService schemaService = new SchemaService(schemaDirToUse, defaultSchemaToUse, metadataSchemaSchema, emf, schemaDAOFactory,
076                    metadataCategoryDAOFactory, metadataAttributeDAOFactory);
077            return schemaService;
078        }
079    }
