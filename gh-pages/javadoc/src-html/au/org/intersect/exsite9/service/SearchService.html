007    package au.org.intersect.exsite9.service;
008    
009    import java.util.Arrays;
010    import java.util.HashSet;
011    import java.util.List;
012    import java.util.Set;
013    
014    import javax.persistence.EntityManager;
015    import javax.persistence.EntityManagerFactory;
016    import javax.persistence.TypedQuery;
017    
018    import au.org.intersect.exsite9.domain.MetadataCategoryType;
019    import au.org.intersect.exsite9.domain.Project;
020    import au.org.intersect.exsite9.domain.ResearchFile;
021    import au.org.intersect.exsite9.domain.ResearchFileSearchDefinition;
022    
023    /**
024     * A service for forming and executing a search on the database
025     */
026    public class SearchService implements ISearchService
027    {
028        private static final String METADATA_CATEGORY_ATTRIBUTE_QUERY_STRING = "SELECT DISTINCT f FROM ResearchFile f JOIN f.metadataAssociations a WHERE f.project = :project AND LOWER(a.metadataAttributeValue.value) LIKE LOWER(:searchTerm)";
029        private static final String METADATA_CATEGORY_VALUE_QUERY_STRING = "SELECT DISTINCT f FROM ResearchFile f JOIN f.metadataAssociations a JOIN a.metadataValues va WHERE f.project = :project AND a.metadataCategory.type = :categoryType AND LOWER(va.value) LIKE LOWER(:searchTerm)";
030        private static final String METADATA_CATEGORY_NAME_QUERY_STRING = "SELECT DISTINCT f FROM ResearchFile f JOIN f.metadataAssociations a WHERE f.project = :project AND LOWER(a.metadataCategory.name) LIKE LOWER(:searchTerm)";
031        private final EntityManagerFactory entityManagerFactory;
032    
033        public SearchService(EntityManagerFactory emf)
034        {
035            this.entityManagerFactory = emf;
036        }
037    
038        @Override
039        public List<ResearchFile> getResearchFilesUsingSearchTerm(String searchTerm,
040                ResearchFileSearchDefinition selectedField, Project currentProject)
041        {
042            final EntityManager em = entityManagerFactory.createEntityManager();
043            try
044            {
045                final TypedQuery<ResearchFile> query;
046                StringBuilder sb = new StringBuilder("%");
047                sb.append(searchTerm);
048                sb.append("%");
049    
050                switch (selectedField)
051                {
052                    case ALL_FIELDS:
053                        //Couldn't get the query below to result all the expected results so had to HACK HACK HACK.
054                        //query = em.createQuery("SELECT DISTINCT f FROM ResearchFile f LEFT JOIN f.metadataAssociations a LEFT JOIN a.metadataValues va WHERE f.project = :project AND (a.metadataCategory.name LIKE :searchTerm OR va.value LIKE :searchTerm OR a.metadataAttributeValue.value LIKE :searchTerm)", ResearchFile.class);
055                        query = em.createQuery(METADATA_CATEGORY_NAME_QUERY_STRING, ResearchFile.class);
056                        query.setParameter("searchTerm", sb.toString());
057                        query.setParameter("project", currentProject);
058                        TypedQuery<ResearchFile> query2 = em.createQuery("SELECT DISTINCT f FROM ResearchFile f JOIN f.metadataAssociations a JOIN a.metadataValues va WHERE f.project = :project AND LOWER(va.value) LIKE LOWER(:searchTerm)", ResearchFile.class);
059                        query2.setParameter("searchTerm", sb.toString());
060                        query2.setParameter("project", currentProject);
061                        TypedQuery<ResearchFile> query3 = em.createQuery(METADATA_CATEGORY_ATTRIBUTE_QUERY_STRING, ResearchFile.class);
062                        query3.setParameter("searchTerm", sb.toString());
063                        query3.setParameter("project", currentProject);
064                        
065                        List<ResearchFile> resultsFromCategoryName = query.getResultList();
066                        List<ResearchFile> resultsFromValue = query2.getResultList();
067                        List<ResearchFile> resultsFromAttributeName = query3.getResultList();
068                        
069                        List<ResearchFile> allResults = resultsFromCategoryName;
070                        allResults.addAll(resultsFromValue);
071                        allResults.addAll(resultsFromAttributeName);
072                        
073                        Set<ResearchFile> resultsWithNoDuplicates = new HashSet<ResearchFile>(allResults);
074                        return Arrays.asList(resultsWithNoDuplicates.toArray(new ResearchFile[resultsWithNoDuplicates.size()]));
075                        
076                    case METADATA_CATEGORY_NAME:
077                        query = em.createQuery(METADATA_CATEGORY_NAME_QUERY_STRING, ResearchFile.class);
078                        break;
079    
080                    case METADATA_VALUE:
081                    case FREETEXT_METADATA_VALUE:
082                        query = em.createQuery(METADATA_CATEGORY_VALUE_QUERY_STRING, ResearchFile.class);
083                        query.setParameter("categoryType", selectedField == ResearchFileSearchDefinition.FREETEXT_METADATA_VALUE ? MetadataCategoryType.FREETEXT : MetadataCategoryType.CONTROLLED_VOCABULARY);
084                        break;
085                        
086                    case METADATA_ATTRIBUTE:
087                        query = em.createQuery(METADATA_CATEGORY_ATTRIBUTE_QUERY_STRING, ResearchFile.class);
088                        break;
089    
090                    default:
091                        throw new IllegalArgumentException("Unkown search field");
092                }
093                query.setParameter("searchTerm", sb.toString());
094                query.setParameter("project", currentProject);
095    
096                return query.getResultList();
097            }
098            finally
099            {
100                em.close();
101            }
102        }
103    
104    }
