007    package au.org.intersect.exsite9;
008    
009    import org.eclipse.swt.graphics.Point;
010    import org.eclipse.ui.application.ActionBarAdvisor;
011    import org.eclipse.ui.application.IActionBarConfigurer;
012    import org.eclipse.ui.application.IWorkbenchWindowConfigurer;
013    import org.eclipse.ui.application.WorkbenchWindowAdvisor;
014    
015    /**
016     * Advisor for the application workbench window
017     */
018    public final class ApplicationWorkbenchWindowAdvisor extends WorkbenchWindowAdvisor
019    {
020    
021        public ApplicationWorkbenchWindowAdvisor(final IWorkbenchWindowConfigurer configurer)
022        {
023            super(configurer);
024        }
025    
026        public ActionBarAdvisor createActionBarAdvisor(final IActionBarConfigurer configurer)
027        {
028            return new ApplicationActionBarAdvisor(configurer);
029        }
030    
031        public void preWindowOpen()
032        {
033            final IWorkbenchWindowConfigurer configurer = getWindowConfigurer();
034    
035            // TODO: we probably want to center it on the screen and have a larger initial size.
036            configurer.setInitialSize(new Point(800, 600));
037            configurer.setShowCoolBar(true);
038            configurer.setShowStatusLine(false);
039            configurer.setShowProgressIndicator(true);
040            configurer.setTitle("ExSite9"); //$NON-NLS-1$
041        }
042    }
