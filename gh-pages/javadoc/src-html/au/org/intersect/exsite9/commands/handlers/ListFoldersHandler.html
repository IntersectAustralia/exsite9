008    package au.org.intersect.exsite9.commands.handlers;
009    
010    import org.apache.log4j.Logger;
011    import org.eclipse.core.commands.ExecutionEvent;
012    import org.eclipse.core.commands.ExecutionException;
013    import org.eclipse.core.commands.IHandler;
014    import org.eclipse.core.commands.IHandlerListener;
015    import org.eclipse.core.commands.NotEnabledException;
016    import org.eclipse.core.commands.NotHandledException;
017    import org.eclipse.core.commands.common.NotDefinedException;
018    import org.eclipse.jface.dialogs.MessageDialog;
019    import org.eclipse.jface.wizard.WizardDialog;
020    import org.eclipse.swt.widgets.Shell;
021    import org.eclipse.ui.IWorkbenchWindow;
022    import org.eclipse.ui.PlatformUI;
023    import org.eclipse.ui.handlers.HandlerUtil;
024    import org.eclipse.ui.handlers.IHandlerService;
025    
026    import au.org.intersect.exsite9.domain.Project;
027    import au.org.intersect.exsite9.jobs.SemaphoreManager;
028    import au.org.intersect.exsite9.service.IProjectManager;
029    import au.org.intersect.exsite9.wizard.listfolders.ListFoldersWizard;
030    
031    /**
032     * Command handler to list the folders that a project is currently watching, via the plugin.xml
033     */
034    public class ListFoldersHandler implements IHandler
035    {
036        private static final Logger LOG = Logger.getLogger(ListFoldersHandler.class);
037        
038        @Override
039        public void addHandlerListener(IHandlerListener handlerListener)
040        {
041        }
042    
043        @Override
044        public void dispose()
045        {
046        }
047    
048        @Override
049        public Object execute(ExecutionEvent event) throws ExecutionException
050        {
051            final IWorkbenchWindow activeWorkbenchWindow = HandlerUtil.getActiveWorkbenchWindow(event);
052            final Shell shell = activeWorkbenchWindow.getShell();
053    
054            if( ! SemaphoreManager.getBackgroundJobSemaphore().tryAcquire() )
055            {
056                MessageDialog.openWarning(shell, "Editing Watched Folders", 
057                                                 "You cannot edit the watched folder list whilst ExSite9 is looking for new files.");
058                return null;
059            }
060            
061            try
062            {
063                final ListFoldersWizard wizard = new ListFoldersWizard();
064                final WizardDialog wizardDialog = new WizardDialog(shell, wizard);
065                wizardDialog.open();
066            }
067            finally
068            {
069                SemaphoreManager.getBackgroundJobSemaphore().release();
070            }
071            
072            final IHandlerService handlerService = (IHandlerService) activeWorkbenchWindow.getService(IHandlerService.class);
073            
074            try
075            {
076                handlerService.executeCommand("au.org.intersect.exsite9.commands.ReloadProjectCommand", null);
077            }
078            catch (NotDefinedException e)
079            {
080                LOG.error("Cannot execute reload project command", e);
081            }
082            catch (NotEnabledException e)
083            {
084                LOG.error("Cannot execute reload project command", e);
085            }
086            catch (NotHandledException e)
087            {
088                LOG.error("Cannot execute reload project command", e);
089            }
090            
091            return null;
092        }
093    
094        @Override
095        public boolean isEnabled()
096        {
097            final IProjectManager projectManager = (IProjectManager) PlatformUI.getWorkbench().getService(IProjectManager.class);
098            final Project currentproject = projectManager.getCurrentProject();
099            
100            return (currentproject == null) ? false : true;
101        }
102    
103        @Override
104        public boolean isHandled()
105        {
106            return true;
107        }
108    
109        @Override
110        public void removeHandlerListener(IHandlerListener handlerListener)
111        {
112        }
113    
114    }
