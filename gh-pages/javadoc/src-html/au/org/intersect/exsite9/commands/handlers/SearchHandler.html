007    package au.org.intersect.exsite9.commands.handlers;
008    
009    import java.util.List;
010    
011    import org.eclipse.core.commands.ExecutionEvent;
012    import org.eclipse.core.commands.ExecutionException;
013    import org.eclipse.core.commands.IHandler;
014    import org.eclipse.core.commands.IHandlerListener;
015    import org.eclipse.jface.dialogs.MessageDialog;
016    import org.eclipse.jface.viewers.StructuredSelection;
017    import org.eclipse.jface.wizard.WizardDialog;
018    import org.eclipse.swt.widgets.Shell;
019    import org.eclipse.ui.PlatformUI;
020    import org.eclipse.ui.handlers.HandlerUtil;
021    
022    import au.org.intersect.exsite9.domain.Project;
023    import au.org.intersect.exsite9.domain.ResearchFile;
024    import au.org.intersect.exsite9.service.IProjectManager;
025    import au.org.intersect.exsite9.view.ProjectExplorerView;
026    import au.org.intersect.exsite9.view.ViewUtils;
027    import au.org.intersect.exsite9.wizard.search.SearchWizard;
028    
029    /**
030     * Command handler to initiate a search of research file metadata within the current project, via the plugin.xml
031     */
032    public class SearchHandler implements IHandler
033    {
034    
035        @Override
036        public void addHandlerListener(IHandlerListener arg0)
037        {
038            
039        }
040    
041        @Override
042        public void dispose()
043        {
044            
045        }
046    
047        @Override
048        public Object execute(ExecutionEvent event) throws ExecutionException
049        {
050            final IProjectManager projectManager = (IProjectManager) PlatformUI.getWorkbench().getService(IProjectManager.class);
051            final Project selectedProject = projectManager.getCurrentProject();
052            if (selectedProject == null)
053            {
054                throw new IllegalStateException("Trying to search on a null project");
055            }
056    
057            final Shell shell = HandlerUtil.getActiveWorkbenchWindow(event).getShell();
058            final SearchWizard wizard = new SearchWizard(selectedProject);
059            final WizardDialog wizardDialog = new WizardDialog(shell, wizard);
060            wizardDialog.open();
061    
062            List<ResearchFile> searchResults = wizard.getSearchResults();
063            
064            final ProjectExplorerView projectExplorerView = (ProjectExplorerView) ViewUtils.getViewByID(PlatformUI
065                    .getWorkbench().getActiveWorkbenchWindow(), ProjectExplorerView.ID);
066            
067            projectExplorerView.setSelection(new StructuredSelection(searchResults));
068            projectExplorerView.refresh();
069            
070            if (searchResults.isEmpty())
071            {
072                MessageDialog.openInformation(shell, "Search Results", "Your Search Returned 0 Results.");
073            }
074            
075            return null;
076        }
077    
078        @Override
079        public boolean isEnabled()
080        {
081            return true;
082        }
083    
084        @Override
085        public boolean isHandled()
086        {
087            return true;
088        }
089    
090        @Override
091        public void removeHandlerListener(IHandlerListener arg0)
092        {
093            
094        }
095    
096    }
