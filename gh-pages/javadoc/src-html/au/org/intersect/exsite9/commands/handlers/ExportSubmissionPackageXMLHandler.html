040    /**
041     * Command handler to export a submission package xml from a project, via the plugin.xml
042     */
043    public class ExportSubmissionPackageXMLHandler implements IHandler
044    {
045        private static final String NEW_LINE = System.getProperty("line.separator");
046    
047        @Override
048        public void addHandlerListener(IHandlerListener handlerListener)
049        {
050        }
051    
052        @Override
053        public void dispose()
054        {
055        }
056    
057        @Override
058        public Object execute(ExecutionEvent event) throws ExecutionException
059        {
060            final Shell shell = HandlerUtil.getActiveWorkbenchWindow(event).getShell();
061    
062            final IProjectManager projectManager = (IProjectManager) PlatformUI.getWorkbench().getService(IProjectManager.class);
063            final Project currentproject = projectManager.getCurrentProject();
064            
065            final IStructuredSelection selection = (IStructuredSelection) HandlerUtil.getActiveWorkbenchWindow(event).getActivePage().getSelection();
066            final Object selectedObject = selection.getFirstElement();
067    
068            if (!(selectedObject instanceof SubmissionPackage))
069            {
070                throw new IllegalArgumentException("Trying to export XML from an object that is not a SubmissionPackage");
071            }
072    
073            final SubmissionPackage submissionPackage = (SubmissionPackage) selectedObject;
074    
075            // Before we generate the output XML, we should ensure that all files selected for this SubmissionPackage are still where we think they are.
076            final Collection<ResearchFile> missingFiles = Collections2.filter(submissionPackage.getResearchFiles(), new Predicate<ResearchFile>()
077            {
078                @Override
079                public boolean apply(final ResearchFile input)
080                {
081                    return !input.getFile().exists();
082                }
083            });
084    
085            if (!missingFiles.isEmpty())
086            {
087                final StringBuilder sb = new StringBuilder();
088                for (final Iterator<ResearchFile> iter = missingFiles.iterator(); iter.hasNext(); )
089                {
090                    sb.append(iter.next().getFile().getAbsolutePath());
091                    if (iter.hasNext())
092                    {
093                        sb.append(NEW_LINE);
094                    }
095                }
096                MessageDialog.openWarning(shell, "Files missing", "The following files are missing. They will not be included in the submission package XML."
097                    + NEW_LINE + sb.toString());
098            }
099            
100            final FileDialog fileDialog = new FileDialog(shell, SWT.SAVE);
101            fileDialog.setOverwrite(true);
102            fileDialog.setFileName(submissionPackage.getName() + ".xml");
103            final String filePath = fileDialog.open();
104    
105            if (filePath == null)
106            {
107                return null;
108            }
109    
110            
111            final ISubmissionPackageService submissionPackageService = (ISubmissionPackageService) PlatformUI.getWorkbench().getService(ISubmissionPackageService.class);
112            
113            final String xml = submissionPackageService.buildXMLForSubmissionPackage(currentproject, submissionPackage);
114    
115            final File fileToWrite = new File(filePath);
116            final OutputSupplier<FileOutputStream> outputSupplier = Files.newOutputStreamSupplier(fileToWrite, false);
117            try
118            {
119                ByteStreams.write(xml.getBytes(Charsets.UTF_8), outputSupplier);
120            }
121            catch (final IOException e)
122            {
123                MessageDialog.openError(shell, "Error", "Could not save to file " + fileToWrite.getAbsolutePath());
124            }
125    
126            return null;
127        }
128    
129        @Override
130        public boolean isEnabled()
131        {
132            return true;
133        }
134    
135        @Override
136        public boolean isHandled()
137        {
138            return true;
139        }
140    
141        @Override
142        public void removeHandlerListener(IHandlerListener handlerListener)
143        {
144        }
145
