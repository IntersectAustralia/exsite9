007    package au.org.intersect.exsite9.commands.handlers;
008    
009    import org.eclipse.core.commands.ExecutionEvent;
010    import org.eclipse.core.commands.ExecutionException;
011    import org.eclipse.core.commands.IHandler;
012    import org.eclipse.core.commands.IHandlerListener;
013    import org.eclipse.jface.viewers.IStructuredSelection;
014    import org.eclipse.jface.wizard.WizardDialog;
015    import org.eclipse.swt.widgets.Shell;
016    import org.eclipse.ui.PlatformUI;
017    import org.eclipse.ui.handlers.HandlerUtil;
018    
019    import au.org.intersect.exsite9.domain.Project;
020    import au.org.intersect.exsite9.domain.SubmissionPackage;
021    import au.org.intersect.exsite9.service.IProjectManager;
022    import au.org.intersect.exsite9.wizard.createsubmissionpackage.CreateSubmissionPackageWizard;
023    
024    /**
025     * Handler to Edit a submission package.
026     */
027    public final class EditSubmissionPackageHandler implements IHandler
028    {
029    
030        /**
031         * @{inheritDoc}
032         */
033        @Override
034        public void dispose()
035        {
036    
037        }
038    
039        /**
040         * @{inheritDoc}
041         */
042        @Override
043        public Object execute(final ExecutionEvent event) throws ExecutionException
044        {
045            final IStructuredSelection selection = (IStructuredSelection) HandlerUtil.getActiveWorkbenchWindow(event).getActivePage().getSelection();
046            final Object selectedObject = selection.getFirstElement();
047    
048            if (!(selectedObject instanceof SubmissionPackage))
049            {
050                throw new IllegalArgumentException("Trying to edit an object that is not a SubmissionPackage");
051            }
052    
053            final SubmissionPackage toEdit = (SubmissionPackage) selectedObject;
054    
055            final IProjectManager projectManager = (IProjectManager) PlatformUI.getWorkbench().getService(IProjectManager.class);
056            final Project project = projectManager.getCurrentProject();
057    
058            final Shell shell = HandlerUtil.getActiveWorkbenchWindow(event).getShell();
059            final CreateSubmissionPackageWizard wizard = new CreateSubmissionPackageWizard(toEdit, project.getSubmissionPackages(), "Edit Submission Package");
060    
061            final WizardDialog wizardDialog = new WizardDialog(shell, wizard);
062            wizardDialog.open();
063            
064            return wizard.getSubmissionPackage();
065        }
066    
067        /**
068         * @{inheritDoc}
069         */
070        @Override
071        public boolean isEnabled()
072        {
073            return true;
074        }
075    
076        /**
077         * @{inheritDoc}
078         */
079        @Override
080        public boolean isHandled()
081        {
082            return true;
083        }
084    
085        /**
086         * @{inheritDoc}
087         */
088        @Override
089        public void addHandlerListener(final IHandlerListener handlerListener)
090        {
091        }
092    
093        /**
094         * @{inheritDoc}
095         */
096        @Override
097        public void removeHandlerListener(final IHandlerListener handlerListener)
098        {
099        }
100    
101    }
