007    package au.org.intersect.exsite9.commands.handlers;
008    
009    import org.eclipse.core.commands.ExecutionEvent;
010    import org.eclipse.core.commands.ExecutionException;
011    import org.eclipse.core.commands.IHandler;
012    import org.eclipse.core.commands.IHandlerListener;
013    import org.eclipse.jface.dialogs.MessageDialog;
014    import org.eclipse.jface.viewers.IStructuredSelection;
015    import org.eclipse.swt.widgets.Shell;
016    import org.eclipse.ui.PlatformUI;
017    import org.eclipse.ui.handlers.HandlerUtil;
018    
019    import au.org.intersect.exsite9.domain.Group;
020    import au.org.intersect.exsite9.domain.NewFilesGroup;
021    import au.org.intersect.exsite9.service.IGroupService;
022    
023    /**
024     * Executed to delete a group.
025     */
026    public final class DeleteGroupHandler implements IHandler
027    {
028    
029        /**
030         * @{inheritDoc}
031         */
032        @Override
033        public void addHandlerListener(final IHandlerListener handlerListener)
034        {
035        }
036    
037        /**
038         * @{inheritDoc}
039         */
040        @Override
041        public void dispose()
042        {
043        }
044    
045        /**
046         * @{inheritDoc}
047         */
048        @Override
049        public Object execute(final ExecutionEvent event) throws ExecutionException
050        {
051            final IStructuredSelection selection = (IStructuredSelection) HandlerUtil.getActiveWorkbenchWindow(event).getActivePage().getSelection();
052            final Object selectedObject = selection.getFirstElement();
053    
054            if (!(selectedObject instanceof Group) || (selectedObject instanceof NewFilesGroup))
055            {
056                throw new RuntimeException("Trying to delete a non-group node from the DeleteGroupHandler");
057            }
058    
059            final Group groupToDelete = (Group) selectedObject;
060    
061            final Shell shell = HandlerUtil.getActiveWorkbenchWindow(event).getShell();
062            final String dialogText = "Are you sure you want to delete group '" + groupToDelete.getName() +
063                    "'? All child groups and files will be moved to the parent.";
064    
065            final MessageDialog messageDialog = new MessageDialog(shell, "Are you sure?", null, dialogText, MessageDialog.CONFIRM, new String[]{"OK", "Cancel"}, 1);
066            final boolean confirm = messageDialog.open() == 0;
067    
068            if (confirm)
069            {
070                final IGroupService groupService = (IGroupService) PlatformUI.getWorkbench().getService(IGroupService.class);
071    
072                final String error = groupService.deleteGroupCheck(groupToDelete);
073                if (error != null)
074                {
075                    MessageDialog.openError(shell, "Cannot delete group " + groupToDelete.getName(), error);
076                }
077                else
078                {
079                    groupService.deleteGroup(groupToDelete);
080                }
081            }
082    
083            return null;
084        }
085    
086        /**
087         * @{inheritDoc}
088         */
089        @Override
090        public boolean isEnabled()
091        {
092            return true;
093        }
094    
095        /**
096         * @{inheritDoc}
097         */
098        @Override
099        public boolean isHandled()
100        {
101            return true;
102        }
103    
104        /**
105         * @{inheritDoc}
106         */
107        @Override
108        public void removeHandlerListener(final IHandlerListener handlerListener)
109        {
110    
111        }
112    
113    }
