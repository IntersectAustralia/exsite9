007    package au.org.intersect.exsite9.commands.handlers;
008    
009    import java.util.HashSet;
010    import java.util.Set;
011    
012    import org.eclipse.ui.IWorkbenchWindow;
013    import org.eclipse.ui.activities.IActivityManager;
014    import org.eclipse.ui.activities.IWorkbenchActivitySupport;
015    
016    /**
017     * Utilities used by the Handlers
018     */
019    public final class HandlerUtils
020    {
021        public static void activateShowProjectActivity(final IWorkbenchWindow activeWorkbenchWindow)
022        {
023            final IWorkbenchActivitySupport activitySupport = activeWorkbenchWindow.getWorkbench().getActivitySupport();
024            final IActivityManager activityManager = activitySupport.getActivityManager();
025            final Set<String> enabledActivities = new HashSet<String>();
026            final String id = "au.org.intersect.exsite9.activity.showProject";
027            if (activityManager.getActivity(id).isDefined())
028            {
029                enabledActivities.add(id);
030            }
031            activitySupport.setEnabledActivityIds(enabledActivities);
032        }
033    }
