009    package au.org.intersect.exsite9.commands.handlers;
010    
011    import org.eclipse.core.commands.ExecutionEvent;
012    import org.eclipse.core.commands.ExecutionException;
013    import org.eclipse.core.commands.IHandler;
014    import org.eclipse.core.commands.IHandlerListener;
015    import org.eclipse.jface.wizard.WizardDialog;
016    import org.eclipse.swt.widgets.Shell;
017    import org.eclipse.ui.PlatformUI;
018    import org.eclipse.ui.handlers.HandlerUtil;
019    
020    import au.org.intersect.exsite9.domain.Project;
021    import au.org.intersect.exsite9.service.IProjectManager;
022    import au.org.intersect.exsite9.wizard.editproject.EditProjectWizard;
023    
024    /**
025     * Command handler to edit a project, via the plugin.xml
026     */
027    public class EditProjectHandler implements IHandler
028    {
029    
030        @Override
031        public void addHandlerListener(final IHandlerListener handlerListener)
032        {
033        }
034    
035        @Override
036        public void dispose()
037        {
038        }
039    
040        @Override
041        public Object execute(final ExecutionEvent event) throws ExecutionException
042        {
043            final IProjectManager projectManager = (IProjectManager) PlatformUI.getWorkbench().getService(IProjectManager.class);
044            final Project selectedProject = projectManager.getCurrentProject();
045            if (selectedProject == null)
046            {
047                throw new IllegalStateException("Trying to edit a null project");
048            }
049    
050            final Shell shell = HandlerUtil.getActiveWorkbenchWindow(event).getShell();
051            final EditProjectWizard wizard = new EditProjectWizard(selectedProject);
052            final WizardDialog wizardDialog = new WizardDialog(shell, wizard);
053            wizardDialog.open();
054    
055            return wizard.updateProject();
056        }
057    
058        @Override
059        public boolean isEnabled()
060        {
061            return true;
062        }
063    
064        @Override
065        public boolean isHandled()
066        {
067            return true;
068        }
069    
070        @Override
071        public void removeHandlerListener(final IHandlerListener handlerListener)
072        {
073        }
074    
075    }
