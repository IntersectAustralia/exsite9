009    package au.org.intersect.exsite9.commands.handlers;
010    
011    import org.eclipse.core.commands.ExecutionEvent;
012    import org.eclipse.core.commands.ExecutionException;
013    import org.eclipse.core.commands.IHandler;
014    import org.eclipse.core.commands.IHandlerListener;
015    import org.eclipse.jface.wizard.WizardDialog;
016    import org.eclipse.swt.widgets.Shell;
017    import org.eclipse.ui.PlatformUI;
018    import org.eclipse.ui.handlers.HandlerUtil;
019    
020    import au.org.intersect.exsite9.domain.Project;
021    import au.org.intersect.exsite9.service.IProjectManager;
022    import au.org.intersect.exsite9.wizard.editproject.EditProjectWizard;
023    
024    /**
025     *
026     * @version $Rev: 29 $
027     */
028    public class EditProjectHandler implements IHandler
029    {
030    
031        @Override
032        public void addHandlerListener(final IHandlerListener handlerListener)
033        {
034            // TODO Auto-generated method stub
035            
036        }
037    
038        @Override
039        public void dispose()
040        {
041            // TODO Auto-generated method stub
042            
043        }
044    
045        @Override
046        public Object execute(final ExecutionEvent event) throws ExecutionException
047        {
048            final IProjectManager projectManager = (IProjectManager) PlatformUI.getWorkbench().getService(IProjectManager.class);
049            final Project selectedProject = projectManager.getCurrentProject();
050            if (selectedProject == null)
051            {
052                throw new IllegalStateException("Trying to edit a null project");
053            }
054    
055            final Shell shell = HandlerUtil.getActiveWorkbenchWindow(event).getShell();
056            final EditProjectWizard wizard = new EditProjectWizard(selectedProject);
057            final WizardDialog wizardDialog = new WizardDialog(shell, wizard);
058            wizardDialog.open();
059    
060            return wizard.updateProject();
061        }
062    
063        @Override
064        public boolean isEnabled()
065        {
066            final IProjectManager projectManager = (IProjectManager) PlatformUI.getWorkbench().getService(IProjectManager.class);
067            final Project currentProject = projectManager.getCurrentProject();
068           
069            return (currentProject == null) ? false : true;
070        }
071    
072        @Override
073        public boolean isHandled()
074        {
075            // TODO Auto-generated method stub
076            return true;
077        }
078    
079        @Override
080        public void removeHandlerListener(final IHandlerListener handlerListener)
081        {
082            // TODO Auto-generated method stub
083            
084        }
085    
086    }
