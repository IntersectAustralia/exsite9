007    package au.org.intersect.exsite9.wizard.createsubmissionpackage;

009    import java.util.ArrayList;
010    import java.util.Arrays;
011    import java.util.List;
012    
013    import org.eclipse.jface.viewers.CheckStateChangedEvent;
014    import org.eclipse.jface.viewers.ColumnViewerToolTipSupport;
015    import org.eclipse.jface.viewers.ICheckStateListener;
016    import org.eclipse.jface.wizard.WizardPage;

018    import org.eclipse.swt.layout.GridData;
019    import org.eclipse.swt.layout.GridLayout;
020    import org.eclipse.swt.widgets.Composite;
021    import org.eclipse.swt.widgets.Label;

023    import org.eclipse.ui.dialogs.ContainerCheckedTreeViewer;
024    
025    import au.org.intersect.exsite9.domain.Project;
026    import au.org.intersect.exsite9.domain.ResearchFile;
027    import au.org.intersect.exsite9.domain.SubmissionPackage;
028    import au.org.intersect.exsite9.service.IProjectManager;
029    import au.org.intersect.exsite9.view.provider.ProjectExplorerViewContentProvider;

031    import au.org.intersect.exsite9.view.provider.ProjectViewInputWrapper;
032    
033    /**
034     * The second page of the Create Submission Package Wizard.
035     * Allows the user to configure the files/groups that are to be contained in the submission package.
036     */
037    public final class CreateSubmissionPackageWizardPage2 extends WizardPage implements ICheckStateListener
038    {
039        private ContainerCheckedTreeViewer treeViewer;
040        private Label itemsSelectedLabel;
041        private final SubmissionPackage currentSubmissionPackage;
042    
043        /**
044         * Constructor
045         */
046        public CreateSubmissionPackageWizardPage2(final SubmissionPackage submissionPackage, final String pageTitle)
047        {
048            super(pageTitle);
049            setTitle(pageTitle);
050            setDescription("Choose the items from your project hierarchy to include in the submission package.");
051            this.currentSubmissionPackage = submissionPackage;
052        }
053    
054        /**
055         * @{inheritDoc}
056         */
057        @Override
058        public void createControl(final Composite parent)
059        {
060            final Composite composite = new Composite(parent, SWT.NONE);
061            final GridLayout layout = new GridLayout();
062            layout.numColumns = 1;
063            composite.setLayout(layout);
064    
065            this.treeViewer = new ContainerCheckedTreeViewer(composite, SWT.H_SCROLL | SWT.V_SCROLL | SWT.BORDER);
066            this.treeViewer.setContentProvider(new ProjectExplorerViewContentProvider(false));
067            this.treeViewer.setLabelProvider(new ProjectExplorerViewLabelProvider());
068            this.treeViewer.addCheckStateListener(this);
069            ColumnViewerToolTipSupport.enableFor(this.treeViewer);
070    
071            final IProjectManager projectManager = (IProjectManager) PlatformUI.getWorkbench().getService(IProjectManager.class);
072            final Project currentProject = projectManager.getCurrentProject();
073            final ProjectViewInputWrapper wrapper = new ProjectViewInputWrapper(currentProject);
074    
075            this.treeViewer.setInput(wrapper);
076            this.treeViewer.expandAll();
077    
078            final GridData treeViewerGridData = new GridData(SWT.FILL, SWT.FILL, true, true);
079            this.treeViewer.getControl().setLayoutData(treeViewerGridData);
080    
081            if (this.currentSubmissionPackage != null)
082            {
083                for (final ResearchFile researchFile : this.currentSubmissionPackage.getResearchFiles())
084                {
085                    this.treeViewer.setChecked(researchFile, true);
086                }
087            }
088    
089            this.itemsSelectedLabel = new Label(composite, SWT.NONE);
090            setItemsSelectedLabel();
091    
092            setControl(composite);
093            setPageComplete(true);
094        }
095    
096        private void setItemsSelectedLabel()
097        {
098            this.itemsSelectedLabel.setText("Files selected: " + getCheckedResearchFiles().size());
099            this.itemsSelectedLabel.pack();
100        }
101    
102        @Override
103        public void checkStateChanged(final CheckStateChangedEvent event)
104        {
105            setItemsSelectedLabel();
106        }
107    
108        public List<ResearchFile> getCheckedResearchFiles()
109        {
110            final List<Object> checkedItems = Arrays.asList(this.treeViewer.getCheckedElements());
111            final List<ResearchFile> toReturn = new ArrayList<ResearchFile>();
112    
113            for (final Object item : checkedItems)
114            {
115                if (item instanceof ResearchFile)
116                {
117                    toReturn.add((ResearchFile)item);
118                }
119            }
120            return toReturn;
121        }
122    }
