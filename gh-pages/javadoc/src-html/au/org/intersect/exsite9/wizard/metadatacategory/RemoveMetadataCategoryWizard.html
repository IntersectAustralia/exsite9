007    package au.org.intersect.exsite9.wizard.metadatacategory;
008    
009    import java.util.List;
010    
011    import org.eclipse.jface.wizard.Wizard;
012    import org.eclipse.ui.PlatformUI;
013    
014    import au.org.intersect.exsite9.domain.Group;
015    import au.org.intersect.exsite9.domain.MetadataCategory;
016    import au.org.intersect.exsite9.domain.MetadataValue;
017    import au.org.intersect.exsite9.domain.ResearchFile;
018    import au.org.intersect.exsite9.domain.Schema;
019    import au.org.intersect.exsite9.service.IGroupService;
020    import au.org.intersect.exsite9.service.IMetadataCategoryService;
021    import au.org.intersect.exsite9.service.IResearchFileService;
022    import au.org.intersect.exsite9.service.ISchemaService;
023    
024    /**
025     * Wizard use to remove metadata categories.
026     */
027    public final class RemoveMetadataCategoryWizard extends Wizard
028    {
029        private final Schema schema;
030    
031        private final ListMetadataCategoriesWizardPage listCategoriesPage;
032    
033        public RemoveMetadataCategoryWizard(final Schema schema)
034        {
035            super();
036            setNeedsProgressMonitor(true);
037            this.schema = schema;
038            this.listCategoriesPage = new ListMetadataCategoriesWizardPage("Remove Metadata Category", "Select metadata categories to be removed.", this.schema, true);
039        }
040    
041        @Override
042        public void addPages()
043        {
044            addPage(this.listCategoriesPage);
045        }
046    
047        /**
048         * @{inheritDoc}
049         */
050        @Override
051        public boolean performFinish()
052        {
053            final IMetadataCategoryService metadataCategoryService = (IMetadataCategoryService) PlatformUI.getWorkbench().getService(IMetadataCategoryService.class);
054            final IGroupService groupService = (IGroupService) PlatformUI.getWorkbench().getService(IGroupService.class);
055            final IResearchFileService fileService = (IResearchFileService) PlatformUI.getWorkbench().getService(IResearchFileService.class);
056            final ISchemaService schemaService = (ISchemaService) PlatformUI.getWorkbench().getService(ISchemaService.class);
057    
058            final List<MetadataCategory> categoriesToDelete = listCategoriesPage.getMetadataCategoriesToDelete();
059            for (final MetadataCategory mdc : categoriesToDelete)
060            {
061                for (final MetadataValue mdv : mdc.getValues())
062                {
063                    final List<ResearchFile> researchFiles = fileService.getResearchFilesWithAssociatedMetadata(mdc, mdv);
064                    for (final ResearchFile researchFile : researchFiles)
065                    {
066                        fileService.disassociateMetadata(researchFile, mdc, mdv);
067                    }
068                    final List<Group> groups = groupService.getGroupsWithAssociatedMetadata(mdc, mdv);
069                    for (final Group group : groups)
070                    {
071                        groupService.disassociateMetadata(group, mdc, mdv);
072                    }
073                }
074                schemaService.removeMetadataCategoryFromSchema(schema, mdc);
075                metadataCategoryService.deleteMetadataCategory(mdc);
076            }
077            return true;
078        }
079    
080    }
