007    package au.org.intersect.exsite9.wizard.newproject;
008    
009    import org.eclipse.jface.wizard.Wizard;
010    import org.eclipse.ui.PlatformUI;
011    
012    import au.org.intersect.exsite9.domain.FieldOfResearch;
013    import au.org.intersect.exsite9.domain.Project;
014    import au.org.intersect.exsite9.domain.Schema;
015    import au.org.intersect.exsite9.dto.ProjectFieldsDTO;
016    import au.org.intersect.exsite9.service.IProjectService;
017    import au.org.intersect.exsite9.service.ISchemaService;
018    
019    /**
020     * The Wizard used to create a new project
021     */
022    public final class NewProjectWizard extends Wizard
023    {
024        private final EditOrCreateProjectWizardPage1 page1;
025        private final EditOrCreateProjectWizardPage2 page2;
026        private final EditOrCreateProjectWizardPage3 page3;
027    
028        private Project newProject;
029    
030        /**
031         * Constructor
032         */
033        public NewProjectWizard()
034        {
035            super();
036            setNeedsProgressMonitor(true);
037            setWindowTitle("New Project");
038            final ProjectFieldsDTO projectFieldsDTO = new ProjectFieldsDTO("", "", "", "", "", "", "", "", "", "", "", "", "", "", "", "", "", "", "", null, "", "", "", "", "");
039            page1 = new EditOrCreateProjectWizardPage1("New Project", "Please enter the details of your new project.", projectFieldsDTO);
040            page2 = new EditOrCreateProjectWizardPage2("New Project", "Please enter the details of your new project.", projectFieldsDTO);
041            page3 = new EditOrCreateProjectWizardPage3("New Project", "Please enter the schema details to be used in your new project.", null);
042        }
043    
044        /**
045         * @{inheritDoc
046         */
047        @Override
048        public void addPages()
049        {
050            addPage(this.page1);
051            addPage(this.page2);
052            addPage(this.page3);
053        }
054    
055        /**
056         * @{inheritDoc
057         */
058        @Override
059        public boolean performFinish()
060        {
061            final IProjectService projectService = (IProjectService) PlatformUI.getWorkbench().getService(IProjectService.class);
062            final ISchemaService schemaService = (ISchemaService) PlatformUI.getWorkbench().getService(ISchemaService.class);
063    
064            final Schema schema;
065            if (page3.getUseLocalSchema())
066            {
067                final String schemaName = page3.getLocalSchemaName();
068                final String schemaDescription = page3.getLocalSchemaDescription();
069                final String schemaNamespaceURL = page3.getLocalSchemaNamespaceURL();
070                schema = schemaService.createLocalSchema(schemaName, schemaDescription, schemaNamespaceURL);
071            }
072            else
073            {
074                schema = page3.getImportedSchema();
075                schemaService.createImportedSchema(schema);
076            }
077    
078            final ProjectFieldsDTO projectFieldsDTO = createProjectFieldsDTO(page1, page2);
079            this.newProject = projectService.createProject(projectFieldsDTO, schema);
080            return this.newProject != null;
081        }
082    
083        public Project getNewProject()
084        {
085            return this.newProject;
086        }
087    
088        public static ProjectFieldsDTO createProjectFieldsDTO(final EditOrCreateProjectWizardPage1 page1, final EditOrCreateProjectWizardPage2 page2)
089        {
090            final String name = page1.getProjectName();
091            final String owner = page1.getOwner();
092            final String institution = page1.getInstitution();
093            final String email = page1.getEmail();
094            final String description = page1.getProjectDescription();
095            final String collectionType = page1.getCollectionType();
096            final String rightsStatement = page1.getRightsStatement();
097            final String accessRights = page1.getAccessRights();
098            final String license = page1.getLicense();
099            final String identifier = page1.getIdentifier();
100            final String subject = page1.getSubject();
101            final String electronicLocation = page2.getElectronicLocation();
102            final String physicalLocation = page2.getPhysicalLocation();
103            final String placeOrRegionName = page2.getPlaceOrRegionName();
104            final String geographicalCoverage = page2.getGeographicalCoverage();
105            final String datesOfCapture = page2.getDatesOfCapture();
106            final String citationInformation = page2.getCitationInformation();
107            final String countries = page2.getCountries();
108            final String languages = page2.getLanguages();
109            final FieldOfResearch fieldOfResearch = page2.getFieldOfResearch();
110            final String fundingBody = page2.getFundingBody();
111            final String grantID = page2.getGrantID();
112            final String relatedParty = page2.getRelatedParty();
113            final String relatedGrant = page2.getRelatedActivity();
114            final String relatedInformation = page2.getRelatedInformation();
115    
116            final ProjectFieldsDTO projectFieldsDTO = new ProjectFieldsDTO(name, owner, institution, email, description, collectionType, rightsStatement,
117                accessRights, license, identifier, subject, electronicLocation, physicalLocation, placeOrRegionName, geographicalCoverage, datesOfCapture,
118                citationInformation, countries, languages, fieldOfResearch, fundingBody, grantID, relatedParty, relatedGrant, relatedInformation);
119            return projectFieldsDTO;
120        }
121    }
