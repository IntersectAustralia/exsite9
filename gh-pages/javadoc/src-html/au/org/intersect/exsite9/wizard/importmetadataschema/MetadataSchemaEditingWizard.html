009    import java.util.List;

011    import org.eclipse.jface.dialogs.MessageDialog;
012    import org.eclipse.jface.wizard.Wizard;
013    import org.eclipse.ui.PlatformUI;

015    import com.google.common.base.Objects;

028    
029    /**
030     * A Wizard that changes Metadata Schema.
031     * A convenience class to avoid code duplication between the Edit Project Wizard and the Import Schema Wizard.
032     */
033    public abstract class MetadataSchemaEditingWizard extends Wizard
034    {
035        private final EditOrCreateProjectWizardPage2 schemaEditPage;
036    
037        private final IProjectService projectService;
038        private final ISchemaService  schemaService;
039        private final IGroupService groupService;
040        private final IResearchFileService researchFileService;
041    
042        protected Project currentProject;
043    
044        public MetadataSchemaEditingWizard(final Project selectedProject, final String title)
045        {
046            this.schemaEditPage = new EditOrCreateProjectWizardPage2(title, "Please select the schema to be used with your project", selectedProject.getSchema());
047            this.currentProject = selectedProject;
048    
049            this.projectService = (IProjectService) PlatformUI.getWorkbench().getService(IProjectService.class);
050            this.schemaService = (ISchemaService) PlatformUI.getWorkbench().getService(ISchemaService.class);
051            this.groupService = (IGroupService) PlatformUI.getWorkbench().getService(IGroupService.class);
052            this.researchFileService = (IResearchFileService) PlatformUI.getWorkbench().getService(IResearchFileService.class);
053        }
054    
055        /**
056         * @{inheritDoc
057         */
058        @Override
059        public void addPages()
060        {
061            addPage(this.schemaEditPage);
062        }
063    
064        /**
065         * @{inheritDoc}
066         */
067        @Override
068        public boolean performFinish()
069        {
070            // Update Schema
071            final Schema currentSchema = currentProject.getSchema();
072    
073            final String localSchemaName = schemaEditPage.getLocalSchemaName();
074            final String localSchemaDescription = schemaEditPage.getLocalSchemaDescription();
075            final String localSchemaNamespaceURL = schemaEditPage.getLocalSchemaNamespaceURL();
076            final Schema importedSchema = schemaEditPage.getImportedSchema();
077    
078            if (currentSchema.getLocal())
079            {
080                if (schemaEditPage.getUseLocalSchema())
081                {
082                    schemaService.updateSchema(currentSchema, localSchemaName, localSchemaDescription, localSchemaNamespaceURL);
083                }
084                else
085                {
086                    final boolean confirm = MessageDialog.openConfirm(getShell(), "Are you sure you want to use an imported metadata schema?",
087                        "Using an imported metadata schema will cause all your current metadata to be removed. Are you sure you want to proceed?");
088                    if (confirm)
089                    {
090                        schemaService.createImportedSchema(importedSchema);
091                        projectService.editProject(importedSchema, currentProject.getId());
092                        deleteSchema(currentSchema);
093                    }
094                }
095            }
096            else
097            {
098                if (schemaEditPage.getUseLocalSchema())
099                {
100                    final boolean confirm = MessageDialog.openConfirm(getShell(), "Are you sure you want to use a local metadata schema?",
101                        "Using a local metadata schema will cause all your current metadata to be removed. Are you sure you want to proceed?");
102                    if (confirm)
103                    {
104                        final Schema localSchema = schemaService.createLocalSchema(localSchemaName, localSchemaDescription, localSchemaNamespaceURL);
105                        projectService.editProject(localSchema, currentProject.getId());
106                        deleteSchema(currentSchema);
107                    }
108                }
109                else
110                {
111                    if (!Objects.equal(currentSchema, importedSchema))
112                    {
113                        final boolean confirm = MessageDialog.openConfirm(getShell(), "Are you sure you want to use an imported metadata schema?",
114                                "Using an imported metadata schema will cause all your current metadata to be removed. Are you sure you want to proceed?");
115                        if (confirm)
116                        {
117                            schemaService.createImportedSchema(importedSchema);
118                            projectService.editProject(importedSchema, currentProject.getId());
119                            deleteSchema(currentSchema);
120                        }
121                    }
122                }
123            }
124    
125            return this.currentProject != null;
126        }
127    
128        private void deleteSchema(final Schema schema)
129        {
130            // Iterate across all the Metadata Category -> Value mappings, remove all Group & File associations.
131            for (final MetadataCategory mdc : schema.getMetadataCategories())
132            {
133                for (final MetadataValue mdv : mdc.getValues())
134                {
135                    final List<ResearchFile> researchFiles = this.researchFileService.getResearchFilesWithAssociatedMetadata(mdc, mdv);
136                    for (final ResearchFile researchFile : researchFiles)
137                    {
138                        this.researchFileService.disassociateMetadata(researchFile, mdc, mdv);
139                    }
140                    final List<Group> groups = this.groupService.getGroupsWithAssociatedMetadata(mdc, mdv);
141                    for (final Group group : groups)
142                    {
143                        this.groupService.disassociateMetadata(group, mdc, mdv);
144                    }
145                }
146            }
147            this.schemaService.removeSchema(schema);
148        }
149    }
