001    package au.org.intersect.exsite9.wizard;
002    
003    import org.eclipse.swt.widgets.Composite;
004    
005    import com.richclientgui.toolbox.validation.ValidatingField;
006    import com.richclientgui.toolbox.validation.ValidationToolkit;
007    
008    /**
009     * Utilities for fields in Wizards.
010     */
011    public final class WizardFieldUtils
012    {
013    
014        /**
015         * Creates a text field configured to be optional, with a maximum length of 255 chars.
016         * @param toolkit The toolkit.
017         * @param composite The composite.
018         * @param fieldName The name of the field.
019         * @param defaultValue The default value.
020         * @return The field.
021         */
022        public static ValidatingField<String> createOptional255TextField(final ValidationToolkit<String> toolkit, final Composite composite,
023                final String fieldName, final String defaultValue)
024        {
025            return toolkit.createTextField(composite, new MaximumLengthFieldValidator(fieldName, 255), false, defaultValue);
026        }
027    }
