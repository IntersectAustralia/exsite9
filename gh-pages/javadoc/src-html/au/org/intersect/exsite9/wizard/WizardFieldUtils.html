007    package au.org.intersect.exsite9.wizard;
008    
009    import org.eclipse.swt.widgets.Composite;
010    
011    import com.richclientgui.toolbox.validation.ValidatingField;
012    import com.richclientgui.toolbox.validation.ValidationToolkit;
013    
014    import au.org.intersect.exsite9.validators.MaximumLengthFieldValidator;
015    
016    /**
017     * Utilities for fields in Wizards.
018     */
019    public final class WizardFieldUtils
020    {
021    
022        /**
023         * Creates a text field configured to be optional, with a maximum length of 255 chars.
024         * @param toolkit The toolkit.
025         * @param composite The composite.
026         * @param fieldName The name of the field.
027         * @param defaultValue The default value.
028         * @return The field.
029         */
030        public static ValidatingField<String> createOptional255TextField(final ValidationToolkit<String> toolkit, final Composite composite,
031                final String fieldName, final String defaultValue)
032        {
033            return toolkit.createTextField(composite, new MaximumLengthFieldValidator(fieldName, 255), false, defaultValue);
034        }
035    }
