283        private boolean loadSchema(final boolean showErrors, final File schemaFile)
284        {
285            try
286            {
287                importedSchema = schemaService.parseSchema(schemaFile);
288                importedSchemaNameField.setText(importedSchema.getName());
289                importedSchemaDescriptionField.setText(importedSchema.getDescription());
290                importedSchemaNamespaceURLField.setText(importedSchema.getNamespaceURL());
291                setPageComplete(allFieldsAreValid());
292                return true;
293            }
294            catch (final SAXException e)
295            {
296                if (showErrors)
297                {
298                    MessageDialog.openError(getShell(), "Could not import Schema", "The XML file is an invalid metadata schema. Reason: " + e.getMessage());
299                }
300                LOG.error(e);
301            }
302            catch (final IOException e)
303            {
304                if (showErrors)
305                {
306                    MessageDialog.openError(getShell(), "Could not import Schema", "Error reading the metadata schema. Reason: " + e.getMessage());
307                }
308                LOG.error(e);
309            }
310            catch (final ParserConfigurationException e)
311            {
312                if (showErrors)
313                {
314                    MessageDialog.openError(getShell(), "Could not import Schema", "Error parsing the metadata schema. Reason: " + e.getMessage());
315                }
316                LOG.error(e);
317            }
318            catch (final InvalidSchemaException e)
319            {
320                if (showErrors)
321                {
322                    MessageDialog.openError(getShell(), "Could not import Schema", "This XML file is an invalid metadata schema. Reason: " + e.getMessage());
323                }
324            }
325            return false;
326        }

345        private void enableLocalSchemaFields()
346        {
347            this.localSchemaRadioButton.setSelection(true);
348            this.importSchemaRadioButton.setSelection(false);
349            this.localSchemaNameField.getControl().setEnabled(true);
350            this.localSchemaDescriptionField.setEnabled(true);
351            this.localSchemaNamespaceURLField.setEnabled(true);
352            this.importSchemaBrowseButton.setEnabled(false);
353        }
354    
355        private void enableImportedSchemaFields()
356        {
357            this.localSchemaRadioButton.setSelection(false);
358            this.importSchemaRadioButton.setSelection(true);
359            this.localSchemaNameField.getControl().setEnabled(false);
360            this.localSchemaDescriptionField.setEnabled(false);
361            this.localSchemaNamespaceURLField.setEnabled(false);
362            this.importSchemaBrowseButton.setEnabled(true);
363            this.importSchemaBrowseButton.setFocus();
364        }

366        /**
367         * @{inheritDoc
368         */
369        @Override
370        public void widgetSelected(final SelectionEvent e)
371        {
372            this.errorMessageHandler.clearMessage();
373            if (e.widget == this.localSchemaRadioButton)
374            {
375                enableLocalSchemaFields();
376            }
377            else if (e.widget == this.importSchemaRadioButton)
378            {
379                enableImportedSchemaFields();
380            }
381            setPageComplete(allFieldsAreValid());
382        }

392        private boolean allFieldsAreValid()
393        {
394            if (this.localSchemaRadioButton.getSelection())
395            {
396                return this.localSchemaNameField.isValid() && !this.localSchemaNameField.getContents().isEmpty();
397            }
398            if (this.importSchemaRadioButton.getSelection())
399            {
400                return (this.importedSchema != null) && (! this.importedSchema.getLocal());
401            }
402            return false;
403        }
