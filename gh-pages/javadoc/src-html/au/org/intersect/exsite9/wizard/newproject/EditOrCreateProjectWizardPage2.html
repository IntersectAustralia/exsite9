014    import org.eclipse.swt.SWT;

025    import au.org.intersect.exsite9.domain.FieldOfResearch;
026    import au.org.intersect.exsite9.dto.ProjectFieldsDTO;

030    /**
031     * Second page of the edit or create new project wizards.
032     */
033    public final class EditOrCreateProjectWizardPage2 extends WizardPage
034    {
035        private final IFieldOfResearchService fieldOfResearchService;
036    
037        private Text electronicLocationText;
038        private Text physicalLocationText;
039        private Text placeOrRegionNameText;
040        private Text geographicalCoverageText;
041        private Text datesOfCaptureText;
042        private Text citationInformationText;
043        private Text countriesText;
044        private Text languagesText;
045        private FieldOfResearch fieldOfResearch;
046        private Text fundingBodyText;
047        private Text grantIDText;
048        private Text relatedPartyText;
049        private Text relatedActivityText;
050        private Text relatedInformationText;
051    
052        private Composite container;
053    
054        private ProjectFieldsDTO incomingFieldValues;
055    
056        /**
057         * Constructor
058         */
059        public EditOrCreateProjectWizardPage2(final String pageTitle, final String pageDescription, final ProjectFieldsDTO incomingFieldValues)
060        {
061            super(pageTitle);
062            setTitle(pageTitle);
063            setDescription(pageDescription);
064            this.incomingFieldValues = incomingFieldValues;
065            this.fieldOfResearchService = (IFieldOfResearchService) PlatformUI.getWorkbench().getService(IFieldOfResearchService.class);
066        }
067    
068        @Override
069        public void createControl(final Composite parent)
070        {
071            this.container = new Composite(parent, SWT.NULL);
072            final GridLayout layout = new GridLayout();
073            this.container.setLayout(layout);
074            layout.numColumns = 2;
075    
076            final Label electronicLocationLabel = new Label(this.container, SWT.NULL);
077            electronicLocationLabel.setText("Electronic Location");
078    
079            this.electronicLocationText = new Text(this.container, SWT.SINGLE | SWT.BORDER);
080            this.electronicLocationText.setText(this.incomingFieldValues.getElectronicLocation());
081    
082            final Label physicalLocationLabel = new Label(this.container, SWT.NULL);
083            physicalLocationLabel.setText("Physical Location");
084    
085            this.physicalLocationText = new Text(this.container, SWT.SINGLE | SWT.BORDER);
086            this.physicalLocationText.setText(this.incomingFieldValues.getPhysicalLocation());
087    
088            final Label placeOrRegionNameLabel = new Label(this.container, SWT.NULL);
089            placeOrRegionNameLabel.setText("Place or Region Name");
090    
091            this.placeOrRegionNameText = new Text(this.container, SWT.SINGLE | SWT.BORDER);
092            this.placeOrRegionNameText.setText(this.incomingFieldValues.getPlaceOrRegionName());
093    
094            final Label latitudeLongitudeLabel = new Label(this.container, SWT.NULL);
095            latitudeLongitudeLabel.setText("Geographical Coverage");
096    
097            this.geographicalCoverageText = new Text(this.container, SWT.SINGLE | SWT.BORDER);
098            this.geographicalCoverageText.setText(this.incomingFieldValues.getGeographicalCoverage());
099    
100            final Label datesOfCaptureLabel = new Label(this.container, SWT.NULL);
101            datesOfCaptureLabel.setText("Dates of Capture");
102    
103            this.datesOfCaptureText = new Text(this.container, SWT.SINGLE | SWT.BORDER);
104            this.datesOfCaptureText.setText(this.incomingFieldValues.getDatesOfCapture());
105    
106            final Label citationInformationLabel = new Label(this.container, SWT.NULL);
107            citationInformationLabel.setText("Citation Information");
108    
109            this.citationInformationText = new Text(this.container, SWT.SINGLE | SWT.BORDER);
110            this.citationInformationText.setText(this.incomingFieldValues.getCitationInformation());
111    
112            final Label countriesLabel = new Label(this.container, SWT.NULL);
113            countriesLabel.setText("Countries");
114    
115            this.countriesText = new Text(this.container, SWT.SINGLE | SWT.BORDER);
116            this.countriesText.setText(this.incomingFieldValues.getCountries());
117    
118            final Label languagesLabel = new Label(this.container, SWT.NULL);
119            languagesLabel.setText("Languages");
120    
121            this.languagesText = new Text(this.container, SWT.SINGLE | SWT.BORDER);
122            this.languagesText.setText(this.incomingFieldValues.getLanguages());
123            
124            final Label fieldOfResearchLabel = new Label(this.container, SWT.NULL);
125            fieldOfResearchLabel.setText("Field of Research");
126    
127            final Text fieldOfResearchText = new Text(this.container, SWT.SINGLE | SWT.BORDER);
128            fieldOfResearchText.setEditable(false);
129            fieldOfResearchText.setEnabled(false);
130            this.fieldOfResearch = this.incomingFieldValues.getFieldOfResearch();
131            if (this.fieldOfResearch != null)
132            {
133                fieldOfResearchText.setText(this.fieldOfResearch.toString());
134            }
135    
136            new Label(this.container, SWT.NULL);
137    
138            final Button browseButton = new Button(this.container, SWT.PUSH);
139            browseButton.setText("Select...");
140            browseButton.setLayoutData(new GridData(SWT.RIGHT, SWT.CENTER, false, false));
141    
142            browseButton.addSelectionListener(new SelectionListener()
143            {
144                @Override
145                public void widgetSelected(final SelectionEvent e)
146                {
147                    final ElementListSelectionDialog selectDialog = new ElementListSelectionDialog(getShell(), new LabelProvider());
148                    final List<FieldOfResearch> fieldsOfResearch = fieldOfResearchService.getAll();
149                    selectDialog.setElements(fieldsOfResearch.toArray());
150                    final int response = selectDialog.open();
151                    if (response != Window.OK)
152                    {
153                        return;
154                    }
155                    fieldOfResearch = (FieldOfResearch) selectDialog.getFirstResult();
156                    fieldOfResearchText.setText(fieldOfResearch.toString());
157                }
158                
159                @Override
160                public void widgetDefaultSelected(final SelectionEvent e)
161                {
162                }
163            });
164    
165            final Label fundingBodyLabel = new Label(this.container, SWT.NULL);
166            fundingBodyLabel.setText("Funding Body");
167    
168            this.fundingBodyText = new Text(this.container, SWT.SINGLE | SWT.BORDER);
169            this.fundingBodyText.setText(this.incomingFieldValues.getFundingBody());
170            
171            final Label grantIDLabel = new Label(this.container, SWT.NULL);
172            grantIDLabel.setText("Grant ID");
173    
174            this.grantIDText = new Text(this.container, SWT.NULL | SWT.BORDER);
175            grantIDText.setText(this.incomingFieldValues.getGrantID());
176    
177            final Label relatedPartyLabel = new Label(this.container, SWT.NULL);
178            relatedPartyLabel.setText("Related Party");
179    
180            this.relatedPartyText = new Text(this.container, SWT.SINGLE | SWT.BORDER);
181            this.relatedPartyText.setText(this.incomingFieldValues.getRelatedParty());
182    
183            final Label relatedActivityLabel = new Label(this.container, SWT.NULL);
184            relatedActivityLabel.setText("Related Grant");
185    
186            this.relatedActivityText = new Text(this.container, SWT.SINGLE | SWT.BORDER);
187            this.relatedActivityText.setText(this.incomingFieldValues.getRelatedGrant());
188    
189            final Label relatedInformationLabel = new Label(this.container, SWT.NULL);
190            relatedInformationLabel.setText("Related Information");
191    
192            this.relatedInformationText = new Text(this.container, SWT.SINGLE | SWT.BORDER);
193            this.relatedInformationText.setText(this.incomingFieldValues.getRelatedInformation());
194    
195            final GridData singleLineGridData = new GridData(GridData.FILL_HORIZONTAL);
196    
197            this.electronicLocationText.setLayoutData(singleLineGridData);
198            this.physicalLocationText.setLayoutData(singleLineGridData);
199            this.placeOrRegionNameText.setLayoutData(singleLineGridData);
200            this.geographicalCoverageText.setLayoutData(singleLineGridData);
201            this.datesOfCaptureText.setLayoutData(singleLineGridData);
202            this.citationInformationText.setLayoutData(singleLineGridData);
203            this.countriesText.setLayoutData(singleLineGridData);
204            this.languagesText.setLayoutData(singleLineGridData);
205            fieldOfResearchText.setLayoutData(singleLineGridData);
206            this.fundingBodyText.setLayoutData(singleLineGridData);
207            this.grantIDText.setLayoutData(singleLineGridData);
208            this.relatedPartyText.setLayoutData(singleLineGridData);
209            this.relatedActivityText.setLayoutData(singleLineGridData);
210            this.relatedInformationText.setLayoutData(singleLineGridData);
211    
212            setControl(this.container);
213            setPageComplete(true);
214        }
