007    package au.org.intersect.exsite9.wizard.listexcludedfiles;
008    
009    import java.util.ArrayList;
010    import java.util.List;
011    
012    import org.eclipse.jface.wizard.WizardPage;
013    import org.eclipse.swt.SWT;
014    import org.eclipse.swt.events.SelectionEvent;
015    import org.eclipse.swt.events.SelectionListener;
016    import org.eclipse.swt.layout.GridData;
017    import org.eclipse.swt.layout.GridLayout;
018    import org.eclipse.swt.layout.RowLayout;
019    import org.eclipse.swt.widgets.Button;
020    import org.eclipse.swt.widgets.Composite;
021    
022    import au.org.intersect.exsite9.domain.ResearchFile;
023    
024    /**
025     * Excluded Files wizard page 1.
026     */
027    public final class ListExcludedFilesWizardPage1 extends WizardPage implements SelectionListener
028    {
029        private final List<ResearchFile> excludedFiles;
030    
031        private final List<ResearchFile> excludedFilesToInclude = new ArrayList<ResearchFile>();
032    
033        private org.eclipse.swt.widgets.List excludedFilesList;
034    
035        private Button removeButton;
036    
037        /**
038         * Constructor
039         * @param excludedFiles The list of currently excluded files.
040         */
041        public ListExcludedFilesWizardPage1(final List<ResearchFile> excludedFiles)
042        {
043            super("Excluded Files");
044            setTitle("Excluded Files");
045            setDescription("These are files that have been excluded from the project.");
046            this.excludedFiles = excludedFiles;
047        }
048    
049        /**
050         * @{inheritDoc}
051         */
052        @Override
053        public void createControl(final Composite parent)
054        {
055            final Composite container = new Composite(parent, SWT.NULL);
056            final GridLayout layout = new GridLayout();
057            layout.numColumns = 2;
058            container.setLayout(layout);
059            
060            this.excludedFilesList = new org.eclipse.swt.widgets.List(container, SWT.BORDER | SWT.MULTI | SWT.WRAP | SWT.V_SCROLL);
061    
062            for (final ResearchFile rf : this.excludedFiles)
063            {
064                this.excludedFilesList.add(rf.getFile().getAbsolutePath());
065            }
066    
067            final GridData multiLineGridData = new GridData(GridData.FILL_BOTH);
068            this.excludedFilesList.setLayoutData(multiLineGridData);
069    
070            Composite rowComp = new Composite(container, SWT.NULL);
071    
072            final RowLayout rowLayout = new RowLayout();
073            rowLayout.type = SWT.VERTICAL;
074            rowLayout.pack = false;
075            rowLayout.justify = true;
076            rowComp.setLayout(rowLayout);
077    
078            this.removeButton = new Button(rowComp, SWT.PUSH);
079            this.removeButton.setText("Reinclude");
080            this.removeButton.addSelectionListener(this);
081            this.removeButton.setEnabled(false);
082    
083            this.excludedFilesList.addSelectionListener(new SelectionListener()
084            {
085                @Override
086                public void widgetSelected(final SelectionEvent e)
087                {
088                    removeButton.setEnabled(excludedFilesList.getSelectionCount() > 0);
089                }
090    
091                @Override
092                public void widgetDefaultSelected(final SelectionEvent e)
093                {
094                }
095            });
096    
097            setControl(container);
098            setPageComplete(false);
099        }
100    
101        @Override
102        public void widgetSelected(final SelectionEvent event)
103        {
104            if (this.excludedFilesList.getSelectionCount() == 0)
105            {
106                return;
107            }
108    
109            while (this.excludedFilesList.getSelectionCount() > 0)
110            {
111                final int selectedIndex = this.excludedFilesList.getSelectionIndex();
112                final ResearchFile selectedResearchFile = this.excludedFiles.remove(selectedIndex);
113                this.excludedFilesToInclude.add(selectedResearchFile);
114                this.excludedFilesList.remove(selectedIndex);
115            }
116    
117            this.setPageComplete(true);
118            this.removeButton.setEnabled(excludedFilesList.getSelectionCount() > 0);
119        }
120    
121        @Override
122        public void widgetDefaultSelected(final SelectionEvent event)
123        {
124        }
125    
126        public List<ResearchFile> getExcludedFilesToInclude()
127        {
128            return this.excludedFilesToInclude;
129        }
130    }
