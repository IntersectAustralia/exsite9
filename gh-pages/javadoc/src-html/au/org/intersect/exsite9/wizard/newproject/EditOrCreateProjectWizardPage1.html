009    import org.eclipse.jface.wizard.WizardPage;
010    import org.eclipse.swt.SWT;
011    import org.eclipse.swt.events.KeyEvent;
012    import org.eclipse.swt.events.KeyListener;
013    import org.eclipse.swt.events.SelectionEvent;
014    import org.eclipse.swt.events.SelectionListener;
015    import org.eclipse.swt.layout.GridData;
016    import org.eclipse.swt.layout.GridLayout;
017    import org.eclipse.swt.widgets.Combo;
018    import org.eclipse.swt.widgets.Composite;
019    import org.eclipse.swt.widgets.Label;
020    import org.eclipse.swt.widgets.Text;
021    
022    import com.richclientgui.toolbox.validation.ValidatingField;
023    import com.richclientgui.toolbox.validation.string.StringValidationToolkit;
024    import com.richclientgui.toolbox.validation.validator.IFieldValidator;
025    
026    import au.org.intersect.exsite9.dto.ProjectFieldsDTO;
027    import au.org.intersect.exsite9.validators.ProjectNameValidator;
028    import au.org.intersect.exsite9.validators.NonEmptyValidator;
029    
030    /**
031     * The first page of the new project wizard.
032     */
033    public final class EditOrCreateProjectWizardPage1 extends WizardPage implements KeyListener
034    {
035        private ValidatingField<String> projectNameText;
036        private IFieldValidator<String> projectNameValidator;
037        private ValidatingField<String> ownerText;
038        private IFieldValidator<String> ownerValidator;
039        private ValidatingField<String> emailText;
040        private IFieldValidator<String> emailValidator;
041        private Text institutionText;
042        private Text descriptionText;
043        private Combo collectionDropDown;
044        private Text rightsStatementText;
045        private Text accessRightsText;
046        private Text licenceText;
047        private Text identifierText;
048        private Text subjectText;
049    
050        private ProjectFieldsDTO incomingFieldValues;
051    
052        private StringValidationToolkit stringValidatorToolkit;
053    
054        private Composite container;
055    
056        /**
057         * Constructor
058         */
059        public EditOrCreateProjectWizardPage1(final String pageTitle, final String pageDescription, final ProjectFieldsDTO incomingFieldValues)
060        {
061            super(pageTitle);
062            super.setTitle(pageTitle);
063            super.setDescription(pageDescription);
064            this.incomingFieldValues = incomingFieldValues;
065        }
066    
067        /**
068         * @{inheritDoc
069         */
070        @Override
071        public void createControl(final Composite parent)
072        {
073            this.container = new Composite(parent, SWT.NULL);
074            final GridLayout layout = new GridLayout();
075            this.container.setLayout(layout);
076            layout.numColumns = 2;
077    
078            this.stringValidatorToolkit = new StringValidationToolkit(SWT.TOP | SWT.LEFT, 1, true);
079    
080            final Label identifierLabel = new Label(this.container, SWT.NULL);
081            identifierLabel.setText("Identifier");
082    
083            this.identifierText = new Text(this.container, SWT.SINGLE | SWT.BORDER);
084            this.identifierText.setText(this.incomingFieldValues.getIdentifier());
085    
086            final Label projectNameLabel = new Label(this.container, SWT.NULL);
087            projectNameLabel.setText("Project Name");
088    
089            this.projectNameValidator = new ProjectNameValidator();
090            this.projectNameText = this.stringValidatorToolkit.createTextField(this.container, this.projectNameValidator, true, this.incomingFieldValues.getName());
091            this.projectNameText.getControl().addKeyListener(this);
092    
093            final Label nameLabel = new Label(this.container, SWT.NULL);
094            nameLabel.setText("Name");
095    
096            this.ownerValidator = new NonEmptyValidator("Name");
097            this.ownerText = this.stringValidatorToolkit.createTextField(this.container, this.ownerValidator, true, this.incomingFieldValues.getOwner());
098            this.ownerText.getControl().addKeyListener(this);
099    
100            final Label institutionLabel = new Label(this.container, SWT.NULL);
101            institutionLabel.setText("Institution");
102    
103            this.institutionText = new Text(this.container, SWT.SINGLE | SWT.BORDER);
104            this.institutionText.setText(this.incomingFieldValues.getInstitution());
105    
106            final Label emailLabel = new Label(this.container, SWT.NULL);
107            emailLabel.setText("Email");
108    
109            this.emailValidator = new NonEmptyValidator("Email");
110            this.emailText = this.stringValidatorToolkit.createTextField(this.container, this.emailValidator, true, this.incomingFieldValues.getEmail());
111            this.emailText.getControl().addKeyListener(this);
112            
113            final Label projectDescriptionLabel = new Label(this.container, SWT.NULL);
114            projectDescriptionLabel.setText("Description");
115    
116            this.descriptionText = new Text(this.container, SWT.BORDER | SWT.MULTI | SWT.WRAP | SWT.V_SCROLL);
117            this.descriptionText.setText(this.incomingFieldValues.getDescription());
118            
119            // 3 empty cells due to the description field spanning 4 rows below
120            new Label(container, SWT.NULL);
121            new Label(container, SWT.NULL);
122            new Label(container, SWT.NULL);
123    
124            final Label collectionTypeLabel = new Label(this.container, SWT.NULL);
125            collectionTypeLabel.setText("Collection Type");
126    
127            this.collectionDropDown = new Combo(this.container, SWT.DROP_DOWN | SWT.READ_ONLY | SWT.SINGLE);
128            this.collectionDropDown.setItems(new String[] {"", "Dataset", "Collection"});
129            for (int i = 0; i < this.collectionDropDown.getItemCount(); i++)
130            {
131                if (this.collectionDropDown.getItem(i).equalsIgnoreCase(this.incomingFieldValues.getCollectionType()))
132                {
133                    this.collectionDropDown.select(i);
134                }
135            }
136    
137            this.collectionDropDown.addSelectionListener(new SelectionListener()
138            {
139                @Override
140                public void widgetSelected(final SelectionEvent e)
141                {
142                    setPageComplete(allFieldsAreValid());                
143                }
144                
145                @Override
146                public void widgetDefaultSelected(final SelectionEvent e)
147                {
148                }
149            });
150    
151            final Label rightsStatementLabel = new Label(this.container, SWT.NULL);
152            rightsStatementLabel.setText("Rights Statement");
153    
154            this.rightsStatementText = new Text(this.container, SWT.SINGLE | SWT.BORDER);
155            this.rightsStatementText.setText(this.incomingFieldValues.getRightsStatement());
156    
157            final Label accessRightsLabel = new Label(this.container, SWT.NULL);
158            accessRightsLabel.setText("Access Rights");
159    
160            this.accessRightsText = new Text(this.container, SWT.SINGLE | SWT.BORDER);
161            this.accessRightsText.setText(this.incomingFieldValues.getAccessRights());
162    
163            final Label licenceLabel = new Label(this.container, SWT.NULL);
164            licenceLabel.setText("Licence");
165    
166            this.licenceText = new Text(this.container, SWT.SINGLE | SWT.BORDER);
167            this.licenceText.setText(this.incomingFieldValues.getLicence());
168    
169            final Label subjectLabel = new Label(this.container, SWT.NULL);
170            subjectLabel.setText("Subject");
171    
172            this.subjectText = new Text(this.container, SWT.SINGLE | SWT.BORDER);
173            this.subjectText.setText(this.incomingFieldValues.getSubject());
174    
175    
176            final GridData singleLineGridData = new GridData(GridData.FILL_HORIZONTAL);
177            final GridData multiLineGridData = new GridData(GridData.FILL_BOTH);
178            multiLineGridData.verticalSpan = 4;
179    
180            this.projectNameText.getControl().setLayoutData(singleLineGridData);
181            this.ownerText.getControl().setLayoutData(singleLineGridData);
182            this.institutionText.setLayoutData(singleLineGridData);
183            this.emailText.getControl().setLayoutData(singleLineGridData);
184            this.descriptionText.setLayoutData(multiLineGridData);
185            this.collectionDropDown.setLayoutData(singleLineGridData);
186            this.rightsStatementText.setLayoutData(singleLineGridData);
187            this.accessRightsText.setLayoutData(singleLineGridData);
188            this.licenceText.setLayoutData(singleLineGridData);
189            this.identifierText.setLayoutData(singleLineGridData);
190            this.subjectText.setLayoutData(singleLineGridData);
191    
192            setControl(this.container);
193            setPageComplete(allFieldsAreValid());
194        }
195    
196        /**
197         * @{inheritDoc
198         */
199        @Override
200        public void keyPressed(final KeyEvent e)
201        {
202        }
203    
204        /**
205         * @{inheritDoc
206         */
207        @Override
208        public void keyReleased(final KeyEvent e)
209        {
210            setPageComplete(allFieldsAreValid());
211        }
212    
213        // Not to override super.getName()
214        public String getProjectName()
215        {
216            return this.projectNameText.getContents().trim();
217        }
218    
219        public String getOwner()
220        {
221            return this.ownerText.getContents().trim();
222        }
223    
224        public String getInstitution()
225        {
226            return this.institutionText.getText().trim();
227        }
228    
229        public String getEmail()
230        {
231            return this.emailText.getContents().trim();
232        }
233    
234        // Not to override super.getDescription()
235        public String getProjectDescription()
236        {
237            return this.descriptionText.getText().trim();
238        }
239    
240        public String getCollectionType()
241        {
242            return this.collectionDropDown.getItem(this.collectionDropDown.getSelectionIndex());
243        }
244    
245        public String getRightsStatement()
246        {
247            return this.rightsStatementText.getText().trim();
248        }
249    
250        public String getAccessRights()
251        {
252            return this.accessRightsText.getText().trim();
253        }
254    
255        public String getLicense()
256        {
257            return this.licenceText.getText().trim();
258        }
259    
260        public String getIdentifier()
261        {
262            return this.identifierText.getText().trim();
263        }
264    
265        public String getSubject()
266        {
267            return this.subjectText.getText().trim();
268        }
269        
270        private boolean allFieldsAreValid()
271        {
272            if (!this.projectNameText.isValid())
273            {
274                setErrorMessage(this.projectNameValidator.getErrorMessage());
275                return false;
276            }
277            if (!this.ownerText.isValid())
278            {
279                setErrorMessage(this.ownerValidator.getErrorMessage());
280                return false;
281            }
282            if (!this.emailText.isValid())
283            {
284                setErrorMessage(this.emailValidator.getErrorMessage());
285                return false;
286            }
287    
288            setErrorMessage(null);
289            return true;
290        }
