083        public EditOrCreateProjectWizardPage1(final String pageTitle, final String pageDescription, final ProjectFieldsDTO incomingFieldValues)
084        {
085            super(pageTitle);
086            setTitle(pageTitle);
087            setDescription(pageDescription);

095        @Override
096        public void createControl(final Composite parent)
097        {
098            this.container = new Composite(parent, SWT.NULL);
099            final GridLayout layout = new GridLayout();
100            this.container.setLayout(layout);
101            layout.numColumns = 2;
102    
103            this.stringValidatorToolkit = new StringValidationToolkit(SWT.TOP | SWT.LEFT, 1, true);
104            this.stringValidatorToolkit.setDefaultErrorMessageHandler(this.errorMessageHandler);
105    
106            final Label projectNameLabel = new Label(this.container, SWT.NULL);
107            projectNameLabel.setText("Project Name");
108    
109            this.projectNameText = this.stringValidatorToolkit.createTextField(this.container, new MaximumLengthFieldValidator("Project Name", 255),
110                    true, this.incomingFieldValues.getName());
111            this.projectNameText.getControl().addKeyListener(this);
112    
113            final Label nameLabel = new Label(this.container, SWT.NULL);
114            nameLabel.setText("Name");
115    
116            this.ownerText = this.stringValidatorToolkit.createTextField(this.container, new TrueFieldValidator(), true, this.incomingFieldValues.getOwner());
117            this.ownerText.getControl().addKeyListener(this);
118    
119            final Label institutionLabel = new Label(this.container, SWT.NULL);
120            institutionLabel.setText("Institution");
121    
122            this.institutionText = new Text(this.container, SWT.SINGLE | SWT.BORDER);
123            this.institutionText.setText(this.incomingFieldValues.getInstitution());
124    
125            final Label emailLabel = new Label(this.container, SWT.NULL);
126            emailLabel.setText("Email");
127    
128            this.emailText = this.stringValidatorToolkit.createTextField(this.container, new TrueFieldValidator(), true, this.incomingFieldValues.getEmail());
129            this.emailText.getControl().addKeyListener(this);
130            
131            final Label projectDescriptionLabel = new Label(this.container, SWT.NULL);
132            projectDescriptionLabel.setText("Description");
133    
134            this.descriptionText = new Text(this.container, SWT.BORDER | SWT.MULTI | SWT.WRAP | SWT.V_SCROLL);
135            this.descriptionText.setText(this.incomingFieldValues.getDescription());
136            
137            // 3 empty cells due to the description field spanning 4 rows below
138            new Label(container, SWT.NULL);
139            new Label(container, SWT.NULL);
140            new Label(container, SWT.NULL);
141    
142            final Label collectionTypeLabel = new Label(this.container, SWT.NULL);
143            collectionTypeLabel.setText("Collection Type");
144    
145            this.collectionDropDown = new Combo(this.container, SWT.DROP_DOWN | SWT.READ_ONLY | SWT.SINGLE);
146            this.collectionDropDown.setItems(new String[] {"", "Dataset", "Collection"});
147            for (int i = 0; i < this.collectionDropDown.getItemCount(); i++)
148            {
149                if (this.collectionDropDown.getItem(i).equalsIgnoreCase(this.incomingFieldValues.getCollectionType()))
150                {
151                    this.collectionDropDown.select(i);
152                }
153            }
154    
155            this.collectionDropDown.addSelectionListener(new SelectionListener()
156            {
157                @Override
158                public void widgetSelected(final SelectionEvent e)
159                {
160                    setPageComplete(allFieldsAreValid());                
161                }
162                
163                @Override
164                public void widgetDefaultSelected(final SelectionEvent e)
165                {
166                }
167            });
168    
169            final Label rightsStatementLabel = new Label(this.container, SWT.NULL);
170            rightsStatementLabel.setText("Rights Statement");

175            final Label accessRightsLabel = new Label(this.container, SWT.NULL);
176            accessRightsLabel.setText("Access Rights");

178            this.accessRightsText = new Text(this.container, SWT.SINGLE | SWT.BORDER);
179            this.accessRightsText.setText(this.incomingFieldValues.getAccessRights());
180    
181            final Label licenceLabel = new Label(this.container, SWT.NULL);
182            licenceLabel.setText("Licence");
183    
184            this.licenceText = new Text(this.container, SWT.SINGLE | SWT.BORDER);
185            this.licenceText.setText(this.incomingFieldValues.getLicence());
186    
187            final Label identifierLabel = new Label(this.container, SWT.NULL);
188            identifierLabel.setText("Identifier");
189    
190            this.identifierText = new Text(this.container, SWT.SINGLE | SWT.BORDER);
191            this.identifierText.setText(this.incomingFieldValues.getIdentifier());
192    
193            final Label subjectLabel = new Label(this.container, SWT.NULL);
194            subjectLabel.setText("Subject");
195    
196            this.subjectText = new Text(this.container, SWT.SINGLE | SWT.BORDER);
197            this.subjectText.setText(this.incomingFieldValues.getSubject());
198    
199            final Label electronicLocationLabel = new Label(this.container, SWT.NULL);
200            electronicLocationLabel.setText("Electronic Location");
201    
202            this.electronicLocationText = new Text(this.container, SWT.SINGLE | SWT.BORDER);
203            this.electronicLocationText.setText(this.incomingFieldValues.getElectronicLocation());
204    
205            final Label physicalLocationLabel = new Label(this.container, SWT.NULL);
206            physicalLocationLabel.setText("Physical Location");
207    
208            this.physicalLocationText = new Text(this.container, SWT.SINGLE | SWT.BORDER);

210

215            this.placeOrRegionNameText.setText(this.incomingFieldValues.getPlaceOrRegionName());

221            this.geographicalCoverageText.setText(this.incomingFieldValues.getGeographicalCoverage());
222    
223            final Label datesOfCaptureLabel = new Label(this.container, SWT.NULL);
224            datesOfCaptureLabel.setText("Dates of Capture");

226            this.datesOfCaptureText = new Text(this.container, SWT.SINGLE | SWT.BORDER);
227            this.datesOfCaptureText.setText(this.incomingFieldValues.getDatesOfCapture());
228    
229            final Label citationInformationLabel = new Label(this.container, SWT.NULL);
230            citationInformationLabel.setText("Citation Information");
231    
232            this.citationInformationText = new Text(this.container, SWT.SINGLE | SWT.BORDER);

234

242            languagesLabel.setText("Languages");

245            this.languagesText.setText(this.incomingFieldValues.getLanguages());
246

248            fieldOfResearchLabel.setText("Field of Research");

258    
259            new Label(this.container, SWT.NULL);
260    
261            final Button browseButton = new Button(this.container, SWT.PUSH);
262            browseButton.setText("Select...");
263            browseButton.setLayoutData(new GridData(SWT.RIGHT, SWT.CENTER, false, false));
264    
265            browseButton.addSelectionListener(new SelectionListener()
266            {
267                @Override
268                public void widgetSelected(final SelectionEvent e)
269                {
270                    final ElementListSelectionDialog selectDialog = new ElementListSelectionDialog(getShell(), new LabelProvider());
271                    final List<FieldOfResearch> fieldsOfResearch = fieldOfResearchService.getAll();
272                    selectDialog.setElements(fieldsOfResearch.toArray());
273                    final int response = selectDialog.open();
274                    if (response != Window.OK)
275                    {
276                        return;
277                    }
278                    fieldOfResearch = (FieldOfResearch) selectDialog.getFirstResult();
279                    fieldOfResearchText.setText(fieldOfResearch.toString());
280                }
281                
282                @Override
283                public void widgetDefaultSelected(final SelectionEvent e)
284                {
285                }
286            });
287    
288            final Label fundingBodyLabel = new Label(this.container, SWT.NULL);
289            fundingBodyLabel.setText("Funding Body");
290    
291            this.fundingBodyText = new Text(this.container, SWT.SINGLE | SWT.BORDER);

294            final Label grantIDLabel = new Label(this.container, SWT.NULL);

300            final Label relatedPartyLabel = new Label(this.container, SWT.NULL);

304            this.relatedPartyText.setText(this.incomingFieldValues.getRelatedParty());

307            relatedActivityLabel.setText("Related Grant");

322            this.projectNameText.getControl().setLayoutData(singleLineGridData);

333            this.electronicLocationText.setLayoutData(singleLineGridData);
334            this.physicalLocationText.setLayoutData(singleLineGridData);
335            this.placeOrRegionNameText.setLayoutData(singleLineGridData);
336            this.geographicalCoverageText.setLayoutData(singleLineGridData);
337            this.datesOfCaptureText.setLayoutData(singleLineGridData);
338            this.citationInformationText.setLayoutData(singleLineGridData);
339            this.countriesText.setLayoutData(singleLineGridData);
340            this.languagesText.setLayoutData(singleLineGridData);
341            fieldOfResearchText.setLayoutData(singleLineGridData);
342            this.fundingBodyText.setLayoutData(singleLineGridData);
343            this.grantIDText.setLayoutData(singleLineGridData);
344            this.relatedPartyText.setLayoutData(singleLineGridData);
345            this.relatedActivityText.setLayoutData(singleLineGridData);
346            this.relatedInformationText.setLayoutData(singleLineGridData);
347    
348            setControl(this.container);
349            setPageComplete(allFieldsAreValid());
350        }

369        public ProjectFieldsDTO getProjectFields()
370        {
371            return new ProjectFieldsDTO(this.projectNameText.getContents().trim(),
372                    this.ownerText.getContents().trim(), this.institutionText.getText().trim(),
373                    this.emailText.getContents().trim(), this.descriptionText.getText().trim(),                
374                    this.collectionDropDown.getItem(this.collectionDropDown.getSelectionIndex()),
375                    this.rightsStatementText.getText().trim(), this.accessRightsText.getText().trim(),
376                    this.licenceText.getText().trim(), this.identifierText.getText().trim(),
377                    this.subjectText.getText().trim(), this.electronicLocationText.getText().trim(),
378                    this.physicalLocationText.getText().trim(), this.placeOrRegionNameText.getText().trim(),
379                    this.geographicalCoverageText.getText().trim(), this.datesOfCaptureText.getText().trim(),
380                    this.citationInformationText.getText().trim(), this.countriesText.getText().trim(),
381                    this.languagesText.getText().trim(), this.fieldOfResearch,
382                    this.fundingBodyText.getText().trim(), this.grantIDText.getText().trim(),
383                    this.relatedPartyText.getText().trim(), this.relatedActivityText.getText().trim(),
384                    this.relatedInformationText.getText().trim());
385        }
386        
387        private boolean allFieldsAreValid()
388        {
389            return this.projectNameText.isValid() && !this.projectNameText.getContents().trim().isEmpty() &&
390                this.ownerText.isValid() && !this.ownerText.getContents().trim().isEmpty() &&
391                this.emailText.isValid() && !this.emailText.getContents().trim().isEmpty();
392        }
