007    package au.org.intersect.exsite9.wizard.metadatacategory;
008    
009    import java.util.Collections;
010    import java.util.List;
011    
012    import org.eclipse.jface.wizard.Wizard;
013    import org.eclipse.ui.PlatformUI;
014    
015    import au.org.intersect.exsite9.domain.Group;
016    import au.org.intersect.exsite9.domain.MetadataAttribute;
017    import au.org.intersect.exsite9.domain.MetadataAttributeValue;
018    import au.org.intersect.exsite9.domain.MetadataCategory;
019    import au.org.intersect.exsite9.domain.MetadataCategoryUse;
020    import au.org.intersect.exsite9.domain.MetadataValue;
021    import au.org.intersect.exsite9.domain.ResearchFile;
022    import au.org.intersect.exsite9.domain.Schema;
023    import au.org.intersect.exsite9.service.IGroupService;
024    import au.org.intersect.exsite9.service.IMetadataCategoryService;
025    import au.org.intersect.exsite9.service.IResearchFileService;
026    
027    /**
028     * Wizard used to Edit a Metadata Category.
029     */
030    public final class EditMetadataCategoryWizard extends Wizard
031    {
032        private final Schema schema;
033    
034        private final ListMetadataCategoriesWizardPage listCategoriesPage;
035        private final AddMetadataCategoryWizardPage1 editCategoryPage;
036    
037        public EditMetadataCategoryWizard(final Schema schema)
038        {
039            super();
040            setNeedsProgressMonitor(true);
041            this.schema = schema;
042            this.listCategoriesPage = new ListMetadataCategoriesWizardPage("Edit Metadata Category", "Choose a metadata category to edit.", this.schema, false);
043            this.editCategoryPage = new AddMetadataCategoryWizardPage1("Edit Metadata Category",
044                    "Edit the details of the metadata category you have selected", schema, null, Collections.<MetadataValue>emptyList(), Collections.<MetadataAttributeValue>emptyList());
045        }
046    
047        @Override
048        public void addPages()
049        {
050            addPage(this.listCategoriesPage);
051            addPage(this.editCategoryPage);
052        }
053    
054        @Override
055        public boolean performFinish()
056        {
057            final IMetadataCategoryService metadataCategoryService = (IMetadataCategoryService) PlatformUI.getWorkbench().getService(IMetadataCategoryService.class);
058            final IGroupService groupService = (IGroupService) PlatformUI.getWorkbench().getService(IGroupService.class);
059            final IResearchFileService fileService = (IResearchFileService) PlatformUI.getWorkbench().getService(IResearchFileService.class);
060    
061            final String categoryTitle = editCategoryPage.getMetadataCategoryName();
062            
063            final MetadataCategoryUse categoryUse = editCategoryPage.getMetadataCategoryUse();
064            final List<MetadataValue> values = editCategoryPage.getMetadataCategoryValues();
065            final String categoryDescription = editCategoryPage.getCategoryDescription();
066            final boolean inextensible = editCategoryPage.getIsInextensible();
067    
068            // Update the modified metadata category
069            final MetadataCategory metadataCategory = listCategoriesPage.getSelectedMetadataCategory();
070            if (metadataCategory == null)
071            {
072                return true;
073            }
074    
075            for (final ResearchFile assignedFile : editCategoryPage.getAssignedFiles())
076            {
077                fileService.disassociateMultipleMetadataValues(assignedFile, metadataCategory, editCategoryPage.getMetadataValuesToBeDisassociated());
078            }
079    
080            for (final Group assignedGroup : editCategoryPage.getAssignedGroups())
081            {
082                groupService.disassociateMultipleMetadataValues(assignedGroup, metadataCategory, editCategoryPage.getMetadataValuesToBeDisassociated());
083            }
084    
085            for (final MetadataAttributeValue mdav : editCategoryPage.getMetadataAttributeValuesToBeDisassociated())
086            {
087                groupService.disassociateMetadataAttributeValue(metadataCategory, mdav);
088                fileService.disassociateMetadataAttributeValue(metadataCategory, mdav);
089            }
090    
091            final MetadataAttribute oldMDA = metadataCategory.getMetadataAttribute();
092            final MetadataAttribute newMDA;
093    
094            if (oldMDA != null)
095            {
096                final String attrName = editCategoryPage.getMetadataAttributeName();
097                if (!attrName.isEmpty())
098                {
099                    newMDA = oldMDA;
100                    newMDA.setName(editCategoryPage.getMetadataAttributeName());
101                    newMDA.setMetadataAttributeValues(editCategoryPage.getMetadataAttributeValues());
102                }
103                else
104                {
105                    newMDA = null;
106                }
107            }
108            else
109            {
110                final String attrName = editCategoryPage.getMetadataAttributeName();
111                if (!attrName.isEmpty())
112                {
113                    newMDA = new MetadataAttribute(attrName, editCategoryPage.getMetadataAttributeValues());
114                }
115                else
116                {
117                    newMDA = null;
118                }
119            }
120    
121            metadataCategoryService.updateMetadataCategory(metadataCategory, categoryTitle, categoryDescription, categoryUse, inextensible, values, newMDA);
122            return true;
123        }
124    }
