001    package au.org.intersect.exsite9.wizard.editproject;
002    
003    import org.eclipse.ui.PlatformUI;
004    
005    import au.org.intersect.exsite9.domain.Project;
006    import au.org.intersect.exsite9.dto.ProjectFieldsDTO;
007    import au.org.intersect.exsite9.service.IProjectService;
008    import au.org.intersect.exsite9.wizard.importmetadataschema.MetadataSchemaEditingWizard;
009    import au.org.intersect.exsite9.wizard.newproject.EditOrCreateProjectWizardPage1;
010    
011    public class EditProjectWizard extends MetadataSchemaEditingWizard
012    {
013        private final EditOrCreateProjectWizardPage1 page1;
014    
015        private final IProjectService projectService;
016    
017        /**
018         * Constructor
019         * 
020         * @param selectedProject
021         */
022        public EditProjectWizard(final Project selectedProject)
023        {
024            super(selectedProject, "Edit Project");
025            setNeedsProgressMonitor(true);
026            this.page1 = new EditOrCreateProjectWizardPage1("Edit Project", "Please amend the details of your project",
027                    new ProjectFieldsDTO(selectedProject.getName(), selectedProject.getOwner(),
028                            selectedProject.getInstitution(), selectedProject.getEmail(),
029                            selectedProject.getDescription(), selectedProject.getCollectionType(),
030                            selectedProject.getRightsStatement(), selectedProject.getAccessRights(),
031                            selectedProject.getLicence(), selectedProject.getIdentifier(), selectedProject.getSubject(),
032                            selectedProject.getElectronicLocation(), selectedProject.getPhysicalLocation(),
033                            selectedProject.getPlaceOrRegionName(), selectedProject.getGeographicalCoverage(),
034                            selectedProject.getDatesOfCapture(), selectedProject.getCitationInformation(),
035                            selectedProject.getCountries(), selectedProject.getLanguages(), selectedProject.getFieldOfResearch(),
036                            selectedProject.getFundingBody(), selectedProject.getGrantID(),
037                            selectedProject.getRelatedParty(), selectedProject.getRelatedGrant(),
038                            selectedProject.getRelatedInformation()));
039    
040            this.projectService = (IProjectService) PlatformUI.getWorkbench().getService(IProjectService.class);
041        }
042    
043        /**
044         * @{inheritDoc
045         */
046        @Override
047        public void addPages()
048        {
049            addPage(this.page1);
050            super.addPages();
051        }
052    
053        @Override
054        public boolean performFinish()
055        {
056            // Update project details
057            super.currentProject = projectService.editProject(page1.getProjectFields(), currentProject.getId());
058    
059            // Update schema details
060            return super.performFinish();
061        }
062    
063        public Project updateProject()
064        {
065            return this.currentProject;
066        }
067    }
