007    package au.org.intersect.exsite9.wizard;
008    
009    import org.eclipse.jface.dialogs.DialogPage;
010    import org.eclipse.jface.wizard.WizardPage;
011    
012    import com.richclientgui.toolbox.validation.IFieldErrorMessageHandler;
013    
014    /**
015     * Displays error messages on wizard pages.
016     */
017    public final class WizardPageErrorHandler implements IFieldErrorMessageHandler
018    {
019        private final WizardPage wizardPage;
020    
021        public WizardPageErrorHandler(final WizardPage wizardPage)
022        {
023            this.wizardPage = wizardPage;
024        }
025    
026        @Override
027        public void handleErrorMessage(final String message, final String input)
028        {
029            this.wizardPage.setMessage(null, DialogPage.WARNING);
030            this.wizardPage.setErrorMessage(message);
031        }
032    
033        @Override
034        public void handleWarningMessage(final String message, final String input)
035        {
036            this.wizardPage.setErrorMessage(null);
037            this.wizardPage.setMessage(message, DialogPage.WARNING);
038        }
039        
040        
041        @Override
042        public void clearMessage()
043        {
044            this.wizardPage.setErrorMessage(null);
045            this.wizardPage.setMessage(null);
046        }
047    }
