001    package au.org.intersect.exsite9.wizard.openproject;
002    
003    import java.util.List;
004    
005    import org.eclipse.jface.wizard.Wizard;
006    import org.eclipse.ui.PlatformUI;
007    
008    import au.org.intersect.exsite9.domain.Project;
009    import au.org.intersect.exsite9.service.IProjectService;
010    
011    public class OpenProjectWizard extends Wizard
012    {
013        private OpenProjectWizardPage1 page1;
014        final IProjectService projectService = (IProjectService) PlatformUI.getWorkbench().getService(IProjectService.class);
015    
016        private Project selectedProject;
017    
018        /**
019         * Constructor 
020         */
021        public OpenProjectWizard()
022        {
023            super();
024            setNeedsProgressMonitor(true);
025        }
026    
027        /**
028         * @{inheritDoc}
029         */
030        @Override
031        public void addPages()
032        {
033            final List<Project> availableProjects = projectService.getAllProjects();
034            page1 = new OpenProjectWizardPage1(availableProjects);
035            addPage(this.page1);
036        }
037    
038        /**
039         * @{inheritDoc}
040         */
041        @Override
042        public boolean performFinish()
043        {
044            this.selectedProject = page1.getSelectedProject();
045            return this.selectedProject != null;
046        }
047    
048        public Project getSelectedProject()
049        {
050            return this.selectedProject;
051        }
052    }
