007    package au.org.intersect.exsite9.wizard.openproject;
008    
009    import java.util.List;
010    
011    import org.eclipse.jface.wizard.Wizard;
012    import org.eclipse.ui.PlatformUI;
013    
014    import au.org.intersect.exsite9.domain.Project;
015    import au.org.intersect.exsite9.service.IProjectService;
016    
017    /**
018     * Wizard used to open an existing {@link Project}, the user is shown the wizard page {@link OpenProjectWizardPage1}
019     */
020    public class OpenProjectWizard extends Wizard
021    {
022        private OpenProjectWizardPage1 page1;
023        final IProjectService projectService = (IProjectService) PlatformUI.getWorkbench().getService(IProjectService.class);
024    
025        private Project selectedProject;
026    
027        /**
028         * Constructor 
029         */
030        public OpenProjectWizard()
031        {
032            super();
033            setNeedsProgressMonitor(true);
034            setWindowTitle("Open Project");
035        }
036    
037        /**
038         * @{inheritDoc}
039         */
040        @Override
041        public void addPages()
042        {
043            final List<Project> availableProjects = projectService.getAllProjects();
044            page1 = new OpenProjectWizardPage1(availableProjects);
045            addPage(this.page1);
046        }
047    
048        /**
049         * @{inheritDoc}
050         */
051        @Override
052        public boolean performFinish()
053        {
054            this.selectedProject = page1.getSelectedProject();
055            return this.selectedProject != null;
056        }
057    
058        public Project getSelectedProject()
059        {
060            return this.selectedProject;
061        }
062    }
