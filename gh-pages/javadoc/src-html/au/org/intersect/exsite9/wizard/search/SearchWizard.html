007    package au.org.intersect.exsite9.wizard.search;
008    
009    import java.util.List;
010    
011    import org.eclipse.jface.wizard.Wizard;
012    import org.eclipse.ui.PlatformUI;
013    
014    import au.org.intersect.exsite9.domain.Project;
015    import au.org.intersect.exsite9.domain.ResearchFile;
016    import au.org.intersect.exsite9.domain.ResearchFileSearchDefinition;
017    import au.org.intersect.exsite9.service.ISearchService;
018    
019    /**
020     * Wizard used to allow a user to do a search, the user is shown the wizard page {@link SearchWizardPage1}
021     */
022    public class SearchWizard extends Wizard
023    {
024        private SearchWizardPage1 page1;
025        private Project currentProject;
026        List<ResearchFile> searchResults;
027    
028        public SearchWizard(Project selectedProject)
029        {
030            super();
031            this.currentProject = selectedProject;
032            setNeedsProgressMonitor(true);
033            setWindowTitle("Search");
034            this.page1 = new SearchWizardPage1(ResearchFileSearchDefinition.toArray());
035        }
036    
037        @Override
038        public void addPages()
039        {
040            addPage(this.page1);
041        }
042    
043        @Override
044        public boolean performFinish()
045        {
046            final ResearchFileSearchDefinition selectedField = page1.getSearchCategory();
047            final String searchTerm = page1.getSearchTerm();
048            final ISearchService searchService = (ISearchService) PlatformUI.getWorkbench()
049                    .getService(ISearchService.class);
050            this.searchResults = searchService.getResearchFilesUsingSearchTerm(searchTerm, selectedField, this.currentProject);
051            return true;
052        }
053        
054        
055        public List<ResearchFile> getSearchResults()
056        {
057            return searchResults;
058        }
059    }
