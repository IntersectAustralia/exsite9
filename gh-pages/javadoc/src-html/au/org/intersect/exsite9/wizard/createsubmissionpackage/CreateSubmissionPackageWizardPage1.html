007    package au.org.intersect.exsite9.wizard.createsubmissionpackage;
008    
009    import java.util.Collection;
010    
011    import org.eclipse.jface.wizard.WizardPage;
012    import org.eclipse.swt.SWT;
013    import org.eclipse.swt.events.KeyEvent;
014    import org.eclipse.swt.events.KeyListener;
015    import org.eclipse.swt.layout.GridData;
016    import org.eclipse.swt.layout.GridLayout;
017    import org.eclipse.swt.widgets.Composite;
018    import org.eclipse.swt.widgets.Label;
019    import org.eclipse.swt.widgets.Text;
020    
021    import com.richclientgui.toolbox.validation.IFieldErrorMessageHandler;
022    import com.richclientgui.toolbox.validation.ValidatingField;
023    import com.richclientgui.toolbox.validation.string.StringValidationToolkit;
024    import com.richclientgui.toolbox.validation.validator.IFieldValidator;
025    
026    import au.org.intersect.exsite9.domain.SubmissionPackage;
027    import au.org.intersect.exsite9.util.DirectoryUtils;
028    import au.org.intersect.exsite9.wizard.WizardPageErrorHandler;
029    
030    /**
031     * First page of the 'Create Submission Package' Wizard.
032     * Allows configuration of a name and description.
033     */
034    public final class CreateSubmissionPackageWizardPage1 extends WizardPage implements KeyListener
035    {
036        private final IFieldErrorMessageHandler errorMessageHandler = new WizardPageErrorHandler(this);
037    
038        private StringValidationToolkit stringValidatorToolkit;
039    
040        private Composite container;
041    
042        private ValidatingField<String> nameField;
043        private Text descriptionField;
044    
045        private SubmissionPackage currentSubmissionPackage;
046        private final Collection<SubmissionPackage> existingSubmissionPackages;
047    
048        public CreateSubmissionPackageWizardPage1(final SubmissionPackage submissionPackage, final Collection<SubmissionPackage> existingSubmissionPackages, final String pageTitle)
049        {
050            super(pageTitle);
051            setTitle(pageTitle);
052            setDescription("Configure properties of the submission package");
053            this.currentSubmissionPackage = submissionPackage;
054            this.existingSubmissionPackages = existingSubmissionPackages;
055        }
056    
057    
058        @Override
059        public void createControl(final Composite parent)
060        {
061            this.container = new Composite(parent, SWT.NULL);
062            final GridLayout layout = new GridLayout();
063            this.container.setLayout(layout);
064            layout.numColumns = 2;
065    
066            this.stringValidatorToolkit = new StringValidationToolkit(SWT.TOP | SWT.LEFT, 1, true);
067            this.stringValidatorToolkit.setDefaultErrorMessageHandler(this.errorMessageHandler);
068    
069            final Label submissionPackageNameLabel = new Label(this.container, SWT.NULL);
070            submissionPackageNameLabel.setText("Name");
071    
072            this.nameField = this.stringValidatorToolkit.createTextField(this.container, new IFieldValidator<String>()
073            {
074                private String errorMessage = "";
075    
076                @Override
077                public boolean warningExist(final String conents)
078                {
079                    return false;
080                }
081    
082                @Override
083                public boolean isValid(final String contents)
084                {
085                    if (contents.trim().isEmpty())
086                    {
087                        this.errorMessage = "Submission package name must not be empty";
088                        return false;
089                    }
090                    if (contents.trim().length() >= 255)
091                    {
092                        this.errorMessage = "Submission package name must be less than 255 characters in length";
093                        return false;
094                    }
095                    if (!DirectoryUtils.isValidDirectoryName(contents.trim()))
096                    {
097                        this.errorMessage = "Submission package name must contain alpha-numeric characters only";
098                        return false;
099                    }
100                    for (final SubmissionPackage existingSubmissionPackage : existingSubmissionPackages)
101                    {
102                        if (currentSubmissionPackage != null && currentSubmissionPackage.equals(existingSubmissionPackage))
103                        {
104                            continue;
105                        }
106                        if (existingSubmissionPackage.getName().equalsIgnoreCase(contents.trim()))
107                        {
108                            this.errorMessage = "A Submission Package with the provided name already exists.";
109                            return false;
110                        }
111                    }
112                    return true;
113                }
114    
115                @Override
116                public String getWarningMessage()
117                {
118                    return "";
119                }
120    
121                @Override
122                public String getErrorMessage()
123                {
124                    return this.errorMessage;
125                }
126            }, true, this.currentSubmissionPackage == null ? "" : this.currentSubmissionPackage.getName());
127    
128            this.nameField.getControl().addKeyListener(this);
129    
130            final Label descriptionLabel = new Label(this.container, SWT.NULL);
131            descriptionLabel.setText("Description");
132    
133            this.descriptionField = new Text(this.container, SWT.BORDER | SWT.MULTI | SWT.WRAP | SWT.V_SCROLL);
134            this.descriptionField.setText(this.currentSubmissionPackage == null ? "" : this.currentSubmissionPackage.getDescription());
135            this.descriptionField.addKeyListener(this);
136    
137            final GridData singleLineGridData = new GridData(GridData.FILL_HORIZONTAL);
138            final GridData multiLineGridData = new GridData(GridData.FILL_BOTH);
139    
140            this.nameField.getControl().setLayoutData(singleLineGridData);
141            this.descriptionField.setLayoutData(multiLineGridData);
142    
143            setControl(this.container);
144            setPageComplete(this.currentSubmissionPackage != null);
145        }
146    
147        /**
148         * @{inheritDoc
149         */
150        @Override
151        public void keyPressed(final KeyEvent e)
152        {
153        }
154    
155        /**
156         * @{inheritDoc
157         */
158        @Override
159        public void keyReleased(final KeyEvent e)
160        {
161            setPageComplete(this.nameField.isValid());
162        }
163    
164        public String getSubmissionPackageName()
165        {
166            return this.nameField.getContents().trim();
167        }
168    
169        public String getSubmissionPackageDescription()
170        {
171            return this.descriptionField.getText().trim();
172        }
173    }
