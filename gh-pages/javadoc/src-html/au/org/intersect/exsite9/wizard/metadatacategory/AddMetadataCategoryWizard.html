009    import java.util.ArrayList;

012    
013    import org.eclipse.jface.wizard.Wizard;
014    import org.eclipse.ui.PlatformUI;

031    /**
032     * Wizard for adding a {@link MetadataCategory} via the {@link AddMetadataCategoryWizardPage1} that is shown to the user
033     */
034    public class AddMetadataCategoryWizard extends Wizard
035    {
036        private final AddMetadataCategoryWizardPage1 page1;
037    
038        private MetadataCategory metadataCategory;
039        private Project project;
040    
041        /**
042         * Constructor
043         * 
044         * @param project
045         * @param metadataCategory
046         *            can be null
047         */
048        public AddMetadataCategoryWizard(Project project, MetadataCategory metadataCategory)
049        {
050            super();
051            setNeedsProgressMonitor(true);
052            this.project = project;
053            this.metadataCategory = metadataCategory;
054            if (metadataCategory == null)
055            {
056                setWindowTitle("New Metadata Category");
057                page1 = new AddMetadataCategoryWizardPage1("New Metadata Category",
058                        "Please enter the details of the metadata category you wish to create", project.getSchema(), null,
059                        new ArrayList<MetadataValue>(), new ArrayList<MetadataAttributeValue>());
060            }
061            else
062            {
063                final List<MetadataAttributeValue> metadataAttribtueValues;
064                metadataAttribtueValues = new ArrayList<MetadataAttributeValue>();
065                setWindowTitle("Edit Metadata Category");
066                if (metadataCategory.getMetadataAttribute() != null)
067                {
068                    metadataAttribtueValues.addAll(metadataCategory.getMetadataAttribute().getMetadataAttributeValues());
069                    Collections.sort(metadataAttribtueValues, new AlphabeticalMetadataAttributeValueComparator());
070                }
071                page1 = new AddMetadataCategoryWizardPage1("Edit Metadata Category",
072                        "Edit the details of the metadata category you have selected", project.getSchema(),
073                        metadataCategory, metadataCategory.getValues(), metadataAttribtueValues);
074            }
075        }
076    
077        @Override
078        public void addPages()
079        {
080            addPage(this.page1);
081        }
082    
083        @Override
084        public boolean performFinish()
085        {
086            final String categoryTitle = page1.getMetadataCategoryName();
087            final String categoryDescription = page1.getCategoryDescription();
088            final MetadataCategoryType categoryType = page1.getMetadataCategoryType();
089            final MetadataCategoryUse categoryUse = page1.getMetadataCategoryUse();
090            final boolean inextensible = page1.getIsInextensible();
091    
092            // Persist the new metadata category.
093            final IMetadataCategoryService metadataCategoryService = (IMetadataCategoryService) PlatformUI.getWorkbench()
094                    .getService(IMetadataCategoryService.class);
095            final IGroupService groupService = (IGroupService) PlatformUI.getWorkbench().getService(IGroupService.class);
096            final IResearchFileService fileService = (IResearchFileService) PlatformUI.getWorkbench().getService(
097                    IResearchFileService.class);
098            final ISchemaService schemaService = (ISchemaService) PlatformUI.getWorkbench()
099                    .getService(ISchemaService.class);
100    
101            for (final ResearchFile assignedFile : page1.getAssignedFiles())
102            {
103                fileService.disassociateMultipleMetadataValues(assignedFile, metadataCategory,
104                        page1.getMetadataValuesToBeDisassociated());
105            }
106    
107            for (final Group assignedGroup : page1.getAssignedGroups())
108            {
109                groupService.disassociateMultipleMetadataValues(assignedGroup, metadataCategory,
110                        page1.getMetadataValuesToBeDisassociated());
111            }
112    
113            for (final MetadataAttributeValue mdav : page1.getMetadataAttributeValuesToBeDisassociated())
114            {
115                groupService.disassociateMetadataAttributeValue(metadataCategory, mdav);
116                fileService.disassociateMetadataAttributeValue(metadataCategory, mdav);
117            }
118    
119            if (this.metadataCategory == null)
120            {
121                final List<MetadataValue> metadataValues;
122                final MetadataAttribute metadataAttribute;
123                if (categoryType == MetadataCategoryType.CONTROLLED_VOCABULARY)
124                {
125                    metadataValues = page1.getMetadataCategoryValues();
126                    metadataAttribute = null;
127                }
128                else
129                {
130                    metadataValues = Collections.emptyList();
131                    final String metadataAttributeName = page1.getMetadataAttributeName();
132                    if (!metadataAttributeName.isEmpty())
133                    {
134                        metadataAttribute = new MetadataAttribute(metadataAttributeName, page1.getMetadataAttributeValues());
135                    }
136                    else
137                    {
138                        metadataAttribute = null;
139                    }
140                }
141                final MetadataCategory newCategory = metadataCategoryService.createNewMetadataCategory(categoryTitle,
142                        categoryDescription, categoryType, categoryUse, inextensible, false, metadataValues,
143                        metadataAttribute);
144                schemaService.addMetadataCategoryToSchema(this.project.getSchema(), newCategory);
145            }
146            else
147            {
148                final MetadataAttribute oldMDA = this.metadataCategory.getMetadataAttribute();
149                final MetadataAttribute newMDA;
150    
151                if (oldMDA != null)
152                {
153                    final String attrName = page1.getMetadataAttributeName();
154                    if (!attrName.isEmpty())
155                    {
156                        newMDA = oldMDA;
157                        newMDA.setName(page1.getMetadataAttributeName());
158                        newMDA.setMetadataAttributeValues(page1.getMetadataAttributeValues());
159                    }
160                    else
161                    {
162                        newMDA = null;
163                    }
164                }
165                else
166                {
167                    final String attrName = page1.getMetadataAttributeName();
168                    if (!attrName.isEmpty())
169                    {
170                        newMDA = new MetadataAttribute(attrName, page1.getMetadataAttributeValues());
171                    }
172                    else
173                    {
174                        newMDA = null;
175                    }
176                }
177    
178                metadataCategoryService.updateMetadataCategory(this.metadataCategory, categoryTitle, categoryDescription,
179                        categoryUse, inextensible, page1.getMetadataCategoryValues(), newMDA);
180            }
181    
182            return this.project != null;
183        }
184    
185        public Project getProjectWithUpdatedMetadataCategories()
186        {
187            return this.project;
188        }
189    
190    }
