007    package au.org.intersect.exsite9.wizard.createsubmissionpackage;
008    
009    import java.util.Collection;
010    import java.util.List;
011    
012    import org.eclipse.jface.wizard.Wizard;
013    import org.eclipse.ui.PlatformUI;
014    
015    import au.org.intersect.exsite9.domain.Project;
016    import au.org.intersect.exsite9.domain.ResearchFile;
017    import au.org.intersect.exsite9.domain.SubmissionPackage;
018    import au.org.intersect.exsite9.service.IProjectManager;
019    import au.org.intersect.exsite9.service.ISubmissionPackageService;
020    
021    /**
022     * A wizard that can be used to create a submission package.
023     */
024    public final class CreateSubmissionPackageWizard extends Wizard
025    {
026        private final CreateSubmissionPackageWizardPage1 page1;
027        private final CreateSubmissionPackageWizardPage2 page2;
028    
029        private SubmissionPackage submissionPackage;
030    
031        /**
032         * Constructor
033         * @param submissionPackage The submission package to fill in the dialogs with. May be {@code null} in which case they will be blank.
034         */
035        public CreateSubmissionPackageWizard(final SubmissionPackage submissionPackage, final Collection<SubmissionPackage> existingSubmissionPackages, final String wizardTitle)
036        {
037            super();
038            setNeedsProgressMonitor(true);
039            this.submissionPackage = submissionPackage;
040            this.page1 = new CreateSubmissionPackageWizardPage1(submissionPackage, existingSubmissionPackages, wizardTitle);
041            this.page2 = new CreateSubmissionPackageWizardPage2(submissionPackage, wizardTitle);
042        }
043    
044        /**
045         * @{inheritDoc}
046         */
047        @Override
048        public void addPages()
049        {
050            addPage(this.page1);
051            addPage(this.page2);
052        }
053    
054        /**
055         * @{inheritDoc}
056         */
057        @Override
058        public boolean performFinish()
059        {
060            final String name = page1.getSubmissionPackageName();
061            final String description = page1.getSubmissionPackageDescription();
062            final List<ResearchFile> researchFiles = page2.getCheckedResearchFiles();
063    
064            final IProjectManager projectManager = (IProjectManager) PlatformUI.getWorkbench().getService(IProjectManager.class);
065            final Project currentProject = projectManager.getCurrentProject();
066    
067            final ISubmissionPackageService submissionPackageService = (ISubmissionPackageService) PlatformUI.getWorkbench().getService(ISubmissionPackageService.class);
068    
069            if (this.submissionPackage == null)
070            {
071                this.submissionPackage = submissionPackageService.createSubmissionPackage(currentProject, name, description, researchFiles);
072            }
073            else
074            {
075                this.submissionPackage = submissionPackageService.updateSubmissionPackage(this.submissionPackage, name, description, researchFiles);
076            }
077    
078            return this.submissionPackage != null;
079        }
080    
081        public SubmissionPackage getSubmissionPackage()
082        {
083            return this.submissionPackage;
084        }
085    }
