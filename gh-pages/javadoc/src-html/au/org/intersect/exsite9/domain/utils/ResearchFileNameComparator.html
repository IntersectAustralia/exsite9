007    package au.org.intersect.exsite9.domain.utils;
008    
009    import java.util.Comparator;
010    
011    import au.org.intersect.exsite9.domain.ResearchFile;
012    
013    /**
014     * A Comparator for {@link ResearchFile}s.
015     * Compares them by their name ignoring case alphabetically ascending or descending.
016     */
017    public class ResearchFileNameComparator implements Comparator<ResearchFile>
018    {
019        protected final int factor;
020    
021        /**
022         * @param ascending {@code true} to compare ascending. {@code false} to compare descending.
023         */
024        public ResearchFileNameComparator(final boolean ascending)
025        {
026            this.factor = ascending ? 1 : -1;
027        }
028    
029        @Override
030        public int compare(final ResearchFile rf1, final ResearchFile rf2)
031        {
032            final String name1 = rf1.getFile().getName();
033            final String name2 = rf2.getFile().getName();
034    
035            final int compareIgnoreCase = name1.compareToIgnoreCase(name2);
036    
037            if (compareIgnoreCase != 0)
038            {
039                return compareIgnoreCase * this.factor;
040            }
041            return 0;
042        }
043    }
