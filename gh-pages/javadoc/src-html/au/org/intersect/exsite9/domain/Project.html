165        public Project(final ProjectFieldsDTO projectFields)
166        {
167            this.name = projectFields.getName();
168            this.owner = projectFields.getOwner();
169            this.institution = projectFields.getInstitution();
170            this.email = projectFields.getEmail();
171            this.description = projectFields.getDescription();
172            this.collectionType = projectFields.getCollectionType();
173            this.rightsStatement = projectFields.getRightsStatement();
174            this.accessRights = projectFields.getAccessRights();
175            this.licence = projectFields.getLicence();
176            this.identifier = projectFields.getIdentifier();
177            this.subject = projectFields.getSubject();
178            this.electronicLocation = projectFields.getElectronicLocation();
179            this.physicalLocation = projectFields.getPhysicalLocation();
180            this.placeOrRegionName = projectFields.getPlaceOrRegionName();
181            this.geographicalCoverage = projectFields.getGeographicalCoverage();
182            this.datesOfCapture = projectFields.getDatesOfCapture();
183            this.citationInformation = projectFields.getCitationInformation();
184            this.countries = projectFields.getCountries();
185            this.languages = projectFields.getLanguages();
186            this.fieldOfResearch = projectFields.getFieldOfResearch();
187            this.fundingBody = projectFields.getFundingBody();
188            this.grantID = projectFields.getGrantID();
189            this.relatedParty = projectFields.getRelatedParty();
190            this.relatedGrant = projectFields.getRelatedGrant();
191            this.relatedInformation = projectFields.getRelatedInformation();
192            this.folders = new ArrayList<Folder>(0);
193            this.rootNode = new RootGroup(this.name);
194            this.newFilesNode = new NewFilesGroup();
195            this.excludedFilesNode = new ExcludedFilesGroup();
196            this.rootNode.getGroups().add(newFilesNode);
197        }
