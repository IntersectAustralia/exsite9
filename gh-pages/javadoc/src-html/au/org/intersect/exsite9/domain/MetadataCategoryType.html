007    package au.org.intersect.exsite9.domain;
008    
009    /**
010     * A Metadata Category type.
011     */
012    public enum MetadataCategoryType
013    {
014        /**
015         * A Metadata Category type that holds controlled metadata value fields.
016         */
017        CONTROLLED_VOCABULARY("Controlled Vocabulary"),
018    
019        /**
020         * A Metadata Category type that holds a free-text metadata value.
021         */
022        FREETEXT("Free Text");
023    
024        private final String name;
025    
026        private MetadataCategoryType(final String name)
027        {
028            this.name = name;
029        }
030    
031        @Override
032        public String toString()
033        {
034            return this.name;
035        }
036    
037        public static String[] toArray()
038        {
039            final String[] toReturn = new String[values().length];
040            int i = 0;
041            for (final MetadataCategoryType type : values())
042            {
043                toReturn[i] = type.toString();
044                i++;
045            }
046            return toReturn;
047        }
048    
049        public static MetadataCategoryType fromString(final String value)
050        {
051            for (final MetadataCategoryType type : values())
052            {
053                if (type.toString().equals(value))
054                {
055                    return type;
056                }
057            }
058            return null;
059        }
060    }
