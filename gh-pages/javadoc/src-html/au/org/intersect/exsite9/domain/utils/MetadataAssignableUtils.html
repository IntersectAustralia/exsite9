007    package au.org.intersect.exsite9.domain.utils;
008    
009    import java.util.HashSet;
010    import java.util.List;
011    import java.util.Set;
012    
013    import au.org.intersect.exsite9.domain.IMetadataAssignable;
014    import au.org.intersect.exsite9.domain.MetadataAssociation;
015    import au.org.intersect.exsite9.domain.MetadataAttributeValue;
016    import au.org.intersect.exsite9.domain.MetadataCategory;
017    import au.org.intersect.exsite9.domain.MetadataValue;
018    import au.org.intersect.exsite9.util.Triplet;
019    
020    /**
021     * Utilities for {@link IMetadataAssignable}
022     */
023    public final class MetadataAssignableUtils
024    {
025        private MetadataAssignableUtils()
026        {
027            // No instances please.
028        }
029    
030        /**
031         * Obtains the set of metadata category to metadata value mappings for the provided group by iterating through its metadata associations.
032         * @param group The group.
033         * @return the set of metadata category to metadata value mappings.
034         */
035        public static Set<Triplet<MetadataCategory, MetadataValue, MetadataAttributeValue>> getCategoryToValueMapping(final IMetadataAssignable metadataAssignable)
036        {
037            final Set<Triplet<MetadataCategory, MetadataValue, MetadataAttributeValue>> toReturn = new HashSet<Triplet<MetadataCategory,MetadataValue, MetadataAttributeValue>>();
038    
039            final List<MetadataAssociation> metadataAssociations = metadataAssignable.getMetadataAssociations();
040            for (final MetadataAssociation metadataAssociation : metadataAssociations)
041            {
042                final MetadataCategory metadataCategory = metadataAssociation.getMetadataCategory();
043                final MetadataAttributeValue metadataAttributeValue = metadataAssociation.getMetadataAttributeValue();
044                for (final MetadataValue metadataValue : metadataAssociation.getMetadataValues())
045                {
046                    toReturn.add(new Triplet<MetadataCategory, MetadataValue, MetadataAttributeValue>(metadataCategory, metadataValue, metadataAttributeValue));
047                }
048            }
049            return toReturn;
050        }
051    }
