007    package au.org.intersect.exsite9.domain.utils;
008    
009    import java.util.Comparator;
010    
011    import au.org.intersect.exsite9.domain.MetadataAttributeValue;
012    
013    /**
014     * Comparator used to sort {@link MetadataAttributeValue}s alphabetically.
015     */
016    public final class AlphabeticalMetadataAttributeValueComparator implements Comparator<MetadataAttributeValue>
017    {
018    
019        /**
020         * @{inheritDoc}
021         */
022        @Override
023        public int compare(final MetadataAttributeValue mav1, final MetadataAttributeValue mav2)
024        {
025            final String val1 = mav1.getValue();
026            final String val2 = mav2.getValue();
027            return val1.compareToIgnoreCase(val2);
028        }
029    }
