007    package au.org.intersect.exsite9.domain;
008    
009    import java.io.File;
010    import java.io.Serializable;
011    import java.text.DecimalFormat;
012    import java.text.SimpleDateFormat;
013    import java.util.ArrayList;
014    import java.util.Date;
015    import java.util.List;
016    
017    import javax.persistence.Column;
018    import javax.persistence.Entity;
019    import javax.persistence.GeneratedValue;
020    import javax.persistence.GenerationType;
021    import javax.persistence.Id;
022    import javax.persistence.ManyToOne;
023    import javax.persistence.OneToMany;
024    import javax.persistence.Table;
025    import javax.persistence.UniqueConstraint;
026    
027    import org.apache.commons.lang3.builder.ToStringBuilder;
028    import org.eclipse.persistence.annotations.Convert;
029    import org.eclipse.persistence.annotations.Converter;
030    
031    import com.google.common.base.Objects;
032    
033    /**
034     * Represents a research data file in a folder that the researcher has associated with a project.
035     */
036    @Entity
037    @Table(name="RESEARCH_FILE", uniqueConstraints=@UniqueConstraint(columnNames={"PROJECT_ID","FILE"}))
038    @Converter(name="fileToStringConverter",
039        converterClass=au.org.intersect.exsite9.domain.utils.FileToStringConverter.class)
040    public final class ResearchFile implements Serializable, IMetadataAssignable
041    {
042    
043        private static final long serialVersionUID = 5047390909436811919L;
044    
045        @Id
046        @GeneratedValue(strategy=GenerationType.IDENTITY)
047        private Long id;
048    
049        @Column(columnDefinition = "longvarchar")
050        @Convert("fileToStringConverter")
051        private File file;
052    
053        @ManyToOne
054        private Project project;
055    
056        @ManyToOne
057        private Group parentGroup;
058        
059        @OneToMany
060        private final List<MetadataAssociation> metadataAssociations = new ArrayList<MetadataAssociation>();
061    
062        public ResearchFile()
063        {
064        }
065    
066        public ResearchFile(final File fileOnDisk)
067        {
068            this.file = fileOnDisk;
069        }
070    
071        public Long getId()
072        {
073            return this.id;
074        }
075        
076        public void setId(Long id)
077        {
078            this.id = id;
079        }
080    
081        public Project getProject()
082        {
083            return project;
084        }
085    
086        public void setProject(final Project project)
087        {
088            this.project = project;
089        }
090    
091        public File getFile()
092        {
093            return this.file;
094        }
095    
096        public void setFile(final File file)
097        {
098            this.file = file;
099        }
100    
101        public void setParentGroup(final Group parentGroup)
102        {
103            this.parentGroup = parentGroup;
104        }
105    
106        public Group getParentGroup()
107        {
108            return this.parentGroup;
109        }
110    
111        /**
112         * {@inheritDoc}
113         */
114        @Override
115        public boolean equals(final Object obj)
116        {
117            if (obj == this)
118            {
119                return true;
120            }
121            if (!(obj instanceof ResearchFile))
122            {
123                return false;
124            }
125            final ResearchFile other = (ResearchFile) obj;
126            return Objects.equal(this.id, other.id);
127        }
128    
129        /**
130         * {@inheritDoc}
131         */
132        @Override
133        public int hashCode()
134        {
135            return Objects.hashCode(this.id);
136        }
137    
138        /**
139         * {@inheritDoc}
140         */
141        @Override
142        public String toString()
143        {
144            return new ToStringBuilder(this).append("id", this.id).append("file", this.file).toString();
145        }
146        
147        public List<MetadataAssociation> getMetadataAssociations()
148        {
149            return this.metadataAssociations;
150        }
151        
152        public boolean isMissingRequiredMetadata()
153        {
154            boolean missing = false;
155            
156            category: 
157            for(final MetadataCategory category : getProject().getSchema().getRequiredMetadataCategories())
158            {
159                for(final MetadataAssociation association : getMetadataAssociations())
160                {
161                    if(association.getMetadataCategory().equals(category))
162                    {
163                        continue category;
164                    }
165                }
166                missing = true;
167                break;
168            }
169            
170            return missing;
171        }
172        
173        public String getDisplayDate()
174        {
175            String displayDate = "";
176            if (this.file != null)
177            {
178                try{
179                    SimpleDateFormat sdf = new SimpleDateFormat("dd/MMM/yyyy hh:mm aa");
180                    displayDate = sdf.format(new Date(this.file.lastModified()));
181                }
182                catch(Exception e){
183                    // continue
184                }
185            }
186            return displayDate;
187        }
188        
189        public String getDisplaySize()
190        {
191            String displaySize = "0 B";
192            if(this.file != null){
193                try{
194                    long size = this.file.length();
195                    final String[] units = new String[] { "B", "KB", "MB", "GB", "TB" };
196                    int digitGroups = (int) (Math.log10(size)/Math.log10(1024));
197                    displaySize =  new DecimalFormat("#,##0.#").format(size/Math.pow(1024, digitGroups)) + " " + units[digitGroups];
198                }
199                catch(Exception e){
200                    // continue
201                }
202            }
203            return displaySize;
204        }
205    
206    }
