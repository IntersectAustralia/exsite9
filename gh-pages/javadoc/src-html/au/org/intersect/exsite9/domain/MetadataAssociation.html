007    package au.org.intersect.exsite9.domain;
008    
009    import java.io.Serializable;
010    import java.util.ArrayList;
011    import java.util.List;
012    
013    import javax.persistence.Entity;
014    import javax.persistence.GeneratedValue;
015    import javax.persistence.GenerationType;
016    import javax.persistence.Id;
017    import javax.persistence.OneToMany;
018    import javax.persistence.Table;
019    
020    import org.apache.commons.lang3.builder.HashCodeBuilder;
021    import org.apache.commons.lang3.builder.ToStringBuilder;
022    
023    import com.google.common.base.Objects;
024    
025    /**
026     * An association between a metadata category and it's values that have been assigned.
027     * Used when adding metadata to {@link ResearchFile}s.
028     */
029    @Entity
030    @Table(name="METADATA_ASSOCIATION")
031    public final class MetadataAssociation implements Serializable
032    {
033        private static final long serialVersionUID = 2152083468103054575L;
034    
035        @Id
036        @GeneratedValue(strategy=GenerationType.IDENTITY)
037        private Long id;
038    
039        private MetadataCategory metadataCategory;
040    
041        private MetadataAttributeValue metadataAttributeValue;
042    
043        @OneToMany
044        private final List<MetadataValue> metadataValues = new ArrayList<MetadataValue>();
045    
046        public MetadataAssociation()
047        {
048        }
049    
050        public MetadataAssociation(final MetadataCategory metadataCategory)
051        {
052            this.metadataCategory = metadataCategory;
053        }
054    
055        public Long getId()
056        {
057            return id;
058        }
059    
060        public void setId(final Long id)
061        {
062            this.id = id;
063        }
064    
065        public MetadataCategory getMetadataCategory()
066        {
067            return metadataCategory;
068        }
069    
070        public void setMetadataCategory(final MetadataCategory metadataCategory)
071        {
072            this.metadataCategory = metadataCategory;
073        }
074    
075        public List<MetadataValue> getMetadataValues()
076        {
077            return metadataValues;
078        }
079    
080        public void setMetadataAttributeValue(final MetadataAttributeValue metadataAttributeValue)
081        {
082            this.metadataAttributeValue = metadataAttributeValue;
083        }
084    
085        public MetadataAttributeValue getMetadataAttributeValue()
086        {
087            return this.metadataAttributeValue;
088        }
089    
090        @Override
091        public int hashCode()
092        {
093            return new HashCodeBuilder().append(this.id).toHashCode();
094        }
095    
096        @Override
097        public boolean equals(final Object obj)
098        {
099            if (this == obj)
100            {
101                return true;
102            }
103            if (!(obj instanceof MetadataAssociation))
104            {
105                return false;
106            }
107            final MetadataAssociation other = (MetadataAssociation) obj;
108            return Objects.equal(this.id, other.id);
109        }
110    
111        @Override
112        public String toString()
113        {
114            return new ToStringBuilder(this).append("id", this.id).append("metadataCategory", this.metadataCategory).toString();
115        }
116    }
