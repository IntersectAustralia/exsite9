007    package au.org.intersect.exsite9.domain;
008    
009    import java.io.Serializable;
010    import java.util.ArrayList;
011    import java.util.List;
012    
013    import javax.persistence.CascadeType;
014    import javax.persistence.Column;
015    import javax.persistence.Entity;
016    import javax.persistence.GeneratedValue;
017    import javax.persistence.GenerationType;
018    import javax.persistence.Id;
019    import javax.persistence.OneToMany;
020    import javax.persistence.OneToOne;
021    import javax.persistence.Table;
022    
023    import org.apache.commons.lang3.builder.HashCodeBuilder;
024    import org.apache.commons.lang3.builder.ToStringBuilder;
025    
026    import com.google.common.base.Objects;
027    
028    /**
029     * A metadata category.
030     */
031    @Entity
032    @Table(name="METADATA_CATEGORY")
033    public final class MetadataCategory implements Serializable
034    {
035        private static final long serialVersionUID = -6662974846879710713L;
036    
037        @Id
038        @GeneratedValue(strategy=GenerationType.IDENTITY)
039        private Long id;
040    
041        @Column(length = 255)
042        private String name;
043        
044        @Column(columnDefinition = "longvarchar")
045        private String description;
046        
047        private MetadataCategoryUse use;
048    
049        private MetadataCategoryType type;
050        
051        private boolean isInextensible;
052        
053        private boolean imported;
054    
055        @OneToMany(cascade=CascadeType.ALL, orphanRemoval=true)
056        private List<MetadataValue> metadataValues = new ArrayList<MetadataValue>();
057    
058        @OneToOne
059        private MetadataAttribute metadataAttribute;
060    
061        public MetadataCategory()
062        {
063        }
064    
065        public MetadataCategory(final String name, final MetadataCategoryType type, final MetadataCategoryUse use)
066        {
067            this.name = name;
068            this.use = use;
069            this.type = type;
070        }
071        
072        public Long getId()
073        {
074            return this.id;
075        }
076    
077        public void setId(final Long id)
078        {
079            this.id = id;
080        }
081    
082        public String getName()
083        {
084            return this.name;
085        }
086    
087        public void setName(final String name)
088        {
089            this.name = name;
090        }
091    
092        public String getDescription()
093        {
094            return description;
095        }
096    
097        public void setDescription(String description)
098        {
099            this.description = description;
100        }
101    
102        public List<MetadataValue> getValues()
103        {
104            return this.metadataValues;
105        }
106        
107        public void setValues(List<MetadataValue> values)
108        {
109            this.metadataValues = values;
110        }
111    
112        public MetadataCategoryUse getUse()
113        {
114            return use;
115        }
116    
117        public void setUse(MetadataCategoryUse use)
118        {
119            this.use = use;
120        }
121    
122        public MetadataCategoryType getType()
123        {
124            return this.type;
125        }
126    
127        public void setType(final MetadataCategoryType metadataCategoryType)
128        {
129            this.type = metadataCategoryType;
130        }
131    
132        public boolean isInextensible()
133        {
134            return isInextensible;
135        }
136        
137        public void setInextensible(boolean isInextensible)
138        {
139            this.isInextensible = isInextensible;
140        }
141    
142        public MetadataAttribute getMetadataAttribute()
143        {
144            return this.metadataAttribute;
145        }
146    
147        public void setMetadataAttribute(final MetadataAttribute metadataAttribute)
148        {
149            this.metadataAttribute = metadataAttribute;
150        }
151        
152        public boolean isImported()
153        {
154            return imported;
155        }
156        
157        public void setImported(boolean imported)
158        {
159            this.imported = imported;
160        }
161        
162        @Override
163        public int hashCode()
164        {
165            return new HashCodeBuilder().append(this.id).toHashCode();
166        }
167    
168        @Override
169        public boolean equals(final Object obj)
170        {
171            if (this == obj)
172            {
173                return true;
174            }
175            if (!(obj instanceof MetadataCategory))
176            {
177                return false;
178            }
179            final MetadataCategory other = (MetadataCategory) obj;
180            return Objects.equal(this.id, other.id);
181        }
182    
183        @Override
184        public String toString()
185        {
186            return new ToStringBuilder(this).append("id", this.id).append("name", this.name).append("type", this.type).append("use",this.use).toString();
187        }
188    }
