007    package au.org.intersect.exsite9.domain.utils;
008    
009    import java.util.Comparator;
010    
011    import au.org.intersect.exsite9.domain.MetadataAttributeValue;
012    import au.org.intersect.exsite9.domain.MetadataCategory;
013    import au.org.intersect.exsite9.domain.MetadataValue;
014    import au.org.intersect.exsite9.util.Triplet;
015    
016    /**
017     * Comparator that can be used to sort MetadataAssociation Triplets
018     */
019    public final class MetadataAssociationTripletComparator implements Comparator<Triplet<MetadataCategory, MetadataValue, MetadataAttributeValue>>
020    {
021    
022        @Override
023        public int compare(final Triplet<MetadataCategory, MetadataValue, MetadataAttributeValue> t1, final Triplet<MetadataCategory, MetadataValue, MetadataAttributeValue> t2)
024        {
025            final long mcId1 = t1.getFirst().getId();
026            final long mcId2 = t2.getFirst().getId();
027    
028            if (mcId1 < mcId2)
029            {
030                return -1;
031            }
032            if (mcId1 > mcId2)
033            {
034                return 1;
035            }
036    
037            final long mvId1 = t1.getSecond().getId();
038            final long mvId2 = t2.getSecond().getId();
039    
040            if (mvId1 < mvId2)
041            {
042                return -1;
043            }
044            if (mvId1 > mvId2)
045            {
046                return 1;
047            }
048    
049            // MetadataAttributeValues can be null.
050            if (t1.getThird() == null && t2.getThird() == null)
051            {
052                return 0;
053            }
054            if (t1.getThird() == null)
055            {
056                return 1;
057            }
058            if (t2.getThird() == null)
059            {
060                return -1;
061            }
062            
063            final long mavId1 = t1.getThird().getId();
064            final long mavId2 = t2.getThird().getId();
065    
066            if (mavId1 < mavId2)
067            {
068                return -1;
069            }
070            if (mavId1 > mavId2)
071            {
072                return 1;
073            }
074    
075            return 0;
076        }
077    }
