007    package au.org.intersect.exsite9.domain;
008    
009    import java.util.Collections;
010    import java.util.List;
011    
012    import javax.persistence.Entity;
013    
014    /**
015     * This group is a holder for files that have been excluded from the project.
016     */
017    @Entity
018    public final class ExcludedFilesGroup extends Group
019    {
020        private static final long serialVersionUID = -5078733345256805564L;
021    
022        public ExcludedFilesGroup()
023        {
024            super("Excluded Files");
025        }
026    
027        /**
028         * You can never add groups to the ExcludedFilesGroup.
029         * @{inheritDoc}
030         */
031        @Override
032        public List<Group> getGroups()
033        {
034            return Collections.unmodifiableList(Collections.<Group>emptyList());
035        }
036    
037        /**
038         * You can never add metadata associations to the ExcludedFilesGroup.
039         * @{inheritDoc}
040         */
041        @Override
042        public List<MetadataAssociation> getMetadataAssociations()
043        {
044            return Collections.unmodifiableList(Collections.<MetadataAssociation>emptyList());
045        }
046    }
