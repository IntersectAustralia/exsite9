001    package au.org.intersect.exsite9.domain;
002    
003    import java.util.Collections;
004    import java.util.List;
005    
006    import javax.persistence.Entity;
007    
008    /**
009     * The Root group - every project has one.
010     */
011    @Entity
012    public final class RootGroup extends Group
013    {
014        private static final long serialVersionUID = -7294236885579971700L;
015        
016        public RootGroup()
017        {
018            super();
019        }
020    
021        public RootGroup(final String projectName)
022        {
023            super(projectName);
024        }
025        
026    
027        /**
028         * You can never add metadata associations to the RootGroup.
029         * @{inheritDoc}
030         */
031        @Override
032        public List<MetadataAssociation> getMetadataAssociations()
033        {
034            return Collections.unmodifiableList(Collections.<MetadataAssociation>emptyList());
035        }
036    
037    }
