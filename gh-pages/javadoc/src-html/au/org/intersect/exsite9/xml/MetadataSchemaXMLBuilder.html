001    package au.org.intersect.exsite9.xml;
002    
003    import javax.xml.parsers.ParserConfigurationException;
004    import javax.xml.transform.TransformerConfigurationException;
005    import javax.xml.transform.TransformerException;
006    
007    import org.apache.log4j.Logger;
008    import org.w3c.dom.Document;
009    import org.w3c.dom.Element;
010    
011    import au.org.intersect.exsite9.domain.MetadataAttribute;
012    import au.org.intersect.exsite9.domain.MetadataAttributeValue;
013    import au.org.intersect.exsite9.domain.MetadataCategory;
014    import au.org.intersect.exsite9.domain.MetadataCategoryType;
015    import au.org.intersect.exsite9.domain.MetadataValue;
016    import au.org.intersect.exsite9.domain.Project;
017    import au.org.intersect.exsite9.domain.Schema;
018    
019    import static au.org.intersect.exsite9.xml.MetadataSchemaXMLUtils.*;
020    
021    public final class MetadataSchemaXMLBuilder extends BaseXMLBuilder
022    {
023        private static final Logger LOG = Logger.getLogger(MetadataSchemaXMLBuilder.class);
024        
025        public static String buildXML(Project project)
026        {
027            try
028            {
029                if (project.getSchema() == null)
030                {
031                    return null;
032                }
033                
034                final Document doc = createNewDocument();
035                
036                appendSchema(doc, project);
037                
038                appendMetadataCategories(doc, project);
039                
040                return transformDocumentToString(doc);
041            }
042            catch(ParserConfigurationException pce)
043            {
044                LOG.error(pce);
045            }
046            catch (final TransformerConfigurationException tce)
047            {
048                LOG.error(tce);
049            }
050            catch (final TransformerException te)
051            {
052                LOG.error(te);
053            }
054            return null;
055        }
056        
057        private static void appendSchema(Document doc, Project project)
058        {
059            final Schema schema = project.getSchema();
060            final Element schemaElement = doc.createElement(ELEMENT_SCHEMA);
061            schemaElement.setAttribute(ATTRIBUTE_DESCRIPTION, schema.getDescription());
062            schemaElement.setAttribute(ATTRIBUTE_NAME, schema.getName());
063            schemaElement.setAttribute(ATTRIBUTE_NAMESPACEURL, schema.getNamespaceURL());
064            doc.appendChild(schemaElement);
065        }
066        
067        private static void appendMetadataCategories(Document doc, Project project)
068        {
069            for(final MetadataCategory category : project.getSchema().getMetadataCategories())
070            {
071                final Element catElement = doc.createElement(ELEMENT_METADATACATEGORY);
072                catElement.setAttribute(ATTRIBUTE_NAME, category.getName());
073                catElement.setAttribute(ATTRIBUTE_USE, category.getUse().toString());
074    
075                final MetadataCategoryType type = category.getType();
076                catElement.setAttribute(ATTRIBUTE_TYPE, type.toString());
077                catElement.setAttribute(ATTRIBUTE_INEXTENSIBLE, ((Boolean)category.isInextensible()).toString());
078                
079                if( (category.getDescription() != null) && (! category.getDescription().isEmpty())){
080                    final Element descElement = doc.createElement(ELEMENT_DESCRIPTION);
081                    descElement.setTextContent(category.getDescription());
082                    catElement.appendChild(descElement);
083                }
084                
085                if (type != MetadataCategoryType.FREETEXT)
086                {
087                    for (final MetadataValue value : category.getValues())
088                    {
089                        final Element valElement = doc.createElement(ELEMENT_VALUE);
090                        valElement.setTextContent(value.getValue());
091                        catElement.appendChild(valElement);
092                    }
093                }
094                else
095                {
096                    final MetadataAttribute metadataAttribute = category.getMetadataAttribute();
097                    if (metadataAttribute != null)
098                    {
099                        final Element attributeElement = doc.createElement(ELEMENT_ATTRIBUTE);
100                        attributeElement.setAttribute(ATTRIBUTE_NAME, metadataAttribute.getName());
101                        for (final MetadataAttributeValue attributeValue : metadataAttribute.getMetadataAttributeValues())
102                        {
103                            final Element valueElement = doc.createElement(ELEMENT_VALUE);
104                            valueElement.setTextContent(attributeValue.getValue());
105                            attributeElement.appendChild(valueElement);
106                        }
107                        catElement.appendChild(attributeElement);
108                    }
109                }
110                doc.getFirstChild().appendChild(catElement);
111            }
112        }
113    }
