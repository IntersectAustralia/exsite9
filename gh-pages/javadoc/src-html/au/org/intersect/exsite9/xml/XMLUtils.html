007    package au.org.intersect.exsite9.xml;
008    
009    /**
010     * Utilities used by exsite9 XML builders and parsers
011     */
012    public class XMLUtils
013    {
014    
015        /**
016         * Checks that string is a valid xml element name
017         * 
018         *   Names can contain letters, numbers and other characters
019         *   Names cannot start with a number or punctuation character
020         *   Names cannot start with the letters xml (case insensitive)
021         *   Names cannot contain spaces
022         * 
023         * @param name The string to check
024         * @return True if valid
025         */
026        public static boolean isValidElementOrAttribute(String name)
027        {
028            
029            // Can't start with punctuation or a number
030            if ( ! name.matches("^[a-zA-Z].*"))
031            {
032                return false;
033            }
034            
035            // Can't start with xml (in any case)
036            if (name.toLowerCase().startsWith("xml"))
037            {
038                return false;
039            }
040            
041            // Can't contain spaces, ampersands, greater than or less than, fwd or back slash
042            // semi-colon, asterisk, open and close (), open and close {}, open and close []
043            // plus sign, single quote, double quote, question mark, exclamation mark, back tick
044            // tilde, at sign, dollar sign, percent sign, caret, equals, comma
045            return !name.matches("^[a-zA-Z].*[\\s&<>/\\\\|;\\*\\(\\)\\{\\}\\[\\]\\+'\"\\?\\!`~@\\$\\%\\^=,].*");
046        }
047    }
