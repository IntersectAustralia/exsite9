001    package au.org.intersect.exsite9.xml;
002    
003    import java.util.List;
004    
005    import javax.xml.parsers.ParserConfigurationException;
006    import javax.xml.transform.TransformerConfigurationException;
007    import javax.xml.transform.TransformerException;
008    
009    import org.apache.log4j.Logger;
010    import org.w3c.dom.Document;
011    import org.w3c.dom.Element;
012    
013    import au.org.intersect.exsite9.domain.Group;
014    import au.org.intersect.exsite9.domain.MetadataAssociation;
015    import au.org.intersect.exsite9.domain.Project;
016    import au.org.intersect.exsite9.domain.ResearchFile;
017    import au.org.intersect.exsite9.domain.SubmissionPackage;
018    
019    public class SIPXMLBuilder extends BaseXMLBuilder
020    {
021        private static final Logger LOG = Logger.getLogger(SIPXMLBuilder.class);
022    
023        /**
024         * Builds XML for a SIP for the provided SubmissionPackage. 
025         * @param project The Project the submission package is contained within.
026         * @param selectedGroups The Groups that contain files are in the SubmissionPackage.
027         * @param submissionPackage The submission package.
028         * @param useGroupPaths {@code true} to use the path of a file relative to its group. {@code false} to use the absolute path of a file.
029         * @return The XML.
030         */
031        public static String buildXML(final Project project, final List<Group> selectedGroups, final SubmissionPackage submissionPackage, final boolean useGroupPaths)
032        {
033            final List<ResearchFile> selectedFiles = submissionPackage.getResearchFiles();
034            try
035            {
036                final Document doc = createNewDocument();
037    
038                final Element rootElement = doc.createElement("project");
039                doc.appendChild(rootElement);
040    
041                final Element projectInfoElement = createProjectInfo(doc, project);
042                rootElement.appendChild(projectInfoElement);
043    
044                final Element groupsElement = doc.createElement("groups");
045                int numGroups = 0;
046    
047                for (final Group group : project.getRootNode().getGroups())
048                {
049                    if (selectedGroups.contains(group))
050                    {
051                        appendGroup(doc, selectedGroups, selectedFiles, groupsElement, group, useGroupPaths, "");
052                        numGroups++;
053                    }
054                }
055    
056                if (numGroups > 0)
057                {
058                    groupsElement.setAttribute("numGroups", Integer.toString(numGroups));
059                    rootElement.appendChild(groupsElement);
060                }
061    
062                final Element filesElement = doc.createElement("files");
063                int numFiles = 0;
064    
065                for (final ResearchFile researchFile : project.getRootNode().getResearchFiles())
066                {
067                    if (selectedFiles.contains(researchFile))
068                    {
069                        if (useGroupPaths)
070                        {
071                            appendResearchFile(doc, filesElement, researchFile, researchFile.getFile().getName(), true);
072                        }
073                        else
074                        {
075                            appendResearchFile(doc, filesElement, researchFile, true);
076                        }
077                        numFiles++;
078                    }
079                }
080    
081                if (numFiles > 0)
082                {
083                    filesElement.setAttribute("numFiles", Integer.toString(numFiles));
084                    rootElement.appendChild(filesElement);
085                }
086    
087                return transformDocumentToString(doc);
088            }
089            catch (final ParserConfigurationException e)
090            {
091                LOG.error(e);
092            }
093            catch (final TransformerConfigurationException e)
094            {
095                LOG.error(e);
096            }
097            catch (final TransformerException e)
098            {
099                LOG.error(e);
100            }
101            return null;
102        }
103        
104        /**
105         * @param parent
106         * @param group
107         */
108        private static void appendGroup(final Document doc, final List<Group> selectedGroups, List<ResearchFile> selectedFiles, final Element parent, final Group group, final boolean useGroupPaths, final String parentGroupPath)
109        {
110            final Element groupElement = doc.createElement("group");
111            groupElement.setAttribute("name", group.getName());
112            final String groupPath = parentGroupPath + group.getName() + "/";
113            
114            for (final MetadataAssociation metadataAssociation : group.getMetadataAssociations())
115            {
116                appendMetadataAssociation(doc, groupElement, metadataAssociation);
117            }
118    
119            final Element groupsElement = doc.createElement("groups");
120            int numGroups = 0;
121    
122            for (final Group childGroup : group.getGroups())
123            {
124                if(selectedGroups.contains(childGroup))
125                {
126                    // Recursion
127                    appendGroup(doc, selectedGroups, selectedFiles, groupsElement, childGroup, useGroupPaths, groupPath);
128                    numGroups++;
129                }
130            }
131    
132            if (numGroups > 0)
133            {
134                groupsElement.setAttribute("numGroups", Integer.toString(numGroups));
135                groupElement.appendChild(groupsElement);
136            }
137    
138            final Element filesElement = doc.createElement("files");
139            int numFiles = 0;
140    
141            for (final ResearchFile childFile : group.getResearchFiles())
142            {
143                if(selectedFiles.contains(childFile))
144                {
145                    if (useGroupPaths)
146                    {
147                        appendResearchFile(doc, filesElement, childFile, groupPath + childFile.getFile().getName(), true);
148                    }
149                    else
150                    {
151                        appendResearchFile(doc, filesElement, childFile, true);
152                    }
153                    numFiles++;
154                }
155            }
156    
157            if (numFiles > 0)
158            {
159                filesElement.setAttribute("numFiles", Integer.toString(numFiles));
160                groupElement.appendChild(filesElement);
161            }
162    
163            parent.appendChild(groupElement);
164        }
165    }
