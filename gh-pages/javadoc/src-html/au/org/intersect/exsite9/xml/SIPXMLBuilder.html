007    package au.org.intersect.exsite9.xml;
008    
009    import java.util.List;
010    
011    import javax.xml.parsers.ParserConfigurationException;
012    import javax.xml.transform.TransformerConfigurationException;
013    import javax.xml.transform.TransformerException;
014    
015    import org.apache.log4j.Logger;
016    import org.w3c.dom.Document;
017    import org.w3c.dom.Element;
018    
019    import au.org.intersect.exsite9.domain.Group;
020    import au.org.intersect.exsite9.domain.MetadataAssociation;
021    import au.org.intersect.exsite9.domain.Project;
022    import au.org.intersect.exsite9.domain.ResearchFile;
023    import au.org.intersect.exsite9.domain.SubmissionPackage;
024    
025    /**
026     * Creates the XML that will be contained in a {@link SubmissionPackage} 
027     */
028    public class SIPXMLBuilder extends BaseXMLBuilder
029    {
030        private static final Logger LOG = Logger.getLogger(SIPXMLBuilder.class);
031    
032        /**
033         * Builds XML for a SIP for the provided SubmissionPackage. 
034         * @param project The Project the submission package is contained within.
035         * @param selectedGroups The Groups that contain files are in the SubmissionPackage.
036         * @param submissionPackage The submission package.
037         * @param useGroupPaths {@code true} to use the path of a file relative to its group. {@code false} to use the absolute path of a file.
038         * @return The XML.
039         */
040        public static String buildXML(final Project project, final List<Group> selectedGroups, final SubmissionPackage submissionPackage, final boolean useGroupPaths)
041        {
042            final List<ResearchFile> selectedFiles = submissionPackage.getResearchFiles();
043            try
044            {
045                final Document doc = createNewDocument();
046    
047                final Element rootElement = doc.createElement("project");
048                doc.appendChild(rootElement);
049    
050                final Element projectInfoElement = createProjectInfo(doc, project);
051                rootElement.appendChild(projectInfoElement);
052    
053                final Element groupsElement = doc.createElement("groups");
054                int numGroups = 0;
055    
056                for (final Group group : project.getRootNode().getGroups())
057                {
058                    if (selectedGroups.contains(group))
059                    {
060                        appendGroup(doc, selectedGroups, selectedFiles, groupsElement, group, useGroupPaths, "");
061                        numGroups++;
062                    }
063                }
064    
065                if (numGroups > 0)
066                {
067                    groupsElement.setAttribute("numGroups", Integer.toString(numGroups));
068                    rootElement.appendChild(groupsElement);
069                }
070    
071                final Element filesElement = doc.createElement("files");
072                int numFiles = 0;
073    
074                for (final ResearchFile researchFile : project.getRootNode().getResearchFiles())
075                {
076                    if (selectedFiles.contains(researchFile))
077                    {
078                        if (useGroupPaths)
079                        {
080                            appendResearchFile(doc, filesElement, researchFile, researchFile.getFile().getName(), true);
081                        }
082                        else
083                        {
084                            appendResearchFile(doc, filesElement, researchFile, true);
085                        }
086                        numFiles++;
087                    }
088                }
089    
090                if (numFiles > 0)
091                {
092                    filesElement.setAttribute("numFiles", Integer.toString(numFiles));
093                    rootElement.appendChild(filesElement);
094                }
095    
096                return transformDocumentToString(doc);
097            }
098            catch (final ParserConfigurationException e)
099            {
100                LOG.error(e);
101            }
102            catch (final TransformerConfigurationException e)
103            {
104                LOG.error(e);
105            }
106            catch (final TransformerException e)
107            {
108                LOG.error(e);
109            }
110            return null;
111        }
112        
113        /**
114         * @param parent
115         * @param group
116         */
117        private static void appendGroup(final Document doc, final List<Group> selectedGroups, List<ResearchFile> selectedFiles, final Element parent, final Group group, final boolean useGroupPaths, final String parentGroupPath)
118        {
119            final Element groupElement = doc.createElement("group");
120            groupElement.setAttribute("name", group.getName());
121            final String groupPath = parentGroupPath + group.getName() + "/";
122            
123            for (final MetadataAssociation metadataAssociation : group.getMetadataAssociations())
124            {
125                appendMetadataAssociation(doc, groupElement, metadataAssociation);
126            }
127    
128            final Element groupsElement = doc.createElement("groups");
129            int numGroups = 0;
130    
131            for (final Group childGroup : group.getGroups())
132            {
133                if(selectedGroups.contains(childGroup))
134                {
135                    // Recursion
136                    appendGroup(doc, selectedGroups, selectedFiles, groupsElement, childGroup, useGroupPaths, groupPath);
137                    numGroups++;
138                }
139            }
140    
141            if (numGroups > 0)
142            {
143                groupsElement.setAttribute("numGroups", Integer.toString(numGroups));
144                groupElement.appendChild(groupsElement);
145            }
146    
147            final Element filesElement = doc.createElement("files");
148            int numFiles = 0;
149    
150            for (final ResearchFile childFile : group.getResearchFiles())
151            {
152                if(selectedFiles.contains(childFile))
153                {
154                    if (useGroupPaths)
155                    {
156                        appendResearchFile(doc, filesElement, childFile, groupPath + childFile.getFile().getName(), true);
157                    }
158                    else
159                    {
160                        appendResearchFile(doc, filesElement, childFile, true);
161                    }
162                    numFiles++;
163                }
164            }
165    
166            if (numFiles > 0)
167            {
168                filesElement.setAttribute("numFiles", Integer.toString(numFiles));
169                groupElement.appendChild(filesElement);
170            }
171    
172            parent.appendChild(groupElement);
173        }
174    }
