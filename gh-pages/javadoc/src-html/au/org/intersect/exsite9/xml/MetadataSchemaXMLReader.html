011    import org.w3c.dom.NodeList;
012    
013    import au.org.intersect.exsite9.domain.MetadataAttribute;
014    import au.org.intersect.exsite9.domain.MetadataAttributeValue;
015    import au.org.intersect.exsite9.domain.MetadataCategory;
016    import au.org.intersect.exsite9.domain.MetadataCategoryType;

019    import au.org.intersect.exsite9.domain.Schema;

022    import au.org.intersect.exsite9.validators.MetadataAttributeValueValidator;

026    import static au.org.intersect.exsite9.xml.MetadataSchemaXMLUtils.*;

028    /**
029     * Used to read the XML when a {@link Schema} is imported
030     */
031    public final class MetadataSchemaXMLReader
032    {
033        /**
034         * Reads a Metadata Schema XML Document. Does not interact with the database to add the schema or its elements.
035         * @param xmlDocument The XML to read. It must conform to the Schema.
036         * @return The parsed Schema.
037         * @throws InvalidSchemaException If there is something wrong with parsing the Schema. 
038         */
039        public static Schema readXML(final Document xmlDocument) throws InvalidSchemaException
040        {
041            final Element rootElement = xmlDocument.getDocumentElement();
042    
043            final Schema schema = parseRootElement(rootElement);
044    
045            final NodeList metadataCategories = rootElement.getElementsByTagName(ELEMENT_METADATACATEGORY);
046            for (int i = 0; i < metadataCategories.getLength(); i++)
047            {
048                final Element metadataCategory = (Element) metadataCategories.item(i);
049                addMetadataCategory(schema, metadataCategory);
050            }
051    
052            return schema;
053        }
054    
055        private static Schema parseRootElement(final Element rootElement) throws InvalidSchemaException
056        {
057            final String name = rootElement.getAttribute(ATTRIBUTE_NAME).trim();
058            if (name.length() >= 255)
059            {
060                throw new InvalidSchemaException("Schema name is too long.");
061            }
062    
063            final String description = rootElement.getAttribute(ATTRIBUTE_DESCRIPTION).trim();
064            final String namespaceURL = rootElement.getAttribute(ATTRIBUTE_NAMESPACEURL).trim();
065            return new Schema(name, description, namespaceURL, Boolean.FALSE);
066        }
067    
068        private static void addMetadataCategory(final Schema schema, final Element metadataCategoryElement) throws InvalidSchemaException
069        {
070            final String name = metadataCategoryElement.getAttribute(ATTRIBUTE_NAME).trim();
071    
072            // Ensure the category name is valid
073            final MetadataCategoryNameValidator validator = new MetadataCategoryNameValidator(schema.getMetadataCategories(), null);
074            if (!validator.isValid(name))
075            {
076                throw new InvalidSchemaException("Metadata category '" + name + "' is invalid. " + validator.getErrorMessage());
077            }
078    
079            final MetadataCategoryUse use = MetadataCategoryUse.valueOf(metadataCategoryElement.getAttribute(ATTRIBUTE_USE).trim());
080            final MetadataCategoryType type = MetadataCategoryType.fromString(metadataCategoryElement.getAttribute(ATTRIBUTE_TYPE).trim());
081            
082            String description = "";
083            final NodeList descriptionNodes = metadataCategoryElement.getElementsByTagName(ELEMENT_DESCRIPTION);
084            if (descriptionNodes.item(0) != null){
085                description = descriptionNodes.item(0).getTextContent().trim();
086            }
087            
088            final boolean inextensible = Boolean.valueOf(metadataCategoryElement.getAttribute(ATTRIBUTE_INEXTENSIBLE).trim());
089    
090            final MetadataCategory metadataCategory = new MetadataCategory(name, type, use);
091            metadataCategory.setDescription(description);
092            metadataCategory.setInextensible(inextensible);
093            metadataCategory.setImported(true);
094    
095            if (type != MetadataCategoryType.FREETEXT)
096            {
097                final NodeList metadataCategoryValues = metadataCategoryElement.getElementsByTagName(ELEMENT_VALUE);
098                for (int i = 0; i < metadataCategoryValues.getLength(); i++)
099                {
100                    final Element metadataCategoryValue = (Element) metadataCategoryValues.item(i);
101                    addMetadataCategoryValue(metadataCategory, metadataCategoryValue);
102                }
103            }
104            else
105            {
106                final NodeList attributeElementList = metadataCategoryElement.getElementsByTagName(ELEMENT_ATTRIBUTE);
107                if (attributeElementList.getLength() > 1)
108                {
109                    throw new InvalidSchemaException("Only one " + ELEMENT_ATTRIBUTE + " element permitted as a child of a free-text metadata category");
110                }
111                if (attributeElementList.getLength() > 0)
112                {
113                    final Element attribtueElement = (Element) attributeElementList.item(0);
114                    final String attributeName = attribtueElement.getAttribute(ATTRIBUTE_NAME);
115                    final MetadataAttributeNameValidator manValidator = new MetadataAttributeNameValidator(false);
116                    if (!manValidator.isValid(attributeName))
117                    {
118                        throw new InvalidSchemaException("Metadata attribute name for '" + name + "' is invalid. " + validator.getErrorMessage());
119                    }
120    
121                    final MetadataAttribute metadataAttribute = new MetadataAttribute();
122                    metadataAttribute.setName(attributeName);
123    
124                    final NodeList attributeNodeValuesList = attribtueElement.getElementsByTagName(ELEMENT_VALUE);
125                    for (int i = 0; i < attributeNodeValuesList.getLength(); i++)
126                    {
127                        final Element metadataAttributeValue = (Element) attributeNodeValuesList.item(i);
128                        addMetadataAttributeValue(metadataAttribute, metadataAttributeValue);
129                    }
130    
131                    metadataCategory.setMetadataAttribute(metadataAttribute);
132                }
133            }
134    
135            schema.getMetadataCategories().add(metadataCategory);
136        }
137    
138        private static void addMetadataCategoryValue(final MetadataCategory metadataCategory, final Element metadataCategoryValueElement) throws InvalidSchemaException
139        {
140            final String value = metadataCategoryValueElement.getTextContent().trim();
141            final MetadataValueValidator validator = new MetadataValueValidator(metadataCategory.getValues());
142            if (!validator.isValid(value))
143            {
144                throw new InvalidSchemaException("Metadata value '" + value + "' in category '" + metadataCategory.getName() + "' is invalid. " + validator.getErrorMessage());
145            }
146            metadataCategory.getValues().add(new MetadataValue(value));
147        }
148    
149        private static void addMetadataAttributeValue(final MetadataAttribute metadataAttribute, final Element metadataAttributeValueElement) throws InvalidSchemaException
150        {
151            final String value = metadataAttributeValueElement.getTextContent().trim();
152            final MetadataAttributeValueValidator validator = new MetadataAttributeValueValidator(metadataAttribute.getMetadataAttributeValues());
153            if (!validator.isValid(value))
154            {
155                throw new InvalidSchemaException("Metadata attribute value '" + value + "' in attribute '" + metadataAttribute.getName() + "' is invalid. " + validator.getErrorMessage());
156            }
157            metadataAttribute.getMetadataAttributeValues().add(new MetadataAttributeValue(value));
158        }
