040        protected static Document createNewDocument() throws ParserConfigurationException
041        {
042            final DocumentBuilderFactory docFactory = DocumentBuilderFactory.newInstance();
043            final DocumentBuilder documentBuilder = docFactory.newDocumentBuilder();
044            
045            return documentBuilder.newDocument();

048        protected static Element createProjectInfo(final Document document, final Project project)
049        {
050            final Element projectInfoElement = document.createElement("project_info");
051            projectInfoElement.setAttribute("identifier", project.getIdentifier());
052            projectInfoElement.setAttribute("collectionType", project.getCollectionType());
053    
054            final Element nameElement = createElementWithTextContent(document, "projectName", project.getName());
055            final Element ownerElement = createElementWithTextContent(document, "name", project.getOwner());
056            final Element institutionElement = createElementWithTextContent(document, "institution", project.getInstitution());
057            final Element emailElement = createElementWithTextContent(document, "email", project.getEmail());
058            final Element descriptionElement = createElementWithTextContent(document, "description", project.getDescription());
059            final Element rightsStatementElement = createElementWithTextContent(document, "rightsStatement", project.getRightsStatement());
060            final Element accessRightsElement = createElementWithTextContent(document, "accessRights", project.getAccessRights());
061            final Element licenseElement = createElementWithTextContent(document, "license", project.getLicence());
062            final Element subjectElement = createElementWithTextContent(document, "subject", project.getSubject());
063            final Element electronicLocationElement = createElementWithTextContent(document, "electronicLocation", project.getElectronicLocation());
064            final Element physicalLocationElement = createElementWithTextContent(document, "physicalLocation", project.getPhysicalLocation());
065            final Element placeOrRegionNameElement = createElementWithTextContent(document, "placeOrRegionName", project.getPlaceOrRegionName());
066            final Element geographicalCoverageElement = createElementWithTextContent(document, "geographicalCoverage", project.getGeographicalCoverage());
067            final Element datesOfCaptureElement = createElementWithTextContent(document, "datesOfCapture", project.getDatesOfCapture());
068            final Element citationInformationElement = createElementWithTextContent(document, "citationInformation", project.getCitationInformation());
069            final Element countriesElement = createElementWithTextContent(document, "countries", project.getCountries());
070            final Element languagesElement = createElementWithTextContent(document, "languages", project.getLanguages());
071    
072            final FieldOfResearch fieldOfResearch = project.getFieldOfResearch();
073            final Element fieldOfResearchElement = createElementWithTextContent(document, "fieldOfResearch", fieldOfResearch == null ? "" : fieldOfResearch.toString());
074    
075            final Element fundingBodyElement = createElementWithTextContent(document, "fundingBody", project.getFundingBody());
076            final Element grantIDElement = createElementWithTextContent(document, "grantID", project.getGrantID());
077            final Element relatedPartyElement = createElementWithTextContent(document, "relatedParty", project.getRelatedParty());
078            final Element relatedGrantElement = createElementWithTextContent(document, "relatedGrant", project.getRelatedGrant());
079            final Element relatedInformationElement = createElementWithTextContent(document, "relatedInformation", project.getRelatedInformation());
080    
081            projectInfoElement.appendChild(nameElement);
082            projectInfoElement.appendChild(ownerElement);
083            projectInfoElement.appendChild(institutionElement);
084            projectInfoElement.appendChild(emailElement);
085            projectInfoElement.appendChild(descriptionElement);
086            projectInfoElement.appendChild(rightsStatementElement);
087            projectInfoElement.appendChild(accessRightsElement);
088            projectInfoElement.appendChild(licenseElement);
089            projectInfoElement.appendChild(subjectElement);
090            projectInfoElement.appendChild(electronicLocationElement);
091            projectInfoElement.appendChild(physicalLocationElement);
092            projectInfoElement.appendChild(placeOrRegionNameElement);
093            projectInfoElement.appendChild(geographicalCoverageElement);
094            projectInfoElement.appendChild(datesOfCaptureElement);
095            projectInfoElement.appendChild(citationInformationElement);
096            projectInfoElement.appendChild(countriesElement);
097            projectInfoElement.appendChild(languagesElement);
098            projectInfoElement.appendChild(fieldOfResearchElement);
099            projectInfoElement.appendChild(fundingBodyElement);
100            projectInfoElement.appendChild(grantIDElement);
101            projectInfoElement.appendChild(relatedPartyElement);
102            projectInfoElement.appendChild(relatedGrantElement);
103            projectInfoElement.appendChild(relatedInformationElement);
104    
105            return projectInfoElement;
106        }

108        protected static String transformDocumentToString(Document doc) throws TransformerConfigurationException, TransformerException
109        {
110            final TransformerFactory transformerFactory = TransformerFactory.newInstance();
111            final Transformer transformer = transformerFactory.newTransformer();
112            transformer.setOutputProperty("{http://xml.apache.org/xslt}indent-amount", "2");
113            transformer.setOutputProperty(OutputKeys.INDENT, "yes");
114            final DOMSource source = new DOMSource(doc);
115    
116            final StringWriter stringWriter = new StringWriter();
117            final StreamResult streamResult = new StreamResult(stringWriter);
118    
119            transformer.transform(source, streamResult);
120    
121            return stringWriter.toString();
122        }
123        
124        protected static void appendMetadataAssociation(final Document doc, final Element parent, final MetadataAssociation metadataAssociation)
125        {
126            for (final MetadataValue metadataValue : metadataAssociation.getMetadataValues())
127            {
128                final MetadataCategory metadataCategory = metadataAssociation.getMetadataCategory();
129                final Element metadataAssociationElement = doc.createElement(metadataCategory.getName());
130                metadataAssociationElement.setTextContent(metadataValue.getValue());
131    
132                final MetadataAttributeValue attributeValue = metadataAssociation.getMetadataAttributeValue();
133                if (attributeValue != null)
134                {
135                    metadataAssociationElement.setAttribute(metadataCategory.getMetadataAttribute().getName(), attributeValue.getValue());
136                }
137    
138                parent.appendChild(metadataAssociationElement);
139            }
140        }
141        
142        protected static void appendResearchFile(final Document doc, final Element parent, final ResearchFile researchFile, final boolean excludeMissingFiles)
143        {
144            final String researchFilePath = researchFile.getFile().getAbsolutePath();
145            appendResearchFile(doc, parent, researchFile, researchFilePath, excludeMissingFiles);
146        }
147    
148        protected static void appendResearchFile(final Document doc, final Element parent, final ResearchFile researchFile, final String researchFilePath, final boolean excludeMissingFiles)
149        {
150            final File file = researchFile.getFile();
151            if (excludeMissingFiles && !file.exists())
152            {
153                return;
154            }
155    
156            final Element researchFileElement = doc.createElement("file");
157    
158            final Element fileNameElement = doc.createElement("name");
159            fileNameElement.setTextContent(file.getName());
160    
161            final Element filePathElement = doc.createElement("path");
162            filePathElement.setTextContent(researchFilePath);
163    
164            researchFileElement.appendChild(fileNameElement);
165            researchFileElement.appendChild(filePathElement);
166    
167            for (final MetadataAssociation metadataAssociation : researchFile.getMetadataAssociations())
168            {
169                appendMetadataAssociation(doc, researchFileElement, metadataAssociation);
170            }
171            parent.appendChild(researchFileElement);
172        }
173    
174        protected static Element createElementWithTextContent(final Document document, final String elementName, final String textContent)
175        {
176            final Element element = document.createElement(elementName);
177            element.setTextContent(textContent);
178            return element;
179        }
