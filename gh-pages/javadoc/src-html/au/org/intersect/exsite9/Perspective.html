007    package au.org.intersect.exsite9;
008    
009    import java.util.HashSet;
010    import java.util.Set;
011    
012    import org.eclipse.ui.IFolderLayout;
013    import org.eclipse.ui.IPageLayout;
014    import org.eclipse.ui.IPerspectiveFactory;
015    import org.eclipse.ui.IViewLayout;
016    import org.eclipse.ui.PlatformUI;
017    import org.eclipse.ui.activities.IWorkbenchActivitySupport;
018    
019    import au.org.intersect.exsite9.view.AssociatedMetadataView;
020    import au.org.intersect.exsite9.view.MetadataBrowserView;
021    import au.org.intersect.exsite9.view.ProjectExplorerView;
022    import au.org.intersect.exsite9.view.SubmissionPackageBrowserView;
023    import au.org.intersect.exsite9.view.ThumbnailView;
024    
025    /**
026     * A Perspective holds the configuration and layout of the application's views.
027     */
028    public final class Perspective implements IPerspectiveFactory
029    {
030        /**
031         * {@inheritDoc}
032         */
033        public void createInitialLayout(final IPageLayout layout)
034        {
035            final IFolderLayout leftFolder = layout.createFolder("folder.left", IPageLayout.LEFT, 0.30f, layout.getEditorArea());
036            final IFolderLayout rightFolder = layout.createFolder("folder.right", IPageLayout.RIGHT, 0.70f, layout.getEditorArea());
037            final IFolderLayout bottomLeftFolder = layout.createFolder("folder.bottomLeft", IPageLayout.BOTTOM, 0.65f, "folder.left");
038    
039            leftFolder.addView(ProjectExplorerView.ID);
040            leftFolder.addView(SubmissionPackageBrowserView.ID);
041            bottomLeftFolder.addView(AssociatedMetadataView.ID);
042            bottomLeftFolder.addView(ThumbnailView.ID);
043            rightFolder.addView(MetadataBrowserView.ID);
044    
045            // Configure the Project Explorer View
046            final IViewLayout projectExplorerViewLayout = layout.getViewLayout(ProjectExplorerView.ID);
047            projectExplorerViewLayout.setCloseable(false);
048            projectExplorerViewLayout.setMoveable(false);
049    
050            // Configure the Submission Package Browser View
051            final IViewLayout submissionPackageBrowserViewLayout = layout.getViewLayout(SubmissionPackageBrowserView.ID);
052            submissionPackageBrowserViewLayout.setCloseable(false);
053            submissionPackageBrowserViewLayout.setMoveable(false);
054    
055            // Configure the Metadata Browser View
056            final IViewLayout metadataBrowserViewLayout = layout.getViewLayout(MetadataBrowserView.ID);
057            metadataBrowserViewLayout.setCloseable(false);
058            metadataBrowserViewLayout.setMoveable(false);
059            
060            // Configure the Associated Metadata View
061            final IViewLayout associatedMetadataViewLayout = layout.getViewLayout(AssociatedMetadataView.ID);
062            associatedMetadataViewLayout.setCloseable(false);
063            associatedMetadataViewLayout.setMoveable(false);
064            
065         // Configure the Thumbnail View
066            final IViewLayout ThumbnailViewLayout = layout.getViewLayout(ThumbnailView.ID);
067            ThumbnailViewLayout.setCloseable(false);
068            ThumbnailViewLayout.setMoveable(false);
069    
070            layout.setEditorAreaVisible(false);
071    
072            // Disables the minimize & maximize buttons on views and editors.
073            layout.setFixed(true);
074    
075            // Clears all the activities - since they are persisted magically.
076            final IWorkbenchActivitySupport activitySupport = PlatformUI.getWorkbench().getActiveWorkbenchWindow().getWorkbench().getActivitySupport();
077            final Set<String> enabledActivities = new HashSet<String>();
078            activitySupport.setEnabledActivityIds(enabledActivities);
079        }
080    }
