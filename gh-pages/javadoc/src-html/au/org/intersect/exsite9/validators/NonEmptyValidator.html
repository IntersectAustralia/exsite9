007    package au.org.intersect.exsite9.validators;
008    
009    import com.richclientgui.toolbox.validation.validator.IFieldValidator;
010    
011    /**
012     * Validator that always returns true if the string is not empty.
013     */
014    public final class NonEmptyValidator implements IFieldValidator<String>
015    {
016        private final String fieldName;
017    
018        public NonEmptyValidator(final String fieldName)
019        {
020            this.fieldName = fieldName;
021        }
022    
023        @Override
024        public String getErrorMessage()
025        {
026            return this.fieldName + " must not be empty";
027        }
028    
029        @Override
030        public String getWarningMessage()
031        {
032            return "";
033        }
034    
035        @Override
036        public boolean isValid(final String contents)
037        {
038            return !contents.trim().isEmpty();
039        }
040    
041        @Override
042        public boolean warningExist(final String contents)
043        {
044            return false;
045        }
046    }
