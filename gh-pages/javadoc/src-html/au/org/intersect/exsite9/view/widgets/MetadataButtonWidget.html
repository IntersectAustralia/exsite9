007    package au.org.intersect.exsite9.view.widgets;
008    
009    import org.eclipse.swt.widgets.Button;
010    import org.eclipse.swt.widgets.Composite;
011    
012    import au.org.intersect.exsite9.domain.MetadataCategory;
013    import au.org.intersect.exsite9.domain.MetadataValue;
014    
015    /**
016     * A Button that holds a metadataCategory and metadataValue - it make it easy to identify these characteristics
017     * of a button that was pressed. 
018     */
019    public final class MetadataButtonWidget extends Button
020    {
021        private final MetadataCategory metadataCategory;
022        private final MetadataValue metadataValue;
023    
024        public MetadataButtonWidget(final Composite parent, final int style, final MetadataCategory metadataCategory, final MetadataValue metadataValue)
025        {
026            super(parent, style);
027            this.metadataCategory = metadataCategory;
028            this.metadataValue = metadataValue;
029        }
030    
031        public MetadataCategory getMetadataCategory()
032        {
033            return this.metadataCategory;
034        }
035    
036        public MetadataValue getMetadataValue()
037        {
038            return this.metadataValue;
039        }
040    
041        @Override
042        protected void checkSubclass()
043        {
044            // screw you SWT!
045        }
046    }
