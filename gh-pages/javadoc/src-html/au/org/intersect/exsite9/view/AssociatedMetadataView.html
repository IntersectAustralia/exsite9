068        @SuppressWarnings("unchecked")
069        @Override
070        public void createPartControl(final Composite parent)
071        {
072            this.parent = parent;
073            final ICommandService commandService = (ICommandService) PlatformUI.getWorkbench().getService(
074                    ICommandService.class);
075    
076            this.tableViewer = new TableViewer(parent, SWT.V_SCROLL | SWT.H_SCROLL | SWT.VIRTUAL);
077    
078            final Table table = this.tableViewer.getTable();
079            table.setHeaderVisible(true);
080            table.setLinesVisible(true);
081    
082            this.tableViewer.setContentProvider(new AssociatedMetadataContentProvider());
083    
084            final TableViewerColumn nameColumn = new TableViewerColumn(this.tableViewer, SWT.NULL);
085            nameColumn.getColumn().setText("Name");
086    
087            final TableViewerColumn valueColumn = new TableViewerColumn(tableViewer, SWT.NULL);
088            valueColumn.getColumn().setText("Value");
089    
090            // use TableColumnLayout to make the columns auto re-sizable and split to width evenly
091            TableColumnLayout layout = new TableColumnLayout();
092            parent.setLayout(layout);
093            layout.setColumnData(nameColumn.getColumn(), new ColumnWeightData(50));
094            layout.setColumnData(valueColumn.getColumn(), new ColumnWeightData(50));
095    
096            nameColumn.setLabelProvider(new ColumnLabelProvider()
097            {
098    
099                @Override
100                public String getText(Object element)
101                {
102                    return ((Pair<String, String>) element).getFirst();
103                }
104    
105            });
106    
107            valueColumn.setLabelProvider(new ColumnLabelProvider()
108            {
109                @Override
110                public String getText(Object element)
111                {
112                    return ((Pair<String, String>) element).getSecond();
113                }
114    
115            });
116    
117            // Listen for selection changes in the treeViewer
118            getSite().getWorkbenchWindow().getSelectionService().addPostSelectionListener(ProjectExplorerView.ID, this);
119    
120            // Commands on which to refresh the table.
121            final Command editProjectCommand = commandService
122                    .getCommand("au.org.intersect.exsite9.commands.EditProjectCommand");
123            editProjectCommand.addExecutionListener(this);
124    
125            final Command addMetadataCategoryCommand = commandService
126                    .getCommand("au.org.intersect.exsite9.commands.AddMetadataCategoryCommand");
127            addMetadataCategoryCommand.addExecutionListener(this);
128    
129            final Command importMetadataSchemaCommand = commandService.getCommand("au.org.intersect.exsite9.commands.ImportMetadataSchemaCommand");
130            importMetadataSchemaCommand.addExecutionListener(this);
131    
132            final Command removeMetadataCategoryCommand = commandService.getCommand("au.org.intersect.exsite9.commands.RemoveMetadataCategoryCommand");
133            removeMetadataCategoryCommand.addExecutionListener(this);
134    
135            final Command editMetadataCategoryCommand = commandService.getCommand("au.org.intersect.exsite9.commands.EditMetadataCategoryCommand");
136            editMetadataCategoryCommand.addExecutionListener(this);
137    
138            final Command removeMetadataCategoryWithWizardCommand = commandService.getCommand("au.org.intersect.exsite9.commands.RemoveMetadataCategoryWithWizardCommand");
139            removeMetadataCategoryWithWizardCommand.addExecutionListener(this);
140        }

153        static class AssociatedMetadataContentProvider implements IStructuredContentProvider
154        {
155            private IGroupService groupService;
156            private IResearchFileService researchFileService;
157            private IProjectService projectService;
158    
159            private final List<IMetadataAssignable> selectedMetadataAssignables = new ArrayList<IMetadataAssignable>();
160    
161            public AssociatedMetadataContentProvider()
162            {
163                this.groupService = (IGroupService) PlatformUI.getWorkbench().getService(IGroupService.class);
164                this.researchFileService = (IResearchFileService) PlatformUI.getWorkbench().getService(
165                        IResearchFileService.class);
166                this.projectService = (IProjectService) PlatformUI.getWorkbench().getService(IProjectService.class);
167            }

172            }

177            }

179            @SuppressWarnings("unchecked")
180            @Override
181            public Object[] getElements(Object selection)
182            {
183                final IStructuredSelection structuredSelection = (IStructuredSelection) selection;
184    
185                final List<Object> selectedObjects = structuredSelection.toList();
186    
187                this.selectedMetadataAssignables.clear();
188    
189                for (final Object selectedObject : selectedObjects)
190                {
191                    if (selectedObject instanceof Group)
192                    {
193                        final Group selectedGroup = this.groupService.findGroupByID(((Group) selectedObject).getId());
194                        this.selectedMetadataAssignables.add(selectedGroup);
195                    }
196                    else if (selectedObject instanceof ResearchFile)
197                    {
198                        final ResearchFile selectedResearchFile = this.researchFileService
199                                .findResearchFileByID(((ResearchFile) selectedObject).getId());
200                        this.selectedMetadataAssignables.add(selectedResearchFile);
201                    }
202                    else if (selectedObject instanceof Project)
203                    {
204                        final Project selectedProject = this.projectService.findProjectById(((Project) selectedObject)
205                                .getId());
206                        // Its only possible for one project to be selected so the assignment to be done here
207                        return getAllNotEmptyFields(selectedProject).toArray();
208                    }
209                }
210    
211                if (!this.selectedMetadataAssignables.isEmpty())
212                {
213                    List<Pair<String, String>> stringsForTable = new ArrayList<Pair<String, String>>();
214    
215                    final Set<Triplet<MetadataCategory, MetadataValue, MetadataAttributeValue>> metadataToBeMapped =
216                            new HashSet<Triplet<MetadataCategory, MetadataValue, MetadataAttributeValue>>(MetadataAssignableUtils.getCategoryToValueMapping(this.selectedMetadataAssignables.get(0)));
217    
218                    for (int i = 1; i < this.selectedMetadataAssignables.size(); i++)
219                    {
220                        metadataToBeMapped.retainAll(MetadataAssignableUtils.getCategoryToValueMapping(this.selectedMetadataAssignables.get(i)));
221                    }
222    
223                    for (final Iterator<Triplet<MetadataCategory, MetadataValue, MetadataAttributeValue>> iterator = metadataToBeMapped.iterator(); iterator.hasNext();)
224                    {
225                        Triplet<MetadataCategory, MetadataValue, MetadataAttributeValue> triplet = (Triplet<MetadataCategory, MetadataValue, MetadataAttributeValue>) iterator.next();
226    
227                        final MetadataAttributeValue attributeValue = triplet.getThird();
228    
229                        final StringBuffer second = new StringBuffer();
230                        second.append(triplet.getSecond().getValue());
231                        if (attributeValue != null)
232                        {
233                            second.append(" (").append(triplet.getFirst().getMetadataAttribute().getName()).append(" = ").append(attributeValue.getValue()).append(")");
234                        }
235                        stringsForTable.add(new Pair<String, String>(triplet.getFirst().getName(), second.toString()));
236                    }
237    
238                    Collections.sort(stringsForTable, new AlphabeticalPairComparator());
239    
240                    return stringsForTable.toArray();
241                }
242                return new Object[0];
243            }
244        }

246        private static List<Pair<String, String>> getAllNotEmptyFields(Project project)
247        {
248            List<Pair<String, String>> fieldsToDisplay = new ArrayList<Pair<String, String>>();
249    
250            checkFieldIsNotEmpty(project.getName(), "Project Name", fieldsToDisplay);
251            checkFieldIsNotEmpty(project.getOwner(), "Name", fieldsToDisplay);
252            checkFieldIsNotEmpty(project.getInstitution(), "Institution", fieldsToDisplay);
253            checkFieldIsNotEmpty(project.getEmail(), "Email", fieldsToDisplay);
254            checkFieldIsNotEmpty(project.getDescription(), "Description", fieldsToDisplay);
255            checkFieldIsNotEmpty(project.getCollectionType(), "Collection Type", fieldsToDisplay);
256            checkFieldIsNotEmpty(project.getRightsStatement(), "Rights Statement", fieldsToDisplay);
257            checkFieldIsNotEmpty(project.getAccessRights(), "Access Rights", fieldsToDisplay);
258            checkFieldIsNotEmpty(project.getLicence(), "Licence", fieldsToDisplay);
259            checkFieldIsNotEmpty(project.getIdentifier(), "Identifier", fieldsToDisplay);
260            checkFieldIsNotEmpty(project.getSubject(), "Subject", fieldsToDisplay);
261            checkFieldIsNotEmpty(project.getElectronicLocation(), "Electronic Location", fieldsToDisplay);
262            checkFieldIsNotEmpty(project.getPhysicalLocation(), "Physical Location", fieldsToDisplay);
263            checkFieldIsNotEmpty(project.getPlaceOrRegionName(), "Place Or Region Name", fieldsToDisplay);
264            checkFieldIsNotEmpty(project.getGeographicalCoverage(), "Geographical Coverage", fieldsToDisplay);
265            checkFieldIsNotEmpty(project.getDatesOfCapture(), "Dates Of Capture", fieldsToDisplay);
266            checkFieldIsNotEmpty(project.getCitationInformation(), "Citation Information", fieldsToDisplay);
267            checkFieldIsNotEmpty(project.getCountries(), "Countries", fieldsToDisplay);
268            checkFieldIsNotEmpty(project.getLanguages(), "Languages", fieldsToDisplay);
269            final FieldOfResearch fieldOfResearch = project.getFieldOfResearch();
270            if (fieldOfResearch != null)
271            {
272                fieldsToDisplay.add(new Pair<String, String>("Field of Research", fieldOfResearch.toString()));
273            }
274            checkFieldIsNotEmpty(project.getFundingBody(), "Funding Body", fieldsToDisplay);
275            checkFieldIsNotEmpty(project.getGrantID(), "Grant ID", fieldsToDisplay);
276            checkFieldIsNotEmpty(project.getRelatedParty(), "Related Party", fieldsToDisplay);
277            checkFieldIsNotEmpty(project.getRelatedGrant(), "Related Grant", fieldsToDisplay);
278            checkFieldIsNotEmpty(project.getRelatedInformation(), "Related Information", fieldsToDisplay);
279    
280            return fieldsToDisplay;
281        }
282    
283        private static void checkFieldIsNotEmpty(final String fieldValue, final String fieldKey, List<Pair<String, String>> fieldsToDisplay)
284        {
285            if (!fieldValue.isEmpty())
286            {
287                fieldsToDisplay.add(new Pair<String, String>(fieldKey, fieldValue));
288            }
289        }
