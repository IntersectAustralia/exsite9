036    /**
037     * This is the ViewPart that will hold the Project Explorer UI component.
038     */
039    public final class ProjectExplorerView extends ViewPart implements IExecutionListener
040    {
041        // This needs to match what is defined in the plugin.xml
042        public static final String ID = ProjectExplorerView.class.getName();
043    
044        private TreeViewer treeViewer;
045    
046        /**
047         * Constructor
048         */
049        public ProjectExplorerView()
050        {
051        }
052    
053        /**
054         * {@inheritDoc}
055         */
056        @Override
057        public void createPartControl(final Composite parent)
058        {
059            this.setPartName("Project View");
060    
061            this.treeViewer = new TreeViewer(parent, SWT.MULTI | SWT.H_SCROLL | SWT.V_SCROLL | SWT.BORDER);
062            this.treeViewer.setContentProvider(new ProjectExplorerViewContentProvider(true));
063            this.treeViewer.setLabelProvider(new ProjectExplorerViewLabelProvider());
064            ColumnViewerToolTipSupport.enableFor(this.treeViewer);
065    
066            final ICommandService commandService = (ICommandService) PlatformUI.getWorkbench().getService(
067                    ICommandService.class);
068    
069            // Set up drag & drop
070            int operations = DND.DROP_MOVE;
071            Transfer[] transferTypes = new Transfer[] {LocalSelectionTransfer.getTransfer()};
072            ProjectExplorerDragListener dragListener = new ProjectExplorerDragListener(treeViewer);
073            ProjectExplorerDropListener dropListener = new ProjectExplorerDropListener(treeViewer);
074            this.treeViewer.addDragSupport(operations, transferTypes, dragListener);
075            this.treeViewer.addDropSupport(operations, transferTypes, dropListener);
076    
077            // This command is defined in the plugin.xml
078            // This is used to the view can load the project with the New Project command is executed.
079            final Command newProjectCommand = commandService
080                    .getCommand("au.org.intersect.exsite9.commands.NewProjectCommand");
081            newProjectCommand.addExecutionListener(this);
082    
083            final Command addFolderToProjectCommand = commandService
084                    .getCommand("au.org.intersect.exsite9.commands.AddFolderToProjectCommand");
085            addFolderToProjectCommand.addExecutionListener(this);
086    
087            final Command addGroupCommand = commandService.getCommand("au.org.intersect.exsite9.commands.AddGroup");
088            addGroupCommand.addExecutionListener(this);
089    
090            final Command deleteGroupCommand = commandService
091                    .getCommand("au.org.intersect.exsite9.commands.DeleteGroupCommand");
092            deleteGroupCommand.addExecutionListener(this);
093    
094            final Command reloadProjectCommand = commandService
095                    .getCommand("au.org.intersect.exsite9.commands.ReloadProjectCommand");
096            reloadProjectCommand.addExecutionListener(this);
097    
098            final Command openProjectCommand = commandService
099                    .getCommand("au.org.intersect.exsite9.commands.OpenProjectCommand");
100            openProjectCommand.addExecutionListener(this);
101    
102            final Command editProjectCommand = commandService
103                    .getCommand("au.org.intersect.exsite9.commands.EditProjectCommand");
104            editProjectCommand.addExecutionListener(this);
105    
106            final Command renameGroupCommand = commandService
107                    .getCommand("au.org.intersect.exsite9.commands.RenameGroupCommand");
108            renameGroupCommand.addExecutionListener(this);
109    
110            final Command listFoldersCommand = commandService
111                    .getCommand("au.org.intersect.exsite9.commands.ListFoldersCommand");
112            listFoldersCommand.addExecutionListener(this);
113    
114            final Command excludeResearchFilesCommand = commandService
115                    .getCommand("au.org.intersect.exsite9.commands.ExcludeResearchFiles");
116            excludeResearchFilesCommand.addExecutionListener(this);
117    
118            final Command listResearchFilesCommand = commandService
119                    .getCommand("au.org.intersect.exsite9.commands.ListExcludedFilesCommand");
120            listResearchFilesCommand.addExecutionListener(this);
121    
122            final Command removeMetadataCategoryCommand = commandService.getCommand("au.org.intersect.exsite9.commands.RemoveMetadataCategoryCommand");
123            removeMetadataCategoryCommand.addExecutionListener(this);
124    
125            final Command editMetadataCategoryCommand = commandService.getCommand("au.org.intersect.exsite9.commands.EditMetadataCategoryCommand");
126            editMetadataCategoryCommand.addExecutionListener(this);
127    
128            final Command removeMetadataCategoryWithWizardCommand = commandService.getCommand("au.org.intersect.exsite9.commands.RemoveMetadataCategoryWithWizardCommand");
129            removeMetadataCategoryWithWizardCommand.addExecutionListener(this);
130    
131            final Command moveToNewGroupCommand = commandService.getCommand("au.org.intersect.exsite9.commands.MoveToNewGroup");
132            moveToNewGroupCommand.addExecutionListener(this);
133    
134            final Command sortResearchFilesInGroupCommand = commandService.getCommand("au.org.intersect.exsite9.commands.SortResearchFilesInGroupCommand");
135            sortResearchFilesInGroupCommand.addExecutionListener(this);
136    
137            initContextMenu();
138        }
139    
140        /**
141         * Initializes the context menu for items in the tree view.
142         */
143        private void initContextMenu()
144        {
145            final MenuManager menuManager = new MenuManager();
146            final Menu menu = menuManager.createContextMenu(this.treeViewer.getTree());
147            this.treeViewer.getTree().setMenu(menu);
148            getSite().registerContextMenu(menuManager, this.treeViewer);
149    
150            // This allows other views to listen to selection changes.
151            getSite().setSelectionProvider(this.treeViewer);
152        }
153    
154        /**
155         * {@inheritDoc}
156         */
157        @Override
158        public void setFocus()
159        {
160            final MetadataBrowserView metadataBrowserView  = (MetadataBrowserView) ViewUtils.getViewByID(
161                    PlatformUI.getWorkbench().getActiveWorkbenchWindow(), MetadataBrowserView.ID);
162            metadataBrowserView.setEnabled(true);
163            
164            final AssociatedMetadataView associatedMetadataView = (AssociatedMetadataView) ViewUtils.getViewByID(PlatformUI
165                    .getWorkbench().getActiveWorkbenchWindow(), AssociatedMetadataView.ID);
166            associatedMetadataView.setEnabled(true);
167        }
168    
169        /**
170         * @{inheritDoc
171         */
172        @Override
173        public void notHandled(final String commandId, final NotHandledException exception)
174        {
175        }
176    
177        /**
178         * @{inheritDoc
179         */
180        @Override
181        public void postExecuteFailure(final String commandId, final ExecutionException exception)
182        {
183        }
184    
185        /**
186         * @{inheritDoc
187         */
188        @Override
189        public void postExecuteSuccess(final String commandId, final Object returnValue)
190        {
191            if (commandId.equals("au.org.intersect.exsite9.commands.NewProjectCommand")
192                    || commandId.equals("au.org.intersect.exsite9.commands.OpenProjectCommand"))
193            {
194                if (returnValue != null)
195                {
196                    displayProjectAndExpand();
197                }
198            }
199    
200            else if (commandId.equals("au.org.intersect.exsite9.commands.EditProjectCommand")
201                    || commandId.equals("au.org.intersect.exsite9.commands.AddFolderToProjectCommand")
202                    || commandId.equals("au.org.intersect.exsite9.commands.AddGroup")
203                    || commandId.equals("au.org.intersect.exsite9.commands.RenameGroupCommand")
204                    || commandId.equals("au.org.intersect.exsite9.commands.DeleteGroupCommand")
205                    || commandId.equals("au.org.intersect.exsite9.commands.ReloadProjectCommand")
206                    || commandId.equals("au.org.intersect.exsite9.commands.ListFoldersCommand")
207                    || commandId.equals("au.org.intersect.exsite9.commands.ExcludeResearchFiles")
208                    || commandId.equals("au.org.intersect.exsite9.commands.ListExcludedFilesCommand")
209                    || commandId.equals("au.org.intersect.exsite9.commands.RemoveMetadataCategoryCommand")
210                    || commandId.equals("au.org.intersect.exsite9.commands.EditMetadataCategoryCommand")
211                    || commandId.equals("au.org.intersect.exsite9.commands.RemoveMetadataCategoryWithWizardCommand")
212                    || commandId.equals("au.org.intersect.exsite9.commands.MoveToNewGroup")
213                    || commandId.equals("au.org.intersect.exsite9.commands.SortResearchFilesInGroupCommand"))
214            {
215                refresh();
216            }
217        }
218    
219        public void refresh()
220        {
221            this.treeViewer.refresh();
222        }
223    
224        public ISelection getSelection()
225        {
226            return this.treeViewer.getSelection();
227        }
228    
229        private void displayProjectAndExpand()
230        {
231            final IProjectManager projectManager = (IProjectManager) PlatformUI.getWorkbench().getService(
232                    IProjectManager.class);
233            final Project project = projectManager.getCurrentProject();
234            if (project != null)
235            {
236                final ProjectViewInputWrapper wrapper = new ProjectViewInputWrapper(project);
237                this.treeViewer.setInput(wrapper);
238                this.treeViewer.expandToLevel(2);
239            }
240        }
241    
242        /**
243         * @{inheritDoc
244         */
245        @Override
246        public void preExecute(final String commandId, final ExecutionEvent event)
247        {
248        }
