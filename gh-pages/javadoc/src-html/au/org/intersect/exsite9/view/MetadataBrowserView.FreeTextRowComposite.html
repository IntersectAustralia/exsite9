193        private void initLayout(final List<MetadataCategory> metadataCategories)
194        {
195            if (this.expandBar != null)
196            {
197                this.expandBar.dispose();
198            }
199    
200            if (this.placeholder != null)
201            {
202                this.placeholder.dispose();
203            }
204    
205            this.metadataButtons.clear();
206            this.freeTextParentComposites.clear();
207            this.freeTextRows.clear();
208            this.rows.clear();
209    
210            if (metadataCategories.isEmpty())
211            {
212                this.placeholder = new Composite(this.parent, SWT.BORDER);
213                return;
214            }
215    
216            this.expandBar = new ExpandBar(this.parent, SWT.BORDER | SWT.V_SCROLL);
217            this.expandBar.addExpandListener(new ExpandListener()
218            {
219                @Override
220                public void itemExpanded(final ExpandEvent e)
221                {
222                    final MetadataCategoryExpandItem expandItem = (MetadataCategoryExpandItem) e.item;
223                    final MetadataCategory metadataCategory = expandItem.getMetadataCategory();
224                    metadataCategoryViewConfigService.setExpanded(metadataCategory, true);
225                }
226                
227                @Override
228                public void itemCollapsed(final ExpandEvent e)
229                {
230                    final MetadataCategoryExpandItem expandItem = (MetadataCategoryExpandItem) e.item;
231                    final MetadataCategory metadataCategory = expandItem.getMetadataCategory();
232                    metadataCategoryViewConfigService.setExpanded(metadataCategory, false);
233                }
234            });
235    
236            final List<MetadataCategory> sorted = new ArrayList<MetadataCategory>(metadataCategories);
237            Collections.sort(sorted, new IDMetadataCategoryComparator());
238    
239            for (final MetadataCategory metadataCategory : sorted)
240            {
241                final Composite expandBarComposite = new Composite(expandBar, SWT.NONE);
242                final RowLayout expandBarLayout = new RowLayout(SWT.VERTICAL);
243                expandBarLayout.wrap = true;
244                expandBarLayout.pack = true;
245                expandBarLayout.justify = false;
246                expandBarComposite.setLayout(expandBarLayout);
247                
248                final Composite headerComposite = new Composite(expandBarComposite, SWT.NONE);
249    
250                final MetadataCategoryExpandItem expandItem = new MetadataCategoryExpandItem(this.expandBar, SWT.NONE, metadataCategory);
251                expandItem.setText(metadataCategory.getName() + " (" + metadataCategory.getUse() + ")");
252                expandItem.setControl(expandBarComposite);
253                expandItem.setHeight(expandBarComposite.computeSize(SWT.DEFAULT, SWT.DEFAULT).y);
254                
255                expandBarComposite.addControlListener(new ControlAdapter()
256                {
257                    @Override
258                    public void controlResized(ControlEvent e)
259                    {
260                        super.controlResized(e);
261                        expandItem.setHeight(expandBarComposite.computeSize(SWT.DEFAULT, SWT.DEFAULT).y);
262                    }
263                });
264    
265                final ToolBar toolBar = new ToolBar(headerComposite, SWT.FLAT | SWT.WRAP | SWT.RIGHT);
266                new ToolItem(toolBar, SWT.SEPARATOR);
267    
268                final ToolItem editButtonItem = new ToolItem(toolBar, SWT.NULL);
269                final Image editImage = Activator.getImageDescriptor("/icons/icon_edit_16.png").createImage();
270                editButtonItem.setImage(editImage);
271                editButtonItem.addSelectionListener(new MetadataCategorySelectionListener(metadataCategory,
272                        "au.org.intersect.exsite9.commands.AddMetadataCategoryCommand", "au.org.intersect.exsite9.commands.AddMetadataCategoryCommand.categoryParameter"));
273                editButtonItem.setToolTipText("Edit category");
274                
275                final ToolItem removeButtonItem = new ToolItem(toolBar, SWT.NULL);
276                final Image removeImage = Activator.getImageDescriptor("/icons/icon_exclude_16.png").createImage();
277                removeButtonItem.setImage(removeImage);
278                removeButtonItem.setToolTipText("Delete category");
279                removeButtonItem.addSelectionListener(new MetadataCategorySelectionListener(metadataCategory,
280                        "au.org.intersect.exsite9.commands.RemoveMetadataCategoryCommand", "au.org.intersect.exsite9.commands.RemoveMetadataCategoryCommand.categoryParameter"));
281    
282                if((metadataCategory.getDescription() != null) && (!metadataCategory.getDescription().isEmpty())){
283                    final ToolItem helpButtonItem = new ToolItem(toolBar, SWT.FLAT);
284                    final Image helpImage = PlatformUI.getWorkbench().getSharedImages().getImage(ISharedImages.IMG_LCL_LINKTO_HELP);
285                    helpButtonItem.setImage(helpImage);
286                    helpButtonItem.setToolTipText(metadataCategory.getDescription());
287                }
288                
289                new ToolItem(toolBar, SWT.SEPARATOR);
290                toolBar.pack();
291    
292    
293                if (metadataCategory.getType() == MetadataCategoryType.CONTROLLED_VOCABULARY)
294                {
295                    final Composite contentComposite = new Composite(expandBarComposite, SWT.NONE);
296                    layoutRow(contentComposite);
297    
298                    final RowLayout buttonLayout = new RowLayout(SWT.HORIZONTAL);
299                    buttonLayout.marginLeft = 10;
300                    buttonLayout.marginRight = 10;
301                    buttonLayout.marginTop = 10;
302                    buttonLayout.marginBottom = 10;
303                    buttonLayout.wrap = true;
304                    buttonLayout.pack = true;
305                    buttonLayout.justify = false;
306                    contentComposite.setLayout(buttonLayout);
307        
308                    // Sort the metadata values.
309                    final List<MetadataValue> sortedMetadataValues = new ArrayList<MetadataValue>(metadataCategory.getValues());
310                    Collections.sort(sortedMetadataValues, new AlphabeticalMetadataValueComparator());
311        
312                    for (final MetadataValue metadataValue : sortedMetadataValues)
313                    {
314                        final MetadataButtonWidget mdbw = new MetadataButtonWidget(contentComposite, SWT.TOGGLE, metadataCategory, metadataValue);
315                        mdbw.setText(metadataValue.getValue());
316                        mdbw.addSelectionListener(this);
317                        final Triplet<MetadataCategory, MetadataValue, MetadataAttributeValue> triplet = new Triplet<MetadataCategory, MetadataValue, MetadataAttributeValue>(metadataCategory, metadataValue, null);
318                        this.metadataButtons.put(triplet, mdbw);
319                    }
320                }
321                else
322                {
323                    freeTextParentComposites.put(metadataCategory, expandBarComposite);
324                }
325                final boolean expanded = metadataCategoryViewConfigService.isExpanded(metadataCategory);
326                expandItem.setExpanded(expanded);
327            }
328    
329            packAndLayout();
330        }

385        private void reloadView()
386        {
387            final IProjectManager projectManager = (IProjectManager) PlatformUI.getWorkbench().getService(IProjectManager.class);
388            final Project project = projectManager.getCurrentProject();
389            if (project != null)
390            {
391                initLayout(project.getSchema().getMetadataCategories());
392    
393                if (!project.getSchema().getMetadataCategories().isEmpty())
394                {
395                    // Refresh this page according to the currently selected items in the RHS.
396                    final ProjectExplorerView projectExplorerView = (ProjectExplorerView) ViewUtils.getViewByID(
397                        PlatformUI.getWorkbench().getActiveWorkbenchWindow(), ProjectExplorerView.ID);
398                    projectExplorerView.refresh();
399                    final ISelection currentSelection = projectExplorerView.getSelection();
400                    selectionChanged(projectExplorerView, currentSelection);
401                }
402            }
403        }

411        @Override
412        public void widgetSelected(final SelectionEvent e)
413        {
414            final MetadataButtonWidget button = (MetadataButtonWidget) e.widget;
415    
416            if (!validMetadataAssignablesSelected(true))
417            {
418                button.setSelection(false);
419                return;
420            }
421    
422            final MetadataCategory metadataCategory = button.getMetadataCategory();
423            final MetadataValue metadataValue = button.getMetadataValue();
424    
425            if (this.selectedMetadataAssignables.size() > 1)
426            {
427                final boolean performOperation = MessageDialog.openConfirm(getSite().getWorkbenchWindow().getShell(),
428                    "Caution","The metadata operation is about to be performed on all selected items. Are you sure you wish to proceed?");
429                if (!performOperation)
430                {
431                    button.setSelection(!button.getSelection());
432                    return;
433                }
434            }
435    
436            if (button.getSelection())
437            {
438                for (final IMetadataAssignable metadataAssignable : this.selectedMetadataAssignables)
439                {
440                    if (metadataAssignable instanceof Group)
441                    {
442                        // We need to reload the object model from the database because the user may have updated it's metadata associations
443                        // and wishes to perform another action on it (without selecting another one in between, which would force a reload from the DB)
444                        final Group freshGroup = this.groupService.findGroupByID(((Group)metadataAssignable).getId());
445                        groupService.associateMetadata(freshGroup, metadataCategory, metadataValue, null);
446                    }
447                    else if (metadataAssignable instanceof ResearchFile)
448                    {
449                        final ResearchFile freshResearchFile = this.researchFileService.findResearchFileByID(((ResearchFile)metadataAssignable).getId());
450                        researchFileService.associateMetadata(freshResearchFile, metadataCategory, metadataValue, null);
451                    }
452                }
453            }
454            else
455            {
456                for (final IMetadataAssignable metadataAssignable : this.selectedMetadataAssignables)
457                {
458                    if (metadataAssignable instanceof Group)
459                    {
460                        final Group freshGroup = this.groupService.findGroupByID(((Group)metadataAssignable).getId());
461                        groupService.disassociateMetadata(freshGroup, metadataCategory, metadataValue);
462                    }
463                    else if (metadataAssignable instanceof ResearchFile)
464                    {
465                        final ResearchFile freshResearchFile = this.researchFileService.findResearchFileByID(((ResearchFile)metadataAssignable).getId());
466                        researchFileService.disassociateMetadata(freshResearchFile, metadataCategory, metadataValue);
467                    }
468                }
469            }
470            refreshRelatedViews();
471        }

473        /**
474         * Ensures the user has selected valid metadata assignables in the project explorer view.
475         * @return {@code true} if selection is valid, {@code false} otherwise.
476         */
477        private boolean validMetadataAssignablesSelected(final boolean showErrors)
478        {
479            if (this.selectedMetadataAssignables.isEmpty())
480            {
481                if (showErrors)
482                {
483                    MessageDialog.openError(getSite().getWorkbenchWindow().getShell(), "Error", "Please select an item to assign metadata with.");
484                }
485                return false;
486            }
487    
488            // Check that the selected groups does not contain a new files group OR the project node - we CANNOT assign metadata
489            // to them.
490            if (!Collections2.filter(this.selectedMetadataAssignables, Predicates.instanceOf(NewFilesGroup.class)).isEmpty())
491            {
492                if (showErrors)
493                {
494                    MessageDialog.openError(getSite().getWorkbenchWindow().getShell(), "Error", "Metadata cannot be assigned to the new files group.");
495                }
496                return false;
497            }
498            else if (!Collections2.filter(this.selectedMetadataAssignables, Predicates.instanceOf(RootGroup.class)).isEmpty())
499            {
500                if (showErrors)
501                {
502                    MessageDialog.openError(getSite().getWorkbenchWindow().getShell(), "Error", "Metadata cannot be assigned to a project.");
503                }
504                return false;
505            }
506            return true;
507        }

509        /**
510         * Refreshes related views - performed after some action is performed on this view that other views may be interested in.
511         */
512        private static void refreshRelatedViews()
513        {
514            // Refresh the table in the Associated Metadata View when a value button is selected or de-selected
515            final AssociatedMetadataView associatedMetadataView = (AssociatedMetadataView) ViewUtils.getViewByID(
516                    PlatformUI.getWorkbench().getActiveWorkbenchWindow(), AssociatedMetadataView.ID);
517            associatedMetadataView.refresh();
518            
519            // Refresh the Project Explorer View according to the currently selected items in the RHS.
520            final ProjectExplorerView projectExplorerView = (ProjectExplorerView) ViewUtils.getViewByID(
521                    PlatformUI.getWorkbench().getActiveWorkbenchWindow(), ProjectExplorerView.ID);
522            projectExplorerView.refresh();        
523            
524        }

531        /**
532         * Listens to selection changes on the tree viewer in the project view on the LHS.
533         * 
534         * @{inheritDoc
535         */
536        @Override
537        public void selectionChanged(final IWorkbenchPart part, final ISelection selection)
538        {
539            if (!(selection instanceof IStructuredSelection))
540            {
541                LOG.error("Unknown selection type");
542                return;
543            }
544    
545            final IStructuredSelection structuredSelection = (IStructuredSelection) selection;
546    
547            @SuppressWarnings("unchecked")
548            final List<Object> selectedObjects = structuredSelection.toList();
549    
550            this.selectedMetadataAssignables.clear();
551    
552            for (final Object selectedObject : selectedObjects)
553            {
554                if (selectedObject instanceof Group && !(selectedObject instanceof NewFilesGroup))
555                {
556                    final Group selectedGroup = this.groupService.findGroupByID(((Group) selectedObject).getId());
557                    this.selectedMetadataAssignables.add(selectedGroup);
558                }
559                else if (selectedObject instanceof ResearchFile)
560                {
561                    final ResearchFile selectedResearchFile = this.researchFileService.findResearchFileByID(((ResearchFile) selectedObject).getId());
562                    this.selectedMetadataAssignables.add(selectedResearchFile);
563                }
564            }
565    
566            resetMetadataValueWidgets();
567            if (!this.selectedMetadataAssignables.isEmpty())
568            {
569                // Determine a common set of buttons that should be pressed and press them.
570                final Set<Triplet<MetadataCategory, MetadataValue, MetadataAttributeValue>> intersection = new HashSet<Triplet<MetadataCategory, MetadataValue, MetadataAttributeValue>>(
571                        MetadataAssignableUtils.getCategoryToValueMapping(this.selectedMetadataAssignables.get(0)));
572    
573                for (int i = 1; i < this.selectedMetadataAssignables.size(); i++)
574                {
575                    intersection.retainAll(MetadataAssignableUtils.getCategoryToValueMapping(this.selectedMetadataAssignables.get(i)));
576                }
577                setMetadataValueWidgets(new ArrayList<Triplet<MetadataCategory, MetadataValue, MetadataAttributeValue>>(intersection));
578            }
579            else
580            {
581                for (final MetadataCategory mdc : this.freeTextParentComposites.keySet())
582                {
583                    if (!this.freeTextRows.containsKey(mdc))
584                    {
585                        final Composite composite = freeTextParentComposites.get(mdc);
586                        final FreeTextRowComposite row = new FreeTextRowComposite(composite, SWT.NULL, mdc, null, null, true);
587                        this.freeTextRows.put(mdc, new ArrayList<MetadataBrowserView.FreeTextRowComposite>(Arrays.asList(row)));
588                        layoutRow(row.getComposite());
589                    }
590                }
591                packAndLayout();
592            }
593        }

605        /**
606         * Sets the metadata values that are selected.
607         * @param metadataCollectionTrio the metadata to select.
608         */
609        private void setMetadataValueWidgets(final List<Triplet<MetadataCategory, MetadataValue, MetadataAttributeValue>> metadataCollectionTrio)
610        {
611            Collections.sort(metadataCollectionTrio, new MetadataAssociationTripletComparator());
612            for (final Triplet<MetadataCategory, MetadataValue, MetadataAttributeValue> triplet : metadataCollectionTrio)
613            {
614                final MetadataCategory metadataCategory = triplet.getFirst();
615                if (metadataCategory.getType() == MetadataCategoryType.CONTROLLED_VOCABULARY)
616                {
617                    final MetadataButtonWidget metadataButtonWidget = this.metadataButtons.get(triplet);
618                    if (metadataButtonWidget != null)
619                    {
620                        metadataButtonWidget.setSelection(true);
621                    }
622                }
623                else
624                {
625                    final Composite composite = freeTextParentComposites.get(metadataCategory);
626                    final MetadataValue metadataValue = triplet.getSecond();
627                    final MetadataAttributeValue metadataAttributeValue = triplet.getThird();
628    
629                    final List<FreeTextRowComposite> rowsForMetadataCategory;
630                    if (this.freeTextRows.containsKey(metadataCategory))
631                    {
632                        rowsForMetadataCategory = this.freeTextRows.get(metadataCategory);
633                    }
634                    else
635                    {
636                        rowsForMetadataCategory = new ArrayList<FreeTextRowComposite>();
637                        this.freeTextRows.put(metadataCategory, rowsForMetadataCategory);
638                    }
639    
640                    final FreeTextRowComposite row = new FreeTextRowComposite(composite, SWT.NULL, metadataCategory, metadataValue, metadataAttributeValue,
641                            rowsForMetadataCategory.isEmpty());
642                    rowsForMetadataCategory.add(row);
643                    layoutRow(row.getComposite());
644                }
645            }
646    
647            for (final MetadataCategory mdc : this.freeTextParentComposites.keySet())
648            {
649                if (!this.freeTextRows.containsKey(mdc))
650                {
651                    final Composite composite = freeTextParentComposites.get(mdc);
652                    final FreeTextRowComposite row = new FreeTextRowComposite(composite, SWT.NULL, mdc, null, null, true);
653                    this.freeTextRows.put(mdc, new ArrayList<MetadataBrowserView.FreeTextRowComposite>(Arrays.asList(row)));
654                    layoutRow(row.getComposite());
655                }
656            }
657    
658            packAndLayout();
659        }

662         * Depresses (i.e. makes them NOT clicked) all the metadata value buttons currently on the page.
663         */
664        private void resetMetadataValueWidgets()
665        {
666            for (final MetadataButtonWidget metadataButtonWidget : this.metadataButtons.values())
667            {
668                metadataButtonWidget.setSelection(false);
669            }
670    
671            for (final List<FreeTextRowComposite> freeTextRows : this.freeTextRows.values())

676                }

681        }

683        /**
684         * Represents a row used for Free Text Metadata Categories.
685         */
686        public final class FreeTextRowComposite
687        {
688            private final Composite contentComposite;
689            private final MetadataCategory metadataCategory;
690            private final MetadataValue metadataValue;
691            private final MetadataAttributeValue metadataAttributeValue;
692            private final boolean first;
693    
694            public FreeTextRowComposite(final Composite parent, final int style, final MetadataCategory mdc, final MetadataValue metadataValue,
695                    final MetadataAttributeValue metadataAttribtueValue, final boolean first)
696            {
697                this.contentComposite = new Composite(parent, style);
698                this.metadataCategory = mdc;
699                this.metadataValue = metadataValue;
700                this.metadataAttributeValue = metadataAttribtueValue;
701                this.first = first;
702                init();
703            }
704    
705            public Composite getComposite()
706            {
707                return contentComposite;
708            }
709    
710            private void init()
711            {
712                final MetadataAttribute metadataAttribute = metadataCategory.getMetadataAttribute();
713                final boolean hasAttributes = metadataAttribute != null;
714    
715                final int numCols = hasAttributes ? 7 : 5;
716    
717                final GridLayout gridLayout = new GridLayout(numCols, false);
718                contentComposite.setLayout(gridLayout);
719    
720                final MetadataAttributeValuesComboWidget combo;
721                if (hasAttributes)
722                {
723                    final Label attributeNameLabel = new Label(contentComposite, SWT.NULL);
724                    attributeNameLabel.setText(metadataAttribute.getName() + ":");
725                    combo = new MetadataAttributeValuesComboWidget(contentComposite, SWT.READ_ONLY);
726                    final List<MetadataAttributeValue> values = new ArrayList<MetadataAttributeValue>(metadataAttribute.getMetadataAttributeValues());
727                    Collections.sort(values, new AlphabeticalMetadataAttributeValueComparator());
728                    combo.setItems(values);
729                }
730                else
731                {
732                    combo = null;
733                }
734                
735                final MetadataTextWidget freeTextField = new MetadataTextWidget(contentComposite, SWT.BORDER | SWT.SINGLE);
736                freeTextField.setText(metadataValue == null ? "" : metadataValue.getValue());
737                freeTextField.setMetadataValue(metadataValue);
738                freeTextField.setToolTipText(freeTextField.getText());
739    
740                if (combo != null && metadataAttributeValue != null)
741                {
742                    combo.select(metadataAttributeValue);
743                    combo.setMetadataAttributeValue(metadataAttributeValue);
744                }
745    
746                final GridData fillGridLayout = new GridData(GridData.FILL_HORIZONTAL);
747                freeTextField.setLayoutData(fillGridLayout);
748    
749                final Button resetButton = new Button(contentComposite, SWT.PUSH);
750                resetButton.setText("Reset");
751                resetButton.setEnabled(false);
752    
753                final Button applyButton = new Button(contentComposite, SWT.PUSH);
754                applyButton.setText("Apply");
755                applyButton.setEnabled(false);
756    
757                final Button minusButton = new Button(contentComposite, SWT.PUSH);
758                minusButton.setText("-");
759                minusButton.setEnabled(!first);
760    
761                final Button plusButton = new Button(contentComposite, SWT.PUSH);
762                plusButton.setText("+");
763    
764                if (combo != null)
765                {
766                    combo.addSelectionListener(new SelectionListener()
767                    {
768                        @Override
769                        public void widgetSelected(final SelectionEvent e)
770                        {
771                            if (!validMetadataAssignablesSelected(false))
772                            {
773                                resetButton.setEnabled(false);
774                                applyButton.setEnabled(false);
775                                return;
776                            }
777                            
778                            final MetadataAttributeValue currentValue = combo.getSelectedMetadataAttributeValue();
779                            final boolean differentAttributeValue = !Objects.equal(metadataAttributeValue, currentValue);
780    
781                            final String originalText = metadataValue == null ? "" : metadataValue.getValue();
782                            final String newText = freeTextField.getText().trim();
783                            final boolean differentTextValue = !Objects.equal(newText, originalText);
784                            freeTextField.setToolTipText(newText);
785                            resetButton.setEnabled(differentAttributeValue || differentTextValue);
786                            applyButton.setEnabled((differentAttributeValue && !newText.isEmpty()) || (differentTextValue && !newText.isEmpty()));
787                        }
788                        
789                        @Override
790                        public void widgetDefaultSelected(SelectionEvent e)
791                        {
792                        }
793                    });
794                }
795    
796                resetButton.addSelectionListener(new SelectionListener()
797                {
798                    @Override
799                    public void widgetSelected(final SelectionEvent e)
800                    {
801                        final String originalText = metadataValue == null ? "" : metadataValue.getValue();
802                        freeTextField.setText(originalText);
803                        freeTextField.setSelection(originalText.length(), originalText.length());
804                        freeTextField.setToolTipText(freeTextField.getText());
805                        if (combo != null)
806                        {
807                            if (metadataAttributeValue != null)
808                            {
809                                combo.select(metadataAttributeValue);
810                            }
811                            else
812                            {
813                                combo.select(0);
814                            }
815                        }
816                        resetButton.setEnabled(false);
817                        applyButton.setEnabled(false);
818                    }
819                    
820                    @Override
821                    public void widgetDefaultSelected(final SelectionEvent e)
822                    {
823                    }
824                });
825    
826                applyButton.addSelectionListener(new SelectionListener()
827                {
828                    @Override
829                    public void widgetSelected(final SelectionEvent event)
830                    {
831                        if (!validMetadataAssignablesSelected(true))
832                        {
833                            return;
834                        }
835    
836                        // Handle the assignment.
837                        if (selectedMetadataAssignables.size() > 1)
838                        {
839                            final boolean performOperation = MessageDialog.openConfirm(getSite().getWorkbenchWindow().getShell(),
840                                "Caution","The metadata operation is about to be performed on all selected items. Are you sure you wish to proceed?");
841                            if (!performOperation)
842                            {
843                                return;
844                            }
845                        }
846    
847                        final String newValue = freeTextField.getText();
848                        final MetadataAttributeValue metadataAttributeValue = combo != null ? combo.getSelectedMetadataAttributeValue() : null;
849    
850                        final MetadataValueValidator validtor = new MetadataValueValidator(Collections.<MetadataValue>emptyList(), true);
851                        if (!validtor.isValid(newValue))
852                        {
853                            MessageDialog.openError(getSite().getWorkbenchWindow().getShell(), "Cannot apply metadata", "Cannot apply metadata. " + validtor.getErrorMessage());
854                            return;
855                        }
856    
857                        // We need to reload the metadata category.
858                        final MetadataCategory updatedMetadataCategory = metadataCategoryService.findById(metadataCategory.getId());
859    
860                        // Disassociate old value
861                        final MetadataValue oldAssociatedValue = freeTextField.getMetadataValue();
862    
863                        if (oldAssociatedValue != null)
864                        {
865                            disassociateFreeText(updatedMetadataCategory, oldAssociatedValue);
866                        }
867    
868                        // Add & Associate new value
869                        if (!newValue.isEmpty())
870                        {
871                            // Persist the newly configured metadata category value.
872                            final MetadataValue metadataValue = metadataCategoryService.addValueToMetadataCategory(updatedMetadataCategory, newValue);
873                            for (final IMetadataAssignable metadataAssignable : selectedMetadataAssignables)
874                            {
875                                if (metadataAssignable instanceof Group)
876                                {
877                                    // We need to reload the object model from the database because the user may have updated it's metadata associations
878                                    // and wishes to perform another action on it (without selecting another one in between, which would force a reload from the DB)
879                                    final Group freshGroup = groupService.findGroupByID(((Group)metadataAssignable).getId());
880                                    groupService.associateMetadata(freshGroup, updatedMetadataCategory, metadataValue, metadataAttributeValue);
881                                }
882                                else if (metadataAssignable instanceof ResearchFile)
883                                {
884                                    final ResearchFile freshResearchFile = researchFileService.findResearchFileByID(((ResearchFile)metadataAssignable).getId());
885                                    researchFileService.associateMetadata(freshResearchFile, updatedMetadataCategory, metadataValue, metadataAttributeValue);
886                                }
887                            }
888                            freeTextField.setMetadataValue(metadataValue);
889                            applyButton.setEnabled(false);
890                            resetButton.setEnabled(false);
891                        }
892                        refreshRelatedViews();
893                    }
894                    
895                    @Override
896                    public void widgetDefaultSelected(final SelectionEvent event)
897                    {
898                    }
899                });
900    
901                plusButton.addSelectionListener(new SelectionListener()
902                {
903                    @Override
904                    public void widgetSelected(final SelectionEvent e)
905                    {
906                        final Composite composite = freeTextParentComposites.get(metadataCategory);
907                        final FreeTextRowComposite row = new FreeTextRowComposite(composite, SWT.NULL, metadataCategory, null, null, false);
908                        freeTextRows.get(metadataCategory).add(row);
909                        layoutRow(row.getComposite());
910                        final Composite parent = contentComposite.getParent();
911                        parent.pack();
912                        parent.layout();
913                    }
914    
915                    @Override
916                    public void widgetDefaultSelected(final SelectionEvent e)
917                    {
918                    }
919                });
920    
921                minusButton.addSelectionListener(new SelectionListener()
922                {
923                    @Override
924                    public void widgetSelected(final SelectionEvent e)
925                    {
926                        // Remove metadata category/value mapping - which includes disassocation.
927                        final MetadataCategory updatedMetadataCategory = metadataCategoryService.findById(metadataCategory.getId());
928                        final MetadataValue oldAssociatedValue = freeTextField.getMetadataValue();
929    
930                        if (oldAssociatedValue != null)
931                        {
932                            disassociateFreeText(updatedMetadataCategory, oldAssociatedValue);
933                        }
934    
935                        final Composite parent = contentComposite.getParent();
936                        contentComposite.dispose();
937                        parent.pack();
938                        parent.layout();
939                        refreshRelatedViews();
940                    }   
941    
942                    @Override
943                    public void widgetDefaultSelected(final SelectionEvent e)
944                    {
945                    }
946                });
947    
948                freeTextField.addKeyListener(new KeyListener()
949                {
950                    @Override
951                    public void keyReleased(final KeyEvent e)
952                    {
953                        if (!validMetadataAssignablesSelected(false))
954                        {
955                            resetButton.setEnabled(false);
956                            applyButton.setEnabled(false);
957                            return;
958                        }
959    
960                        final String originalText = metadataValue == null ? "" : metadataValue.getValue();
961                        final String newText = freeTextField.getText().trim();
962                        final boolean enabled = !originalText.equals(newText);
963                        freeTextField.setToolTipText(newText);
964                        resetButton.setEnabled(enabled);
965                        applyButton.setEnabled(enabled && !newText.isEmpty());
966                    }
967                    
968                    @Override
969                    public void keyPressed(final KeyEvent e)
970                    {
971                    }
972                });
973            }

975            private void disassociateFreeText(final MetadataCategory updatedMetadataCategory, final MetadataValue oldAssociatedValue)
976            {
977                for (final IMetadataAssignable metadataAssignable : selectedMetadataAssignables)
978                {
979                    if (metadataAssignable instanceof Group)
980                    {
981                        // We need to reload the object model from the database because the user may have updated it's metadata associations
982                        // and wishes to perform another action on it (without selecting another one in between, which would force a reload from the DB)
983                        final Group freshGroup = groupService.findGroupByID(((Group)metadataAssignable).getId());
984                        groupService.disassociateMetadata(freshGroup, updatedMetadataCategory, oldAssociatedValue);
985                    }
986                    else if (metadataAssignable instanceof ResearchFile)
987                    {
988                        final ResearchFile freshResearchFile = researchFileService.findResearchFileByID(((ResearchFile)metadataAssignable).getId());
989                        researchFileService.disassociateMetadata(freshResearchFile, updatedMetadataCategory, oldAssociatedValue);
990                    }
991                }
992            }
