001    package au.org.intersect.exsite9.view.listener;
002    
003    import java.util.ArrayList;
004    
005    import org.apache.log4j.Logger;
006    import org.eclipse.core.commands.Command;
007    import org.eclipse.core.commands.ExecutionException;
008    import org.eclipse.core.commands.IParameter;
009    import org.eclipse.core.commands.NotEnabledException;
010    import org.eclipse.core.commands.NotHandledException;
011    import org.eclipse.core.commands.Parameterization;
012    import org.eclipse.core.commands.ParameterizedCommand;
013    import org.eclipse.core.commands.common.NotDefinedException;
014    import org.eclipse.swt.events.SelectionEvent;
015    import org.eclipse.swt.events.SelectionListener;
016    import org.eclipse.ui.IWorkbenchWindow;
017    import org.eclipse.ui.PlatformUI;
018    import org.eclipse.ui.commands.ICommandService;
019    import org.eclipse.ui.handlers.IHandlerService;
020    
021    import au.org.intersect.exsite9.domain.MetadataCategory;
022    
023    /**
024     * A Selection Listener for a MetadataCategory capable of executing a command and providing the Metadata Category ID as a parameter to the command.
025     */
026    public final class MetadataCategorySelectionListener implements SelectionListener
027    {
028        private static final Logger LOG = Logger.getLogger(MetadataCategorySelectionListener.class);
029    
030        private final MetadataCategory metadataCategory;
031        private final String commandId;
032        private final String parameterId;
033    
034        public MetadataCategorySelectionListener(final MetadataCategory metadataCategory, final String commandId, final String parameterId)
035        {
036            this.metadataCategory = metadataCategory;
037            this.commandId = commandId;
038            this.parameterId = parameterId;
039        }
040    
041        @Override
042        public void widgetSelected(final SelectionEvent event)
043        {
044            // Fire the command including the metadata category ID as an argument.
045            final IWorkbenchWindow window = PlatformUI.getWorkbench().getActiveWorkbenchWindow();
046            final ICommandService commandService = (ICommandService) window.getService(ICommandService.class);
047            final Command command = commandService.getCommand(this.commandId);
048    
049            final IParameter iparam;
050            try
051            {
052                iparam = command.getParameter(this.parameterId);
053            }
054            catch (final NotDefinedException e)
055            {
056                LOG.error("Expecting parameter " + this.parameterId + " on command " + this.commandId, e);
057                return;
058            }
059    
060            final Parameterization params = new Parameterization(iparam, metadataCategory.getId().toString());
061            final ArrayList<Parameterization> parameters = new ArrayList<Parameterization>();
062            parameters.add(params);
063    
064            // Build the parameterized command
065            final ParameterizedCommand pc = new ParameterizedCommand(command, parameters.toArray(new Parameterization[parameters.size()]));
066    
067            // Execute the command
068            final IHandlerService handlerService = (IHandlerService) window.getService(IHandlerService.class);
069            try
070            {
071                handlerService.executeCommand(pc, null);
072            }
073            catch (final ExecutionException e)
074            {
075                LOG.error("Cannot execute paramertized command", e);
076            }
077            catch (final NotDefinedException e)
078            {
079                LOG.error("Cannot execute paramertized command", e);
080            }
081            catch (final NotEnabledException e)
082            {
083                LOG.error("Cannot execute paramertized command", e);
084            }
085            catch (final NotHandledException e)
086            {
087                LOG.error("Cannot execute paramertized command", e);
088            }
089        }
090    
091        @Override
092        public void widgetDefaultSelected(final SelectionEvent event)
093        {
094        }
095    
096    }
