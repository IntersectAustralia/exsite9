011    package au.org.intersect.exsite9.view.dialogs;
012    
013    import java.util.Arrays;
014    
015    import org.eclipse.jface.viewers.ILabelProvider;
016    import org.eclipse.swt.widgets.Composite;
017    import org.eclipse.swt.widgets.Control;
018    import org.eclipse.swt.widgets.Shell;
019    
020    /**
021     * A class to select elements out of a list of elements.
022     * 
023     * @since 2.0
024     */
025    public class ElementListSelectionDialog extends
026            AbstractElementListSelectionDialog {
027    
028        private Object[] fElements;
029    
030        /**
031         * Creates a list selection dialog.
032         * @param parent   the parent widget.
033         * @param renderer the label renderer.
034         */
035        public ElementListSelectionDialog(Shell parent, ILabelProvider renderer) {
036            super(parent, renderer);
037        }
038    
039        /**
040         * Sets the elements of the list.
041         * @param elements the elements of the list.
042         */
043        public void setElements(Object[] elements) {
044            fElements = elements;
045        }
046    
047        /*
048         * @see SelectionStatusDialog#computeResult()
049         */
050        protected void computeResult() {
051            setResult(Arrays.asList(getSelectedElements()));
052        }
053    
054        /*
055         * @see Dialog#createDialogArea(Composite)
056         */
057        protected Control createDialogArea(Composite parent) {
058            Composite contents = (Composite) super.createDialogArea(parent);
059    
060            createMessageArea(contents);
061            createFilterText(contents);
062            createFilteredList(contents);
063    
064            setListElements(fElements);
065    
066            setSelection(getInitialElementSelections().toArray());
067    
068            return contents;
069        }
070    }
