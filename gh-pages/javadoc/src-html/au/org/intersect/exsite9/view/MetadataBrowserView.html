183        private void initLayout(final List<MetadataCategory> metadataCategories)
184        {
185            if (this.expandBar != null)
186            {
187                this.expandBar.dispose();
188            }
189    
190            if (this.placeholder != null)
191            {
192                this.placeholder.dispose();
193            }
194    
195            this.metadataButtons.clear();
196            this.metadataFreeTextFields.clear();
197            this.metadataAttributeValueCombos.clear();
198            this.rows.clear();
199    
200            if (metadataCategories.isEmpty())
201            {
202                this.placeholder = new Composite(this.parent, SWT.BORDER);
203                return;
204            }
205    
206            this.expandBar = new ExpandBar(this.parent, SWT.BORDER | SWT.V_SCROLL);
207    
208            final List<MetadataCategory> sorted = new ArrayList<MetadataCategory>(metadataCategories);
209            Collections.sort(sorted, new IDMetadataCategoryComparator());
210    
211            for (final MetadataCategory metadataCategory : sorted)
212            {
213                final Composite expandBarComposite = new Composite(expandBar, SWT.NONE);
214                final RowLayout expandBarLayout = new RowLayout(SWT.VERTICAL);
215                expandBarLayout.wrap = true;
216                expandBarLayout.pack = true;
217                expandBarLayout.justify = false;
218                expandBarComposite.setLayout(expandBarLayout);
219                
220                final Composite headerComposite = new Composite(expandBarComposite, SWT.NONE);
221                final Composite contentComposite = new Composite(expandBarComposite, SWT.NONE);
222                final ExpandItem expandItem = new ExpandItem(this.expandBar, SWT.NONE);
223                expandItem.setText(metadataCategory.getName() + " (" + metadataCategory.getUse() + ")");
224                expandItem.setControl(expandBarComposite);
225                expandItem.setHeight(expandBarComposite.computeSize(SWT.DEFAULT, SWT.DEFAULT).y);
226                
227                expandBarComposite.addControlListener(new ControlAdapter()
228                {
229                    @Override
230                    public void controlResized(ControlEvent e)
231                    {
232                        super.controlResized(e);
233                        expandItem.setHeight(expandBarComposite.computeSize(SWT.DEFAULT, SWT.DEFAULT).y);
234                    }
235                });
236    
237                final ToolBar toolBar = new ToolBar(headerComposite, SWT.FLAT | SWT.WRAP | SWT.RIGHT);
238                new ToolItem(toolBar, SWT.SEPARATOR);
239    
240                final ToolItem editButtonItem = new ToolItem(toolBar, SWT.NULL);
241                final Image editImage = Activator.getImageDescriptor("/icons/icon_edit_16.png").createImage();
242                editButtonItem.setImage(editImage);
243                editButtonItem.addSelectionListener(new MetadataCategorySelectionListener(metadataCategory,
244                        "au.org.intersect.exsite9.commands.AddMetadataCategoryCommand", "au.org.intersect.exsite9.commands.AddMetadataCategoryCommand.categoryParameter"));
245                editButtonItem.setToolTipText("Edit category");
246                
247                final ToolItem removeButtonItem = new ToolItem(toolBar, SWT.NULL);
248                final Image removeImage = Activator.getImageDescriptor("/icons/icon_exclude_16.png").createImage();
249                removeButtonItem.setImage(removeImage);
250                removeButtonItem.setToolTipText("Remove category");
251                removeButtonItem.addSelectionListener(new MetadataCategorySelectionListener(metadataCategory,
252                        "au.org.intersect.exsite9.commands.RemoveMetadataCategoryCommand", "au.org.intersect.exsite9.commands.RemoveMetadataCategoryCommand.categoryParameter"));
253    
254                if((metadataCategory.getDescription() != null) && (!metadataCategory.getDescription().isEmpty())){
255                    final ToolItem helpButtonItem = new ToolItem(toolBar, SWT.FLAT);
256                    final Image helpImage = PlatformUI.getWorkbench().getSharedImages().getImage(ISharedImages.IMG_LCL_LINKTO_HELP);
257                    helpButtonItem.setImage(helpImage);
258                    helpButtonItem.setToolTipText(metadataCategory.getDescription());
259                }
260                
261                new ToolItem(toolBar, SWT.SEPARATOR);
262                toolBar.pack();
263    
264                final RowData rowData = new RowData();
265                rowData.width = parent.getClientArea().width;
266                contentComposite.setLayoutData(rowData);
267                rows.add(rowData);
268    
269                if (metadataCategory.getType() == MetadataCategoryType.CONTROLLED_VOCABULARY)
270                {
271                    final RowLayout buttonLayout = new RowLayout(SWT.HORIZONTAL);
272                    buttonLayout.marginLeft = 10;
273                    buttonLayout.marginRight = 10;
274                    buttonLayout.marginTop = 10;
275                    buttonLayout.marginBottom = 10;
276                    buttonLayout.wrap = true;
277                    buttonLayout.pack = true;
278                    buttonLayout.justify = false;
279                    contentComposite.setLayout(buttonLayout);
280        
281                    // Sort the metadata values.
282                    final List<MetadataValue> sortedMetadataValues = new ArrayList<MetadataValue>(metadataCategory.getValues());
283                    Collections.sort(sortedMetadataValues, new AlphabeticalMetadataValueComparator());
284        
285                    for (final MetadataValue metadataValue : sortedMetadataValues)
286                    {
287                        final MetadataButtonWidget mdbw = new MetadataButtonWidget(contentComposite, SWT.TOGGLE, metadataCategory, metadataValue);
288                        mdbw.setText(metadataValue.getValue());
289                        mdbw.addSelectionListener(this);
290                        final Triplet<MetadataCategory, MetadataValue, MetadataAttributeValue> triplet = new Triplet<MetadataCategory, MetadataValue, MetadataAttributeValue>(metadataCategory, metadataValue, null);
291                        this.metadataButtons.put(triplet, mdbw);
292                    }
293                }
294                else
295                {
296                    buildFreeTextRow(contentComposite, metadataCategory);
297                }
298    
299                contentComposite.pack();
300                contentComposite.layout(true);
301                expandItem.setExpanded(true);
302            }
303    
304            this.expandBar.pack();
305            this.expandBar.layout();
306    
307            this.parent.layout();
308        }

310        /**
311         * Builds a free text row.
312         * @param contentComposite The composite to build the row with.
313         * @param metadataCategory The metadata Category to use.
314         */
315        private void buildFreeTextRow(final Composite contentComposite, final MetadataCategory metadataCategory)
316        {
317            final MetadataAttribute metadataAttribute = metadataCategory.getMetadataAttribute();
318            final boolean hasAttribute = metadataAttribute != null;
319    
320            final int numCols = hasAttribute ? 7 : 5;
321    
322            final GridLayout gridLayout = new GridLayout(numCols, false);
323            contentComposite.setLayout(gridLayout);
324    
325            final MetadataAttributeValuesComboWidget combo;
326            if (hasAttribute)
327            {
328                final Label attributeNameLabel = new Label(contentComposite, SWT.NULL);
329                attributeNameLabel.setText(metadataAttribute.getName() + ":");
330    
331                combo = new MetadataAttributeValuesComboWidget(contentComposite, SWT.READ_ONLY);
332                combo.setItems(metadataAttribute.getMetadataAttributeValues());
333                this.metadataAttributeValueCombos.put(metadataCategory, combo);
334            }
335            else
336            {
337                combo = null;
338            }
339            
340            final MetadataTextWidget freeTextField = new MetadataTextWidget(contentComposite, SWT.BORDER | SWT.SINGLE);
341            this.metadataFreeTextFields.put(metadataCategory, freeTextField);
342    
343            final GridData multiLineGridData = new GridData(GridData.FILL_HORIZONTAL);
344            freeTextField.setLayoutData(multiLineGridData);
345    
346            final Button resetButton = new Button(contentComposite, SWT.PUSH);
347            resetButton.setText("Reset");
348            resetButton.setEnabled(false);
349    
350            final Button applyButton = new Button(contentComposite, SWT.PUSH);
351            applyButton.setText("Apply");
352            applyButton.setEnabled(false);
353    
354            final Button minusButton = new Button(contentComposite, SWT.PUSH);
355            minusButton.setText("-");
356    
357            final Button plusButton = new Button(contentComposite, SWT.PUSH);
358            plusButton.setText("+");
359    
360            if (combo != null)
361            {
362                combo.addSelectionListener(new SelectionListener()
363                {
364                    @Override
365                    public void widgetSelected(final SelectionEvent e)
366                    {
367                        if (!validMetadataAssignablesSelected(false))
368                        {
369                            resetButton.setEnabled(false);
370                            applyButton.setEnabled(false);
371                            return;
372                        }
373                        
374                        final MetadataAttributeValue oldValue = metadataAttributeValueCombos.get(metadataCategory).getMetadataAttributeValue();
375                        final MetadataAttributeValue currentValue = combo.getSelectedMetadataAttributeValue();
376                        final boolean differentAttributeValue = !Objects.equal(oldValue, currentValue);
377    
378                        final MetadataValue metadataValue = metadataFreeTextFields.get(metadataCategory).getMetadataValue();
379                        final String originalText = metadataValue == null ? "" : metadataValue.getValue();
380                        final boolean differentTextValue = !Objects.equal(freeTextField.getText().trim(), originalText);
381                        freeTextField.setToolTipText(freeTextField.getText());
382                        resetButton.setEnabled(differentAttributeValue || differentTextValue);
383                        applyButton.setEnabled(differentTextValue);
384                    }
385                    
386                    @Override
387                    public void widgetDefaultSelected(SelectionEvent e)
388                    {
389                    }
390                });
391            }
392    
393            resetButton.addSelectionListener(new SelectionListener()
394            {
395                @Override
396                public void widgetSelected(final SelectionEvent e)
397                {
398                    final MetadataValue metadataValue = metadataFreeTextFields.get(metadataCategory).getMetadataValue();
399                    final String originalText = metadataValue == null ? "" : metadataValue.getValue();
400                    freeTextField.setText(originalText);
401                    freeTextField.setSelection(originalText.length(), originalText.length());
402                    freeTextField.setToolTipText(freeTextField.getText());
403                    if (combo != null)
404                    {
405                        final MetadataAttributeValue metadataAttributeValue = metadataAttributeValueCombos.get(metadataCategory).getMetadataAttributeValue();
406                        if (metadataAttributeValue != null)
407                        {
408                            combo.select(metadataAttributeValue);
409                        }
410                        else
411                        {
412                            combo.select(0);
413                        }
414                    }
415                    resetButton.setEnabled(false);
416                    applyButton.setEnabled(false);
417                }
418                
419                @Override
420                public void widgetDefaultSelected(final SelectionEvent e)
421                {
422                }
423            });
424    
425            applyButton.addSelectionListener(new SelectionListener()
426            {
427                @Override
428                public void widgetSelected(final SelectionEvent event)
429                {
430                    if (!validMetadataAssignablesSelected(true))
431                    {
432                        return;
433                    }
434    
435                    // Handle the assignment.
436                    if (selectedMetadataAssignables.size() > 1)
437                    {
438                        final boolean performOperation = MessageDialog.openConfirm(getSite().getWorkbenchWindow().getShell(),
439                            "Caution","The metadata operation is about to be performed on all selected items. Are you sure you wish to proceed?");
440                        if (!performOperation)
441                        {
442                            return;
443                        }
444                    }
445    
446                    final String newValue = freeTextField.getText();
447                    final MetadataAttributeValue metadataAttributeValue = combo != null ? combo.getSelectedMetadataAttributeValue() : null;
448    
449                    final MetadataValueValidator validtor = new MetadataValueValidator(Collections.<MetadataValue>emptyList(), true);
450                    if (!validtor.isValid(newValue))
451                    {
452                        MessageDialog.openError(getSite().getWorkbenchWindow().getShell(), "Cannot apply metadata", "Cannot apply metadata. " + validtor.getErrorMessage());
453                        return;
454                    }
455    
456                    // We need to reload the metadata category.
457                    final MetadataCategory updatedMetadataCategory = metadataCategoryService.findById(metadataCategory.getId());
458    
459                    // Disassociate old value
460                    final MetadataValue oldAssociatedValue = freeTextField.getMetadataValue();
461    
462                    if (oldAssociatedValue != null)
463                    {
464                        for (final IMetadataAssignable metadataAssignable : selectedMetadataAssignables)
465                        {
466                            if (metadataAssignable instanceof Group)
467                            {
468                                // We need to reload the object model from the database because the user may have updated it's metadata associations
469                                // and wishes to perform another action on it (without selecting another one in between, which would force a reload from the DB)
470                                final Group freshGroup = groupService.findGroupByID(((Group)metadataAssignable).getId());
471                                groupService.disassociateMetadata(freshGroup, updatedMetadataCategory, oldAssociatedValue);
472                            }
473                            else if (metadataAssignable instanceof ResearchFile)
474                            {
475                                final ResearchFile freshResearchFile = researchFileService.findResearchFileByID(((ResearchFile)metadataAssignable).getId());
476                                researchFileService.disassociateMetadata(freshResearchFile, updatedMetadataCategory, oldAssociatedValue);
477                            }
478                        }
479                    }
480    
481                    // Add & Associate new value
482                    if (!newValue.isEmpty())
483                    {
484                        // Persist the newly configured metadata category value.
485                        final MetadataValue metadataValue = metadataCategoryService.addValueToMetadataCategory(updatedMetadataCategory, newValue);
486                        for (final IMetadataAssignable metadataAssignable : selectedMetadataAssignables)
487                        {
488                            if (metadataAssignable instanceof Group)
489                            {
490                                // We need to reload the object model from the database because the user may have updated it's metadata associations
491                                // and wishes to perform another action on it (without selecting another one in between, which would force a reload from the DB)
492                                final Group freshGroup = groupService.findGroupByID(((Group)metadataAssignable).getId());
493                                groupService.associateMetadata(freshGroup, updatedMetadataCategory, metadataValue, metadataAttributeValue);
494                            }
495                            else if (metadataAssignable instanceof ResearchFile)
496                            {
497                                final ResearchFile freshResearchFile = researchFileService.findResearchFileByID(((ResearchFile)metadataAssignable).getId());
498                                researchFileService.associateMetadata(freshResearchFile, updatedMetadataCategory, metadataValue, metadataAttributeValue);
499                            }
500                        }
501                    }
502                    reloadView();
503                    refreshRelatedViews();
504                }
505                
506                @Override
507                public void widgetDefaultSelected(final SelectionEvent event)
508                {
509                }
510            });
511    
512            plusButton.addSelectionListener(new SelectionListener()
513            {
514                @Override
515                public void widgetSelected(final SelectionEvent e)
516                {
517                }
518    
519                @Override
520                public void widgetDefaultSelected(final SelectionEvent e)
521                {
522                }
523            });
524    
525            minusButton.addSelectionListener(new SelectionListener()
526            {
527                @Override
528                public void widgetSelected(final SelectionEvent e)
529                {
530                }
531    
532                @Override
533                public void widgetDefaultSelected(final SelectionEvent e)
534                {
535                }
536            });
537    
538            freeTextField.addKeyListener(new KeyListener()
539            {
540                @Override
541                public void keyReleased(final KeyEvent e)
542                {
543                    if (!validMetadataAssignablesSelected(false))
544                    {
545                        resetButton.setEnabled(false);
546                        applyButton.setEnabled(false);
547                        return;
548                    }
549    
550                    final MetadataValue metadataValue = metadataFreeTextFields.get(metadataCategory).getMetadataValue();
551                    final String originalText = metadataValue == null ? "" : metadataValue.getValue();
552                    final boolean enabled = !originalText.equals(freeTextField.getText());
553                    freeTextField.setToolTipText(freeTextField.getText());
554                    resetButton.setEnabled(enabled);
555                    applyButton.setEnabled(enabled);
556                }
557                
558                @Override
559                public void keyPressed(final KeyEvent e)
560                {
561                }
562            });
563        }

594                    || commandId.equals("au.org.intersect.exsite9.commands.RemoveMetadataCategoryCommand")

596                    || commandId.equals("au.org.intersect.exsite9.commands.RemoveMetadataCategoryWithWizardCommand"))
597            {
598                reloadView();
599            }

602        private void reloadView()
603        {
604            final IProjectManager projectManager = (IProjectManager) PlatformUI.getWorkbench().getService(IProjectManager.class);
605            final Project project = projectManager.getCurrentProject();
606            if (project != null)
607            {
608                initLayout(project.getSchema().getMetadataCategories());
609    
610                if (!project.getSchema().getMetadataCategories().isEmpty())
611                {
612                    // Refresh this page according to the currently selected items in the RHS.
613                    final ProjectExplorerView projectExplorerView = (ProjectExplorerView) ViewUtils.getViewByID(
614                        PlatformUI.getWorkbench().getActiveWorkbenchWindow(), ProjectExplorerView.ID);
615                    projectExplorerView.refresh();
616                    final ISelection currentSelection = projectExplorerView.getSelection();
617                    selectionChanged(projectExplorerView, currentSelection);
618                }
619            }
620        }

628        @Override
629        public void widgetSelected(final SelectionEvent e)
630        {
631            final MetadataButtonWidget button = (MetadataButtonWidget) e.widget;
632    
633            if (!validMetadataAssignablesSelected(true))
634            {
635                button.setSelection(false);
636                return;
637            }
638    
639            final MetadataCategory metadataCategory = button.getMetadataCategory();
640            final MetadataValue metadataValue = button.getMetadataValue();
641    
642            if (this.selectedMetadataAssignables.size() > 1)
643            {
644                final boolean performOperation = MessageDialog.openConfirm(getSite().getWorkbenchWindow().getShell(),
645                    "Caution","The metadata operation is about to be performed on all selected items. Are you sure you wish to proceed?");
646                if (!performOperation)
647                {
648                    button.setSelection(!button.getSelection());
649                    return;
650                }
651            }
652    
653            if (button.getSelection())
654            {
655                for (final IMetadataAssignable metadataAssignable : this.selectedMetadataAssignables)
656                {
657                    if (metadataAssignable instanceof Group)
658                    {
659                        // We need to reload the object model from the database because the user may have updated it's metadata associations
660                        // and wishes to perform another action on it (without selecting another one in between, which would force a reload from the DB)
661                        final Group freshGroup = this.groupService.findGroupByID(((Group)metadataAssignable).getId());
662                        groupService.associateMetadata(freshGroup, metadataCategory, metadataValue, null);
663                    }
664                    else if (metadataAssignable instanceof ResearchFile)
665                    {
666                        final ResearchFile freshResearchFile = this.researchFileService.findResearchFileByID(((ResearchFile)metadataAssignable).getId());
667                        researchFileService.associateMetadata(freshResearchFile, metadataCategory, metadataValue, null);
668                    }
669                }
670            }
671            else
672            {
673                for (final IMetadataAssignable metadataAssignable : this.selectedMetadataAssignables)
674                {
675                    if (metadataAssignable instanceof Group)
676                    {
677                        final Group freshGroup = this.groupService.findGroupByID(((Group)metadataAssignable).getId());
678                        groupService.disassociateMetadata(freshGroup, metadataCategory, metadataValue);
679                    }
680                    else if (metadataAssignable instanceof ResearchFile)
681                    {
682                        final ResearchFile freshResearchFile = this.researchFileService.findResearchFileByID(((ResearchFile)metadataAssignable).getId());
683                        researchFileService.disassociateMetadata(freshResearchFile, metadataCategory, metadataValue);
684                    }
685                }
686            }
687            refreshRelatedViews();
688        }

690        /**
691         * Ensures the user has selected valid metadata assignables in the project explorer view.
692         * @return {@code true} if selection is valid, {@code false} otherwise.
693         */
694        private boolean validMetadataAssignablesSelected(final boolean showErrors)
695        {
696            // Check that the selected groups does not contain a new files group OR the project node - we CANNOT assign metadata
697            // to them.
698            if (!Collections2.filter(this.selectedMetadataAssignables, Predicates.instanceOf(NewFilesGroup.class)).isEmpty())
699            {
700                if (showErrors)
701                {
702                    MessageDialog.openError(getSite().getWorkbenchWindow().getShell(), "Error", "Metadata cannot be assigned to the new files group.");
703                }
704                return false;
705            }
706            else if (!Collections2.filter(this.selectedMetadataAssignables, Predicates.instanceOf(RootGroup.class)).isEmpty())
707            {
708                if (showErrors)
709                {
710                    MessageDialog.openError(getSite().getWorkbenchWindow().getShell(), "Error", "Metadata cannot be assigned to a project.");
711                }
712                return false;
713            }
714            return true;
715        }

740        /**
741         * Listens to selection changes on the tree viewer in the project view on the LHS.
742         * 
743         * @{inheritDoc
744         */
745        @Override
746        public void selectionChanged(final IWorkbenchPart part, final ISelection selection)
747        {
748            if (!(selection instanceof IStructuredSelection))
749            {
750                LOG.error("Unknown selection type");
751                return;
752            }
753    
754            final IStructuredSelection structuredSelection = (IStructuredSelection) selection;
755    
756            @SuppressWarnings("unchecked")
757            final List<Object> selectedObjects = structuredSelection.toList();
758    
759            this.selectedMetadataAssignables.clear();
760    
761            for (final Object selectedObject : selectedObjects)
762            {
763                if (selectedObject instanceof Group)
764                {
765                    final Group selectedGroup = this.groupService.findGroupByID(((Group) selectedObject).getId());
766                    this.selectedMetadataAssignables.add(selectedGroup);
767                }
768                else if (selectedObject instanceof ResearchFile)
769                {
770                    final ResearchFile selectedResearchFile = this.researchFileService.findResearchFileByID(((ResearchFile) selectedObject).getId());
771                    this.selectedMetadataAssignables.add(selectedResearchFile);
772                }
773                else if (selectedObject instanceof Project)
774                {
775                    Project project = ((Project) selectedObject);
776                    final Group selectedGroup = this.groupService.findGroupByID(project.getRootNode().getId());
777                    this.selectedMetadataAssignables.add(selectedGroup);
778                }
779            }
780    
781            resetMetadataValueWidgets();
782            if (!this.selectedMetadataAssignables.isEmpty())
783            {
784                // Determine a common set of buttons that should be pressed and press them.
785                final Set<Triplet<MetadataCategory, MetadataValue, MetadataAttributeValue>> intersection = new HashSet<Triplet<MetadataCategory, MetadataValue, MetadataAttributeValue>>(
786                        MetadataAssignableUtils.getCategoryToValueMapping(this.selectedMetadataAssignables.get(0)));
787    
788                for (int i = 1; i < this.selectedMetadataAssignables.size(); i++)
789                {
790                    intersection.retainAll(MetadataAssignableUtils.getCategoryToValueMapping(this.selectedMetadataAssignables.get(i)));
791                }
792                setMetadataValueWidgets(intersection);
793            }
794        }

806        /**
807         * Sets the metadata values that are selected.
808         * @param metadataCollectionTrio the metadata to select.
809         */
810        private void setMetadataValueWidgets(final Collection<Triplet<MetadataCategory, MetadataValue, MetadataAttributeValue>> metadataCollectionTrio)
811        {
812            for (final Triplet<MetadataCategory, MetadataValue, MetadataAttributeValue> triplet : metadataCollectionTrio)
813            {
814                final MetadataCategory metadataCategory = triplet.getFirst();
815                if (metadataCategory.getType() == MetadataCategoryType.CONTROLLED_VOCABULARY)
816                {
817                    final MetadataButtonWidget metadataButtonWidget = this.metadataButtons.get(triplet);
818                    if (metadataButtonWidget != null)
819                    {
820                        metadataButtonWidget.setSelection(true);
821                    }
822                }
823                else
824                {
825                    final MetadataValue metadataValue = triplet.getSecond();
826                    final MetadataTextWidget freeTextField = this.metadataFreeTextFields.get(metadataCategory);
827                    freeTextField.setText(metadataValue.getValue());
828                    freeTextField.setMetadataValue(metadataValue);
829                    freeTextField.setToolTipText(freeTextField.getText());
830    
831                    final MetadataAttributeValue metadataAttributeValue = triplet.getThird();
832                    if (metadataAttributeValue != null)
833                    {
834                        final MetadataAttributeValuesComboWidget comboWidget = this.metadataAttributeValueCombos.get(metadataCategory);
835                        comboWidget.select(metadataAttributeValue);
836                        comboWidget.setMetadataAttributeValue(metadataAttributeValue);
837                    }
838                }
839            }
840        }

842        /**
843         * Depresses (i.e. makes them NOT clicked) all the metadata value buttons currently on the page.
844         */
845        private void resetMetadataValueWidgets()
846        {
847            for (final MetadataButtonWidget metadataButtonWidget : this.metadataButtons.values())
848            {
849                metadataButtonWidget.setSelection(false);
850            }
851            for (final MetadataTextWidget freeTextFields : this.metadataFreeTextFields.values())
852            {
853                freeTextFields.setMetadataValue(null);
854                freeTextFields.setText("");
855                freeTextFields.setToolTipText(freeTextFields.getText());
856            }
857            for (final MetadataAttributeValuesComboWidget comboWidget : this.metadataAttributeValueCombos.values())
858            {
859                comboWidget.setMetadataAttributeValue(null);
860                comboWidget.select(0);
861            }
862        }
