007    package au.org.intersect.exsite9.view;
008    
009    import org.eclipse.ui.IViewPart;
010    import org.eclipse.ui.IViewReference;
011    import org.eclipse.ui.IWorkbenchWindow;
012    
013    /**
014     * Used to perform actions on exsite9 views
015     */
016    public final class ViewUtils
017    {
018        public static IViewPart getViewByID(final IWorkbenchWindow window, final String viewID)
019        {
020            final IViewReference[] refs = window.getActivePage().getViewReferences();
021            for (final IViewReference viewRef : refs)
022            {
023                if (viewRef.getId().equals(viewID))
024                {
025                    return viewRef.getView(false);
026                }
027            }
028            return null;
029        }
030    }
