008    package au.org.intersect.exsite9.view.listener;
009    
010    import org.eclipse.jface.util.LocalSelectionTransfer;
011    import org.eclipse.jface.viewers.ITreeSelection;
012    import org.eclipse.jface.viewers.TreeViewer;
013    import org.eclipse.swt.dnd.DragSourceEvent;
014    import org.eclipse.swt.dnd.DragSourceListener;
015    
016    import au.org.intersect.exsite9.domain.NewFilesGroup;
017    import au.org.intersect.exsite9.domain.Project;
018    
019    /**
020     * Custom listener for listening for drag events on the tree viewer of the project view
021     */
022    public class ProjectExplorerDragListener implements DragSourceListener
023    {
024        private final TreeViewer treeViewer;
025        
026        public ProjectExplorerDragListener(TreeViewer treeViewer)
027        {
028            this.treeViewer = treeViewer;
029        }
030        
031        @Override
032        public void dragStart(DragSourceEvent event)
033        {
034            // If we have a single selection we can cancel the drag/drop if the selection is
035            // the project group or the new files group
036            if(((ITreeSelection)treeViewer.getSelection()).size() == 1)
037            {
038                Object obj = ((ITreeSelection)treeViewer.getSelection()).toList().get(0);
039                if ((obj instanceof NewFilesGroup) || (obj instanceof Project))
040                {
041                    event.doit = false;
042                }
043            }
044        }
045    
046        @Override
047        public void dragSetData(DragSourceEvent event)
048        {
049            LocalSelectionTransfer.getTransfer().setSelection(treeViewer.getSelection());
050        }
051    
052        @Override
053        public void dragFinished(DragSourceEvent event)
054        {
055        }
056    }
