009    import org.eclipse.core.commands.Command;
010    import org.eclipse.core.commands.ExecutionEvent;
011    import org.eclipse.core.commands.ExecutionException;
012    import org.eclipse.core.commands.IExecutionListener;
013    import org.eclipse.core.commands.NotHandledException;
014    import org.eclipse.jface.action.MenuManager;
015    import org.eclipse.jface.viewers.TreeViewer;
016    import org.eclipse.swt.SWT;
017    import org.eclipse.swt.widgets.Composite;
018    import org.eclipse.swt.widgets.Menu;
019    import org.eclipse.ui.PlatformUI;
020    import org.eclipse.ui.commands.ICommandService;
021    import org.eclipse.ui.part.ViewPart;
022    
023    import au.org.intersect.exsite9.domain.Project;
024    import au.org.intersect.exsite9.service.IProjectManager;
025    import au.org.intersect.exsite9.view.provider.ProjectViewInputWrapper;
026    import au.org.intersect.exsite9.view.provider.SubmissionPackageBrowserViewContentProvider;
027    import au.org.intersect.exsite9.view.provider.SubmissionPackageBrowserViewLabelProvider;
028    
029    /**
030     * The view that shows the submission packages
031     */
032    public final class SubmissionPackageBrowserView extends ViewPart implements IExecutionListener
033    {
034        public static final String ID = SubmissionPackageBrowserView.class.getName();
035    
036        private TreeViewer treeViewer;
037    
038        /**
039         * Constructor
040         */
041        public SubmissionPackageBrowserView()
042        {
043        }
044    
045        /**
046         * @{inheritDoc
047         */
048        @Override
049        public void createPartControl(final Composite parent)
050        {
051            final ICommandService commandService = (ICommandService) PlatformUI.getWorkbench().getService(
052                    ICommandService.class);
053    
054            final Command createSubmissionPackageCommand = commandService
055                    .getCommand("au.org.intersect.exsite9.commands.CreateSubmissionPackageCommand");
056            createSubmissionPackageCommand.addExecutionListener(this);
057    
058            final Command deleteSubmissionPackageCommand = commandService
059                    .getCommand("au.org.intersect.exsite9.commands.DeleteSubmissionPackageCommand");
060            deleteSubmissionPackageCommand.addExecutionListener(this);
061    
062            final Command openProjectCommand = commandService
063                    .getCommand("au.org.intersect.exsite9.commands.OpenProjectCommand");
064            openProjectCommand.addExecutionListener(this);
065    
066            final Command editProjectCommand = commandService
067                    .getCommand("au.org.intersect.exsite9.commands.EditProjectCommand");
068            editProjectCommand.addExecutionListener(this);
069    
070            final Command newProjectCommand = commandService
071                    .getCommand("au.org.intersect.exsite9.commands.NewProjectCommand");
072            newProjectCommand.addExecutionListener(this);
073    
074            final Command editSubmissionPackageCommand = commandService
075                    .getCommand("au.org.intersect.exsite9.commands.EditSubmissionPackageCommand");
076            editSubmissionPackageCommand.addExecutionListener(this);
077    
078            this.setPartName("Submission Package Browser");
079    
080            this.treeViewer = new TreeViewer(parent, SWT.SINGLE | SWT.H_SCROLL | SWT.V_SCROLL | SWT.BORDER);
081            this.treeViewer.setContentProvider(new SubmissionPackageBrowserViewContentProvider());
082            this.treeViewer.setLabelProvider(new SubmissionPackageBrowserViewLabelProvider());
083    
084            loadCurrentProject();
085    
086            // Initialize the context menu
087            final MenuManager menuManager = new MenuManager();
088            final Menu menu = menuManager.createContextMenu(this.treeViewer.getTree());
089            this.treeViewer.getTree().setMenu(menu);
090            getSite().registerContextMenu(menuManager, this.treeViewer);
091    
092            // This allows other views to listen to selection changes.
093            getSite().setSelectionProvider(this.treeViewer);
094        }
095    
096        private void loadCurrentProject()
097        {
098            final IProjectManager projectManager = (IProjectManager) PlatformUI.getWorkbench().getService(
099                    IProjectManager.class);
100            final Project project = projectManager.getCurrentProject();
101            if (project != null)
102            {
103                final ProjectViewInputWrapper wrapper = new ProjectViewInputWrapper(project);
104                this.treeViewer.setInput(wrapper);
105                this.treeViewer.expandAll();
106            }
107        }
108    
109        /**
110         * @{inheritDoc
111         */
112        @Override
113        public void setFocus()
114        {
115            final MetadataBrowserView metadataBrowserView = (MetadataBrowserView) ViewUtils.getViewByID(PlatformUI
116                    .getWorkbench().getActiveWorkbenchWindow(), MetadataBrowserView.ID);
117            metadataBrowserView.setEnabled(false);
118            
119            final AssociatedMetadataView associatedMetadataView = (AssociatedMetadataView) ViewUtils.getViewByID(PlatformUI
120                    .getWorkbench().getActiveWorkbenchWindow(), AssociatedMetadataView.ID);
121            associatedMetadataView.setEnabled(false);
122        }
123    
124        @Override
125        public void notHandled(final String commandId, final NotHandledException exception)
126        {
127        }
128    
129        @Override
130        public void postExecuteFailure(final String commandId, final ExecutionException exception)
131        {
132        }
133    
134        @Override
135        public void postExecuteSuccess(final String commandId, final Object returnValue)
136        {
137            loadCurrentProject();
138        }
139    
140        @Override
141        public void preExecute(final String commandId, final ExecutionEvent event)
142        {
143        }
144    }
