007    package au.org.intersect.exsite9.view.provider;
008    
009    import org.eclipse.jface.viewers.StyledCellLabelProvider;
010    import org.eclipse.jface.viewers.StyledString;
011    import org.eclipse.jface.viewers.ViewerCell;
012    import org.eclipse.ui.ISharedImages;
013    import org.eclipse.ui.PlatformUI;
014    
015    import au.org.intersect.exsite9.Activator;
016    import au.org.intersect.exsite9.domain.Project;
017    import au.org.intersect.exsite9.domain.SubmissionPackage;
018    
019    /**
020     * 
021     */
022    public final class SubmissionPackageBrowserViewLabelProvider extends StyledCellLabelProvider
023    {
024    
025        public SubmissionPackageBrowserViewLabelProvider()
026        {
027        }
028    
029        /**
030         * @{inheritDoc}
031         */
032        @Override
033        public void update(final ViewerCell cell)
034        {
035            final Object element = cell.getElement();
036            final StyledString text = new StyledString();
037    
038            if (element instanceof Project)
039            {
040                final Project project = (Project) element;
041                text.append(project.getName());
042                text.append(" (" + project.getSubmissionPackages().size() + ")", StyledString.COUNTER_STYLER);
043                cell.setImage(PlatformUI.getWorkbench().getSharedImages().getImage(ISharedImages.IMG_ETOOL_HOME_NAV));
044            }
045            else if (element instanceof SubmissionPackage)
046            {
047                final SubmissionPackage submissionPackage = (SubmissionPackage) element;
048                text.append(submissionPackage.getName());
049    
050                cell.setImage(Activator.getImageDescriptor("/icons/icon_package_16.png").createImage());
051            }
052            else
053            {
054                throw new IllegalArgumentException("Unknown type in submission package tree");
055            }
056    
057            cell.setText(text.toString());
058            cell.setStyleRanges(text.getStyleRanges());
059            super.update(cell);
060        }
061    }
