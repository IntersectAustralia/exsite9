007    package au.org.intersect.exsite9.view.provider;
008    
009    import java.util.ArrayList;
010    import java.util.Collections;
011    import java.util.Comparator;
012    import java.util.List;
013    
014    import org.eclipse.jface.viewers.ITreeContentProvider;
015    import org.eclipse.jface.viewers.Viewer;
016    import org.eclipse.ui.PlatformUI;
017    
018    import com.google.common.base.Predicates;
019    import com.google.common.collect.Collections2;
020    
021    import au.org.intersect.exsite9.domain.Group;
022    import au.org.intersect.exsite9.domain.NewFilesGroup;
023    import au.org.intersect.exsite9.domain.Project;
024    import au.org.intersect.exsite9.domain.ResearchFile;
025    import au.org.intersect.exsite9.domain.ResearchFileSortField;
026    import au.org.intersect.exsite9.domain.SortFieldDirection;
027    import au.org.intersect.exsite9.domain.utils.AlphabeticalGroupComparator;
028    import au.org.intersect.exsite9.domain.utils.ResearchFileExtensionComparator;
029    import au.org.intersect.exsite9.domain.utils.ResearchFileModifiedDataComparator;
030    import au.org.intersect.exsite9.domain.utils.ResearchFileNameComparator;
031    import au.org.intersect.exsite9.domain.utils.ResearchFileSizeComparator;
032    import au.org.intersect.exsite9.service.IGroupService;
033    import au.org.intersect.exsite9.service.IProjectService;
034    import au.org.intersect.exsite9.view.ProjectExplorerView;
035    
036    /**
037     * Provides content to {@link ProjectExplorerView}
038     */
039    public final class ProjectExplorerViewContentProvider implements ITreeContentProvider
040    {
041        private final boolean includeNewFilesGroup;
042        private final IGroupService groupService;
043        private final IProjectService projectService;
044    
045        /**
046         * Constructor
047         */
048        public ProjectExplorerViewContentProvider(final boolean includeNewFilesGroup)
049        {
050            this((IGroupService) PlatformUI.getWorkbench().getService(IGroupService.class), (IProjectService) PlatformUI.getWorkbench().getService(IProjectService.class), includeNewFilesGroup);
051        }
052    
053        ProjectExplorerViewContentProvider(final IGroupService groupService, final IProjectService projectService, final boolean includeNewFilesGroup)
054        {
055            this.groupService = groupService;
056            this.projectService = projectService;
057            this.includeNewFilesGroup = includeNewFilesGroup;
058        }
059    
060        /**
061         * {@inheritDoc}
062         */
063        @Override
064        public void dispose()
065        {
066        }
067    
068        /**
069         * {@inheritDoc}
070         */
071        @Override
072        public void inputChanged(final Viewer viewer, final Object oldInput, final Object newInput)
073        {
074        }
075    
076        /**
077         * {@inheritDoc}
078         */
079        @Override
080        public Object[] getElements(final Object inputElement)
081        {
082            if (inputElement instanceof ProjectViewInputWrapper)
083            {
084                final ProjectViewInputWrapper viewInput = (ProjectViewInputWrapper) inputElement;
085                return new Object[]{this.projectService.findProjectById(viewInput.getProject().getId())};
086            }
087            return Collections.emptyList().toArray();
088        }
089    
090        /**
091         * {@inheritDoc}
092         */
093        @Override
094        public Object[] getChildren(final Object parentElement)
095        {
096            final Group rootGroup;
097            if (parentElement instanceof Project)
098            {
099                final Project project = (Project) parentElement;
100                rootGroup = this.groupService.findGroupByID(project.getRootNode().getId());
101            }
102            else if (parentElement instanceof Group)
103            {
104                rootGroup = this.groupService.findGroupByID(((Group) parentElement).getId());
105            }
106            else
107            {
108                return Collections.emptyList().toArray();
109            }
110    
111            // We include the NewFilesGroup below (if needed)
112            final List<Group> groups = new ArrayList<Group>(Collections2.filter(rootGroup.getGroups(), Predicates.not(Predicates.instanceOf(NewFilesGroup.class))));
113            final List<ResearchFile> researchFiles = new ArrayList<ResearchFile>(rootGroup.getResearchFiles());
114    
115            Collections.sort(groups, new AlphabeticalGroupComparator());
116    
117            final boolean ascending = rootGroup.getResearchFileSortDirection() == SortFieldDirection.ASC;
118    
119            final Comparator<ResearchFile> researchFileComparator;
120            final ResearchFileSortField researchFileSortField = rootGroup.getResearchFileSortField();
121            if (researchFileSortField == ResearchFileSortField.NAME)
122            {
123                researchFileComparator = new ResearchFileNameComparator(ascending);
124            }
125            else if (researchFileSortField == ResearchFileSortField.SIZE)
126            {
127                researchFileComparator = new ResearchFileSizeComparator(ascending);
128            }
129            else if (researchFileSortField == ResearchFileSortField.EXTENSION)
130            {
131                researchFileComparator = new ResearchFileExtensionComparator(ascending);
132            }
133            else
134            {
135                // researchFileSortField == ResearchFileSortField.MODIFIED_DATE
136                researchFileComparator = new ResearchFileModifiedDataComparator(ascending);
137            }
138    
139            Collections.sort(researchFiles, researchFileComparator);
140    
141            final List<Object> toReturn = new ArrayList<Object>();
142            toReturn.addAll(groups);
143            toReturn.addAll(researchFiles);
144    
145            // Add the new files group last
146            if (this.includeNewFilesGroup)
147            {
148                toReturn.addAll(Collections2.filter(rootGroup.getGroups(), Predicates.instanceOf(NewFilesGroup.class)));
149            }
150    
151            return toReturn.toArray();
152        }
153    
154        /**
155         * {@inheritDoc}
156         */
157        @Override
158        public Object getParent(final Object element)
159        {
160            if (element instanceof Group)
161            {
162                final Group group = (Group) element;
163                return group.getParentGroup();
164            }
165            if (element instanceof ResearchFile)
166            {
167                final ResearchFile researchFile = (ResearchFile) element;
168                return researchFile.getParentGroup();
169            }
170            return null;
171        }
172    
173        /**
174         * {@inheritDoc}
175         */
176        @Override
177        public boolean hasChildren(final Object element)
178        {
179            if (element instanceof Project)
180            {
181                final Project project = (Project) element;
182                final Group group = project.getRootNode();
183                return !group.getGroups().isEmpty() || !group.getResearchFiles().isEmpty();
184            }
185            if (element instanceof Group)
186            {
187                final Group group = (Group) element;
188                return !group.getGroups().isEmpty() || !group.getResearchFiles().isEmpty();
189            }
190            return false;
191        }
192    }
