007    package au.org.intersect.exsite9.view.provider;
008    
009    import java.io.File;
010    
011    import org.eclipse.jface.viewers.StyledCellLabelProvider;
012    import org.eclipse.jface.viewers.StyledString;
013    import org.eclipse.jface.viewers.ViewerCell;
014    import org.eclipse.swt.graphics.Point;
015    import org.eclipse.ui.ISharedImages;
016    import org.eclipse.ui.PlatformUI;
017    
018    import au.org.intersect.exsite9.Activator;
019    import au.org.intersect.exsite9.domain.Group;
020    import au.org.intersect.exsite9.domain.NewFilesGroup;
021    import au.org.intersect.exsite9.domain.Project;
022    import au.org.intersect.exsite9.domain.ResearchFile;
023    import au.org.intersect.exsite9.view.ProjectExplorerView;
024    
025    /**
026     * Provides labels to {@link ProjectExplorerView}
027     */
028    public final class ProjectExplorerViewLabelProvider extends StyledCellLabelProvider
029    {
030        private static final String NEW_LINE = System.getProperty("line.separator");
031    
032        /**
033         * Constructor
034         */
035        public ProjectExplorerViewLabelProvider()
036        {
037        }
038    
039        /**
040         * @{inheritDoc
041         */
042        @Override
043        public void update(final ViewerCell cell)
044        {
045            final Object element = cell.getElement();
046            final StyledString text = new StyledString();
047    
048            if (element instanceof Project)
049            {
050                final Project project = (Project) element;
051                text.append(project.getName());
052                cell.setImage(PlatformUI.getWorkbench().getSharedImages().getImage(ISharedImages.IMG_ETOOL_HOME_NAV));
053            }
054            else if (element instanceof Group)
055            {
056                final Group group = (Group) element;
057                if (group instanceof NewFilesGroup)
058                {
059                    text.append(group.getName(), StyledString.QUALIFIER_STYLER);
060                }
061                else
062                {
063                    text.append(group.getName());
064                }
065                cell.setImage(PlatformUI.getWorkbench().getSharedImages().getImage(ISharedImages.IMG_OBJ_FOLDER));
066                text.append(" (" + (group.getGroups().size() + group.getResearchFiles().size()) + ")",
067                        StyledString.COUNTER_STYLER);
068            }
069            else if (element instanceof ResearchFile)
070            {
071                final ResearchFile rf = (ResearchFile) element;
072                final File file = rf.getFile();
073                text.append(file.getName());
074    
075                if (!file.exists())
076                {
077                    cell.setImage(PlatformUI.getWorkbench().getSharedImages().getImage(ISharedImages.IMG_TOOL_DELETE));
078                }
079                else if (rf.getMetadataAssociations().isEmpty())
080                {
081                    cell.setImage(Activator.getImageDescriptor("/icons/icon_warning_12.png").createImage());
082                }
083                else if (rf.isMissingRequiredMetadata())
084                {
085                    cell.setImage(Activator.getImageDescriptor("/icons/icon_warning_12.png").createImage());
086                }
087                else
088                {
089                    cell.setImage(PlatformUI.getWorkbench().getSharedImages().getImage(ISharedImages.IMG_OBJ_FILE));
090                }
091                
092                text.append("    [" + rf.getDisplayDate() + "]");
093                text.append("    [" + rf.getDisplaySize() + "]");
094            }
095            else
096            {
097                throw new IllegalArgumentException("Unknown type in project explorer tree");
098            }
099    
100            cell.setText(text.toString());
101            cell.setStyleRanges(text.getStyleRanges());
102            super.update(cell);
103        }
104    
105        /**
106         * @{inheritDoc
107         */
108        @Override
109        public String getToolTipText(final Object element)
110        {
111            if (!(element instanceof ResearchFile))
112            {
113                return null;
114            }
115    
116            final ResearchFile rf = (ResearchFile) element;
117            final File file = rf.getFile();
118    
119            final StringBuilder sb = new StringBuilder();
120    
121            if (!file.exists())
122            {
123                sb.append("THIS FILE IS MISSING").append(NEW_LINE);
124            }
125            if (rf.getMetadataAssociations().isEmpty())
126            {
127                sb.append("THIS FILE HAS NO METADATA ASSOCIATIONS").append(NEW_LINE);
128            }
129            else if (rf.isMissingRequiredMetadata())
130            {
131                sb.append("THIS FILE IS MISSING REQUIRED METADATA").append(NEW_LINE);
132            }
133            sb.append(rf.getFile().getAbsolutePath());
134            return sb.toString();
135        }
136    
137        /**
138         * @{inheritDoc
139         */
140        @Override
141        public int getToolTipTimeDisplayed(final Object object)
142        {
143            return 10 * 1000;
144        }
145    
146        /**
147         * @{inheritDoc
148         */
149        @Override
150        public int getToolTipDisplayDelayTime(final Object object)
151        {
152            return 200;
153        }
154    
155        /**
156         * @{inheritDoc
157         */
158        public Point getToolTipShift(final Object object)
159        {
160            return new Point(5, 5);
161        }
162    }
