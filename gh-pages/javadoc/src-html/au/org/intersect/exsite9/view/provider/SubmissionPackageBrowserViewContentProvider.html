007    package au.org.intersect.exsite9.view.provider;
008    
009    import java.util.ArrayList;
010    import java.util.Collections;
011    import java.util.List;
012    
013    import org.eclipse.jface.viewers.ITreeContentProvider;
014    import org.eclipse.jface.viewers.Viewer;
015    import org.eclipse.ui.PlatformUI;
016    
017    import au.org.intersect.exsite9.domain.Project;
018    import au.org.intersect.exsite9.domain.SubmissionPackage;
019    import au.org.intersect.exsite9.service.IProjectService;
020    import au.org.intersect.exsite9.service.ISubmissionPackageService;
021    import au.org.intersect.exsite9.view.SubmissionPackageBrowserView;
022    
023    /**
024     * Provides content to {@link SubmissionPackageBrowserView}
025     */
026    public final class SubmissionPackageBrowserViewContentProvider implements ITreeContentProvider
027    {
028        private final IProjectService projectService;
029        private final ISubmissionPackageService submissionPackageService;
030    
031        public SubmissionPackageBrowserViewContentProvider()
032        {
033            this((IProjectService) PlatformUI.getWorkbench().getService(IProjectService.class), (ISubmissionPackageService) PlatformUI.getWorkbench().getService(ISubmissionPackageService.class));
034        }
035    
036        SubmissionPackageBrowserViewContentProvider(final IProjectService projectService, final ISubmissionPackageService submissionPackageService)
037        {
038            this.projectService = projectService;
039            this.submissionPackageService = submissionPackageService;
040        }
041    
042    
043        @Override
044        public void dispose()
045        {
046        }
047    
048        @Override
049        public void inputChanged(final Viewer viewer, final Object oldInput, final Object newInput)
050        {
051        }
052    
053        @Override
054        public Object[] getElements(final Object inputElement)
055        {
056            if (inputElement instanceof ProjectViewInputWrapper)
057            {
058                final ProjectViewInputWrapper viewInput = (ProjectViewInputWrapper) inputElement;
059                return new Object[]{this.projectService.findProjectById(viewInput.getProject().getId())};
060            }
061            return Collections.emptyList().toArray();
062        }
063    
064        @Override
065        public Object[] getChildren(final Object parentElement)
066        {
067            if (parentElement instanceof Project)
068            {
069                final Project project = (Project) parentElement;
070                final List<SubmissionPackage> toReturn = new ArrayList<SubmissionPackage>(project.getSubmissionPackages().size());
071                for (final SubmissionPackage subPack : project.getSubmissionPackages())
072                {
073                    toReturn.add(this.submissionPackageService.findSubmissionPackageById(subPack.getId()));
074                }
075                return toReturn.toArray();
076            }
077            return Collections.emptyList().toArray();
078        }
079    
080        @Override
081        public Object getParent(final Object element)
082        {
083            return null;
084        }
085    
086        @Override
087        public boolean hasChildren(final Object element)
088        {
089            if (element instanceof Project)
090            {
091                final Project project = (Project) element;
092                return !project.getSubmissionPackages().isEmpty();
093            }
094            return false;
095        }
096    }
