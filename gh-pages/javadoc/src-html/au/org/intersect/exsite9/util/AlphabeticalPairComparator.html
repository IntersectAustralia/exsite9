007    package au.org.intersect.exsite9.util;
008    
009    import java.util.Comparator;
010    
011    /**
012     * Comparator to compare a pair of Strings alphabetically.
013     */
014    public final class AlphabeticalPairComparator implements Comparator<Pair<String, String>>
015    {
016        /**
017         * @{inheritDoc}
018         */
019        @Override
020        public int compare(final Pair<String, String> pair1, final Pair<String, String> pair2)
021        {
022            final int result = compareStrings(pair1.getFirst(), pair2.getFirst());
023            return result == 0 ? compareStrings(pair1.getSecond(), pair2.getSecond()) : result;
024        }
025    
026        private int compareStrings(final String s1, final String s2)
027        {
028            return s1.compareToIgnoreCase(s2);
029        }
030    }
