007    package au.org.intersect.exsite9.util;
008    
009    import java.awt.Desktop;
010    import java.io.File;
011    import java.io.IOException;
012    
013    /**
014     * Utilities for the Desktop.
015     */
016    public final class DesktopUtils
017    {
018        /**
019         * Opens the provided file with it's default application configured on the system.
020         * Does nothing if it is not supported on the current platform.
021         * @param file The file to open.
022         * @throws IOException If the file could not be opened.
023         */
024        public static void openWithDefaultApplication(final File file) throws IOException
025        {
026            if (!Desktop.isDesktopSupported())
027            {
028                return;
029            }
030    
031            try
032            {
033                Desktop.getDesktop().open(file);
034            }
035            catch (final IOException e)
036            {
037                // See http://bugs.sun.com/bugdatabase/view_bug.do?bug_id=6780505
038                Desktop.getDesktop().browse(file.toURI());
039            }
040        }
041    }
