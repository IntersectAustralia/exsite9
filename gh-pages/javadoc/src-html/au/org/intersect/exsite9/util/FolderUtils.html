001    package au.org.intersect.exsite9.util;
002    
003    import java.io.File;
004    import java.util.ArrayList;
005    import java.util.List;
006    
007    import org.apache.commons.io.FileUtils;
008    import org.apache.commons.io.filefilter.AgeFileFilter;
009    import org.apache.commons.io.filefilter.HiddenFileFilter;
010    import org.apache.commons.io.filefilter.IOFileFilter;
011    import org.apache.commons.io.filefilter.TrueFileFilter;
012    
013    import au.org.intersect.exsite9.domain.Folder;
014    
015    public final class FolderUtils
016    {
017        private FolderUtils()
018        {
019            // No instances please!
020        }
021    
022        /**
023         * Recursively looks for files in the folder & it's sub folders that are new since the last time
024         * it checked.
025         * Returns an empty list if there are no files or this is not a folder.
026         * Time resolution of ext3 & hfs+ is 1 second so when we store the time of the
027         * last check we round it down. 
028         * @return A list of the files identified in the folder since the last time the folder was checked
029         */
030        public static List<File> identifyNewFiles(final Folder folder)
031        {
032            final IOFileFilter ageFilter = new AgeFileFilter(folder.getLastCheckTimeInMillis(),false);
033            return listFiles(folder, ageFilter, TrueFileFilter.INSTANCE);
034        }
035    
036        /**
037         * Recursively looks for files in the folder & it's sub folders.
038         * Returns an empty list if there are no files or this is not a folder.
039         * 
040         * @return A list of the files in the folder
041         */
042        public static List<File> getAllFilesInFolder(final Folder folder)
043        {
044            return listFiles(folder, HiddenFileFilter.VISIBLE, TrueFileFilter.INSTANCE);
045        }
046    
047        private static List<File> listFiles(final Folder folder, final IOFileFilter fileFilter, final IOFileFilter dirFilter)
048        {
049            final List<File> newFileList = new ArrayList<File>(FileUtils.listFiles(folder.getFolder(), fileFilter, dirFilter));
050            folder.setLastCheckTimeInMillis(System.currentTimeMillis());
051            return newFileList;
052        }
053    }
