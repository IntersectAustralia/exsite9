001    package au.org.intersect.exsite9.util;
002    
003    import java.util.regex.Pattern;
004    
005    /**
006     * Provides utilities for Directories.
007     */
008    public final class DirectoryUtils
009    {
010        private DirectoryUtils()
011        {
012            // No instances please!
013        }
014    
015        /**
016         * Regex to check for illegal characters in directory names.
017         * Illegal characters are: \ / : * " ? < > |
018         */
019        private static final Pattern ILLEGAL_DIRNAME_REGEX = Pattern.compile("^.*[\\\\/:\\*\"\\?<>\\|].*");
020    
021        /**
022         * Determines if the provided String is a valid directory name.
023         * It will ensure the dirName is valid on Linux, Mac and Windows.
024         * @param dirName The String to check.
025         * @return {@code true} if the String can be used as a valid directory name.
026         */
027        public static boolean isValidDirectoryName(final String dirName)
028        {
029            return !ILLEGAL_DIRNAME_REGEX.matcher(dirName).matches();
030        }
031    }
