007    package au.org.intersect.exsite9.util;
008    
009    import java.util.regex.Pattern;
010    
011    /**
012     * Provides utilities for Directories.
013     */
014    public final class DirectoryUtils
015    {
016        private DirectoryUtils()
017        {
018            // No instances please!
019        }
020    
021        /**
022         * Regex to check for illegal characters in directory names.
023         * Illegal characters are: \ / : * " ? < > |
024         */
025        private static final Pattern ILLEGAL_DIRNAME_REGEX = Pattern.compile("^.*[\\\\/:\\*\"\\?<>\\|].*");
026    
027        /**
028         * Determines if the provided String is a valid directory name.
029         * It will ensure the dirName is valid on Linux, Mac and Windows.
030         * @param dirName The String to check.
031         * @return {@code true} if the String can be used as a valid directory name.
032         */
033        public static boolean isValidDirectoryName(final String dirName)
034        {
035            return !ILLEGAL_DIRNAME_REGEX.matcher(dirName).matches();
036        }
037    }
