007    package au.org.intersect.exsite9.util;
008    
009    import org.apache.commons.lang3.builder.HashCodeBuilder;
010    import org.apache.commons.lang3.builder.ToStringBuilder;
011    
012    import com.google.common.base.Objects;
013    
014    /**
015     * 
016     */
017    public final class Triplet<F, S, T> extends Pair<F, S>
018    {
019        private final T third;
020    
021        public Triplet(final F first, final S second, final T third)
022        {
023            super(first, second);
024            this.third = third;
025        }
026    
027        public T getThird()
028        {
029            return third;
030        }
031    
032        @Override
033        public boolean equals(final Object obj)
034        {
035            if (obj == this)
036            {
037                return true;
038            }
039            if (!(obj instanceof Triplet<?, ?, ?>))
040            {
041                return false;
042            }
043            final Triplet<?, ?, ?> other = (Triplet<?, ?, ?>) obj;
044            return Objects.equal(this.third, other.third) && super.equals(obj);
045        }
046    
047        @Override
048        public int hashCode()
049        {
050            return new HashCodeBuilder().appendSuper(super.hashCode()).append(this.third).toHashCode();
051        }
052    
053        @Override
054        public String toString()
055        {
056            return new ToStringBuilder(this).appendSuper(super.toString()).append("third", this.third).toString();
057        }
058    }
