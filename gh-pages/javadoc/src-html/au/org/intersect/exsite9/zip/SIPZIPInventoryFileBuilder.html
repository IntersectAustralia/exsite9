007    package au.org.intersect.exsite9.zip;
008    
009    import java.text.SimpleDateFormat;
010    import java.util.Date;
011    import java.util.List;
012    
013    import au.org.intersect.exsite9.domain.Group;
014    import au.org.intersect.exsite9.domain.Project;
015    import au.org.intersect.exsite9.domain.ResearchFile;
016    import au.org.intersect.exsite9.domain.SubmissionPackage;
017    
018    /**
019     * Used to create an inventory file for the ZIP file
020     */
021    public class SIPZIPInventoryFileBuilder
022    {
023        
024        
025        public static String buildInventoryFile(final Project project, final SubmissionPackage submissionPackage)
026        {        
027            StringBuffer result = new StringBuffer();
028            final List<ResearchFile> selectedFiles = submissionPackage.getResearchFiles();
029            
030            for (ResearchFile researchFile : selectedFiles)
031            {
032                StringBuffer fileDetails = new StringBuffer();
033                
034                Group parent = researchFile.getParentGroup();
035                
036                fileDetails.append(researchFile.getFile().getName());
037                
038                while(!parent.equals(project.getRootNode()))
039                {
040                    //insert parent name at the front of the path.
041                    fileDetails.insert(0, parent.getName() + "/");              
042                    parent = parent.getParentGroup();
043                }
044                
045                fileDetails.append(" | ");                      
046                fileDetails.append(Long.toString(researchFile.getFile().length()) + " Bytes");
047                if (!researchFile.getFile().exists())
048                {
049                    fileDetails.append(" (MISSING)");
050                }
051               
052                fileDetails.append(" | ");
053                long lastModifiedStamp = researchFile.getFile().lastModified();
054                SimpleDateFormat dateFormat = new SimpleDateFormat("dd/MM/yyyy HH:mm:ss");
055                Date dateStamp = new Date(lastModifiedStamp);
056                fileDetails.append(dateFormat.format(dateStamp));
057                             
058                fileDetails.append(System.getProperty("line.separator"));
059                fileDetails.append(System.getProperty("line.separator"));
060                
061                result.append(fileDetails.toString());
062            }
063            
064            return result.toString();
065        }
066    }
