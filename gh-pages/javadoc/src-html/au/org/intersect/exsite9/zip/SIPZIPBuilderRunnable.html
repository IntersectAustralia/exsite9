039    /**
040     * Used to build the ZIP file that contains the submission information package of a project
041     */
042    public final class SIPZIPBuilderRunnable implements IRunnableWithProgress
043    {
044        private static final int BUF_SIZE = 0x1000; // 4K
045    
046        private static final Logger LOG = Logger.getLogger(SIPZIPBuilderRunnable.class);
047        private static final String PROGRESS_MESSAGE = "Building ZIP file. This can take some time.";
048    
049        private final Project project;
050        private final List<Group> selectedGroups;
051        private final SubmissionPackage submissionPackage;
052        private final File destinationFile;
053        private IProgressMonitor progressMonitor;
054    
055        public SIPZIPBuilderRunnable(final Project project, final List<Group> selectedGroups, final SubmissionPackage submissionPackage, final File destinationFile)
056        {
057            this.project = project;
058            this.selectedGroups = selectedGroups;
059            this.submissionPackage = submissionPackage;
060            this.destinationFile = destinationFile;
061        }
062    
063        /**
064         * @{inheritDoc}
065         */
066        @Override
067        public void run(final IProgressMonitor monitor) throws InvocationTargetException, InterruptedException
068        {
069            this.progressMonitor = monitor;
070            
071            // Determine the size of all files we need to compress - so we can provide progress.
072            long totalSizeBytes = 0;
073            for (final ResearchFile researchFile : this.submissionPackage.getResearchFiles())
074            {
075                final File file = researchFile.getFile();
076                if (file.exists())
077                {
078                    totalSizeBytes += file.length();
079                }
080            }
081            
082            // Throw an exception if we don't have enough room on the device
083            FileSystemView fsv = FileSystemView.getFileSystemView();
084            
085            File destinationRoot = destinationFile.getParentFile();
086            while(destinationRoot.getParentFile() != null)
087            {
088                if( (destinationRoot.getUsableSpace() > 0) || (fsv.isFileSystemRoot(destinationRoot))){
089                    break;
090                }
091                destinationRoot = destinationRoot.getParentFile();
092            }        
093            
094            if((destinationRoot.getUsableSpace() > 0) || (fsv.isFileSystemRoot(destinationRoot)))
095            {
096                long freeSpaceBytes = destinationRoot.getUsableSpace();
097                LOG.info("Device has " + freeSpaceBytes + " bytes available.");
098                
099                if(totalSizeBytes > freeSpaceBytes)
100                {
101                    LOG.info("No room for ZIP file.");
102                    String msg = "Not enough free space on device " + destinationRoot.getAbsolutePath();
103                    NotEnoughSpaceForZIPException nesfze = new NotEnoughSpaceForZIPException(msg);
104                    throw new InvocationTargetException(nesfze);
105                }
106            }
107            
108            final int totalSizeKiloBytes = (int) (totalSizeBytes / 1024);
109            this.progressMonitor.beginTask(PROGRESS_MESSAGE, totalSizeKiloBytes);
110    
111            try
112            {
113                LOG.info("Writing ZIP file " + destinationFile.getAbsolutePath());
114    
115                final ZipArchiveOutputStream zipStream = new ZipArchiveOutputStream(destinationFile);
116                zipStream.setUseZip64(Zip64Mode.AsNeeded);
117                final WritableByteChannel zipByteChannel = Channels.newChannel(zipStream);
118    
119                try
120                {
121                    // Add the groups and files to the zip
122                    final String rootFolder = submissionPackage.getName();
123                    createDirForGroup(rootFolder, project.getRootNode(), zipStream, zipByteChannel, "", selectedGroups, submissionPackage);
124    
125                    // Put the SIP XML in place.
126                    final String xml = SIPXMLBuilder.buildXML(project, selectedGroups, submissionPackage, true);
127    
128                    final ZipArchiveEntry sipXMLZipEntry = new ZipArchiveEntry(rootFolder + "/" + submissionPackage.getName() + ".xml");
129                    zipStream.putArchiveEntry(sipXMLZipEntry);
130        
131                    final ReadableByteChannel sipXmlByteChannel = Channels.newChannel(new ByteArrayInputStream(xml.getBytes(Charsets.UTF_8)));
132                    try
133                    {
134                        ByteStreams.copy(sipXmlByteChannel, zipByteChannel);
135                    }
136                    finally
137                    {
138                        zipStream.closeArchiveEntry();
139                        sipXmlByteChannel.close();
140                    }
141        
142                    // Also put the SIP Inventory in place.
143                    final String inventoryInput = SIPZIPInventoryFileBuilder.buildInventoryFile(project, submissionPackage);
144                    final ZipArchiveEntry inventoryFileZipEntry = new ZipArchiveEntry(rootFolder + "/" + submissionPackage.getName() + "_Inventory.txt");
145                    zipStream.putArchiveEntry(inventoryFileZipEntry);
146                    final ReadableByteChannel inventoryChannel = Channels.newChannel(new ByteArrayInputStream(inventoryInput.getBytes(Charsets.UTF_8)));
147    
148                    try
149                    {
150                        ByteStreams.copy(inventoryChannel, zipByteChannel);
151                    }
152                    finally
153                    {
154                        zipStream.closeArchiveEntry();
155                        inventoryChannel.close();
156                    }
157                }
158                catch (final InterruptedException e)
159                {
160                    // User has requested the operation to cancel. Delete the partially built file.
161                    zipStream.close();
162                    this.destinationFile.delete();
163                    throw e;
164                }
165                finally
166                {
167                    zipStream.close();
168                }
169                LOG.info("Completed writing ZIP file " + destinationFile.getAbsolutePath());
170            }
171            catch (final IOException e)
172            {
173                LOG.error("Could not create ZIP file " + destinationFile.getAbsolutePath(), e);
174                throw new InvocationTargetException(e);
175            }
176            finally
177            {
178                this.progressMonitor.done();
179            }
180        }
181    
182        private void createDirForGroup(final String dirName, final Group group, final ZipArchiveOutputStream zipStream, final WritableByteChannel zipByteChannel, final String parentPath, final List<Group> selectedGroups, final SubmissionPackage submissionPackage) throws IOException, InterruptedException
183        {
184            final String groupPath = parentPath + dirName + "/";
185            final ZipArchiveEntry groupDirZipEntry = new ZipArchiveEntry(groupPath);
186            zipStream.putArchiveEntry(groupDirZipEntry);
187            zipStream.closeArchiveEntry();
188    
189            copyResearchFiles(group, zipStream, zipByteChannel, groupPath, submissionPackage);
190    
191            for (final Group child : group.getGroups())
192            {
193                if (selectedGroups.contains(child))
194                {
195                    createDirForGroup(child.getName(), child, zipStream, zipByteChannel, groupPath, selectedGroups, submissionPackage);
196                }
197            }
198        }
199    
200        private void copyResearchFiles(final Group group, final ZipArchiveOutputStream zipStream, final WritableByteChannel zipByteChannel, final String parentPath, final SubmissionPackage submissionPackage) throws IOException, InterruptedException
201        {
202            for (final ResearchFile researchFile : group.getResearchFiles())
203            {
204                if (submissionPackage.getResearchFiles().contains(researchFile))
205                {
206                    final File theFile = researchFile.getFile();
207                    if (!theFile.exists())
208                    {
209                        continue;
210                    }
211    
212                    final ZipArchiveEntry zipEntry = new ZipArchiveEntry(parentPath + theFile.getName());
213                    zipEntry.setSize(theFile.length());
214                    zipStream.putArchiveEntry(zipEntry);
215    
216                    final ReadableByteChannel fileByteChannel = Channels.newChannel(new FileInputStream(theFile));
217                    try
218                    {
219                        copy(fileByteChannel, zipByteChannel);
220                    }
221                    finally
222                    {
223                        zipStream.closeArchiveEntry();
224                        fileByteChannel.close();
225                    }
226                }
227            }
228        }
229    
230        /**
231         * Copied from Google Guava {@link ByteStreams#copy(ReadableByteChannel, WritableByteChannel)} and modified to provide progress feedback and the ability to cancel
232         * mid-way through a copy.
233         *
234         * Copies all bytes from the readable channel to the writable channel.
235         * Does not close or flush either channel.
236         * @param from the readable channel to read from
237         * @param to the writable channel to write to
238         * @return the number of bytes copied
239         * @throws IOException if an I/O error occurs
240         */
241        private long copy(final ReadableByteChannel from, final WritableByteChannel to) throws IOException, InterruptedException
242        {
243            final ByteBuffer buf = ByteBuffer.allocate(BUF_SIZE);
244            long total = 0;
245            while (from.read(buf) != -1)
246            {
247                buf.flip();
248                while (buf.hasRemaining())
249                {
250                    final int bytesRead = to.write(buf);
251                    total += bytesRead;
252                    this.progressMonitor.worked(bytesRead / 1024);
253                    if (this.progressMonitor.isCanceled())
254                    {
255                        throw new InterruptedException("User Cancelled");
256                    }
257                }
258                buf.clear();
259            }
260            return total;
261        }
