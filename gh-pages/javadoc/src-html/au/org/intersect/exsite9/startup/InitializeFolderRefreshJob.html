007    package au.org.intersect.exsite9.startup;
008    
009    import org.eclipse.ui.IStartup;
010    
011    import au.org.intersect.exsite9.Activator;
012    import au.org.intersect.exsite9.initialization.ConfigurationInitializer;
013    import au.org.intersect.exsite9.jobs.IdentifyAllNewFilesForProjectRepeatingJob;
014    
015    /**
016     * Initializes the folder refresh job.
017     */
018    public class InitializeFolderRefreshJob implements IStartup
019    {
020    
021        /**
022         * @{inheritDoc}
023         */
024        @Override
025        public void earlyStartup()
026        {
027            // Start the scheduled find files job
028            final int refreshTimeMins = Activator.getDefault().getPreferenceStore().getInt(ConfigurationInitializer.FOLDER_RELOAD_INTERVAL_KEY);
029            final long delayMillis = refreshTimeMins * 60 * 1000;
030            final IdentifyAllNewFilesForProjectRepeatingJob findFilesJob = new IdentifyAllNewFilesForProjectRepeatingJob(delayMillis);
031    
032            if (refreshTimeMins != 0)
033            {
034                findFilesJob.schedule(delayMillis);
035            }
036        }
037    }
