007    package au.org.intersect.exsite9.dto;
008    
009    /**
010     * Data Transfer Object to cut down on the parameters that have to be passed around when a node in the tree is moved (dragged and dropped)
011     */
012    public final class HierarchyMoveDTO
013    {
014        private final Object child;
015        private final Object oldParent;
016        private final Object newParent;
017        
018        public HierarchyMoveDTO(Object selection, Object oldParent, Object newParent)
019        {
020            this.child = selection;
021            this.oldParent = oldParent;
022            this.newParent = newParent;
023        }
024    
025        public Object getChild()
026        {
027            return child;
028        }
029    
030        public Object getOldParent()
031        {
032            return oldParent;
033        }
034    
035        public Object getNewParent()
036        {
037            return newParent;
038        }
039        
040    }
