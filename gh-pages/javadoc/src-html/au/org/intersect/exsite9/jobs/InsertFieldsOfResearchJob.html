007    package au.org.intersect.exsite9.jobs;
008    
009    import java.io.BufferedReader;
010    import java.io.File;
011    import java.io.FileReader;
012    import java.io.IOException;
013    import java.sql.Connection;
014    import java.sql.ResultSet;
015    import java.sql.SQLException;
016    import java.sql.SQLIntegrityConstraintViolationException;
017    import java.sql.Statement;
018    import java.util.ArrayList;
019    import java.util.List;
020    
021    import javax.persistence.EntityManager;
022    import javax.persistence.EntityManagerFactory;
023    
024    import org.apache.log4j.Logger;
025    import org.eclipse.core.runtime.IProgressMonitor;
026    import org.eclipse.core.runtime.IStatus;
027    import org.eclipse.core.runtime.Platform;
028    import org.eclipse.core.runtime.Status;
029    import org.eclipse.core.runtime.jobs.Job;
030    import org.eclipse.persistence.internal.jpa.EntityManagerImpl;
031    import org.eclipse.ui.PlatformUI;
032    
033    /**
034     * Job used to insert the fields of research codes into the database.
035     */
036    public final class InsertFieldsOfResearchJob extends Job
037    {
038        private static final Logger LOG = Logger.getLogger(InsertFieldsOfResearchJob.class);
039        public static final String JOB_NAME = "Loading Fields of Research";
040    
041        private final EntityManagerFactory emf;
042        private final File sqlFile;
043    
044        /**
045         * Constructor
046         */
047        public InsertFieldsOfResearchJob()
048        {
049            super(JOB_NAME);
050            this.emf = (EntityManagerFactory) PlatformUI.getWorkbench().getService(EntityManagerFactory.class);
051            final String workspaceDir = Platform.getInstallLocation().getURL().getPath();
052            final File configurationDir = new File(workspaceDir, "configuration");
053            this.sqlFile = new File(configurationDir, "fieldsOfResearch.sql");
054        }
055    
056        /**
057         * Constructor for use in unit tests only.
058         * @param emf
059         * @param sqlFile
060         */
061        InsertFieldsOfResearchJob(final EntityManagerFactory emf, final File sqlFile)
062        {
063            super(JOB_NAME);
064            this.emf = emf;
065            this.sqlFile = sqlFile;
066        }
067    
068    
069        @Override
070        protected IStatus run(final IProgressMonitor monitor)
071        {
072            monitor.beginTask("Inserting Fields of Research", IProgressMonitor.UNKNOWN);
073            try
074            {
075                final List<String> statements = new ArrayList<String>();
076                try
077                {
078                    final BufferedReader br = new BufferedReader(new FileReader(this.sqlFile));
079                    String line;
080                    while ((line = br.readLine()) != null)
081                    {
082                        statements.add(line);
083                    }
084                }
085                catch (final IOException e)
086                {
087                    LOG.error("Could not read file " + this.sqlFile.getAbsolutePath(), e);
088                    return Status.CANCEL_STATUS;
089                }
090    
091                final EntityManager em = this.emf.createEntityManager();
092                final Connection connection = ((EntityManagerImpl)em.getDelegate()).getServerSession().getAccessor().getConnection();
093                try
094                {
095                    final Statement countStatement = connection.createStatement();
096                    final ResultSet rs = countStatement.executeQuery("SELECT COUNT(*) FROM FIELDOFRESEARCH");
097                    countStatement.close();
098                    rs.next();
099    
100                    if (rs.getLong(1) != statements.size())
101                    {
102                        LOG.info("Inserting field of research codes");
103    
104                        final Statement stmt = connection.createStatement();
105                        int successInserts = 0;
106                        int alreadyExists = 0;
107                        for (final String statement : statements)
108                        {
109                            try
110                            {
111                                stmt.execute(statement);
112                                successInserts++;
113                            }
114                            catch (final SQLIntegrityConstraintViolationException e)
115                            {
116                                alreadyExists++;
117                                // We already have the Field of Research we are trying to insert.
118                            }
119                        }
120    
121                        LOG.info("Insertion of field of research codes complete. Attempted: " + statements.size() + " Successful: " + successInserts + " Skipped: " + alreadyExists);
122                        stmt.close();
123                    }
124                    else
125                    {
126                        LOG.info("Database contains " + statements.size() + " field of research codes. Insert skipped.");
127                    }
128                }
129                catch (final SQLException e)
130                {
131                    LOG.error("Could not execute script " + this.sqlFile.getAbsolutePath(), e);
132                }
133                finally
134                {
135                    try
136                    {
137                        connection.close();
138                    }
139                    catch (final SQLException e)
140                    {
141                        LOG.error("Could not close connection", e);
142                    }
143                    em.close();
144                }
145            }
146            finally
147            {
148                monitor.done();
149            }
150            return Status.OK_STATUS;
151        }
152    
153    }
