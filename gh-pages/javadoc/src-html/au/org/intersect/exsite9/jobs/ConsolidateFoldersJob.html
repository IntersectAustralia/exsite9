001    package au.org.intersect.exsite9.jobs;
002    
003    import java.util.List;
004    
005    import org.eclipse.core.runtime.IProgressMonitor;
006    import org.eclipse.core.runtime.IStatus;
007    import org.eclipse.core.runtime.Status;
008    import org.eclipse.core.runtime.jobs.Job;
009    import org.eclipse.swt.widgets.Display;
010    import org.eclipse.ui.PlatformUI;
011    
012    import au.org.intersect.exsite9.domain.Folder;
013    import au.org.intersect.exsite9.domain.Project;
014    import au.org.intersect.exsite9.service.IProjectManager;
015    import au.org.intersect.exsite9.service.IResearchFileService;
016    import au.org.intersect.exsite9.view.ProjectExplorerView;
017    import au.org.intersect.exsite9.view.ViewUtils;
018    
019    public class ConsolidateFoldersJob extends Job
020    {
021    
022        private static final String jobName = "ConsolidateFoldersForProject";
023        
024        private final Folder parentFolder;
025        private final List<Folder> subFolders;
026        
027        public ConsolidateFoldersJob(Folder parentFolder, List<Folder>subFolders)
028        {
029            super(jobName);
030            this.parentFolder = parentFolder;
031            this.subFolders = subFolders;
032        }
033        
034        @Override
035        protected IStatus run(IProgressMonitor monitor)
036        {
037            try
038            {
039                SemaphoreManager.getBackgroundJobSemaphore().acquire();
040                
041                final IProjectManager projectManager = (IProjectManager) PlatformUI.getWorkbench().getService(IProjectManager.class);
042                final Project project = projectManager.getCurrentProject();
043         
044                if (project == null || project.getId() == null)
045                {
046                    return Status.CANCEL_STATUS;
047                }
048         
049                monitor.beginTask("Consolidating Folders", IProgressMonitor.UNKNOWN);
050                
051                final IResearchFileService fileService = (IResearchFileService) PlatformUI.getWorkbench().getService(IResearchFileService.class);
052                
053                for(final Folder subFolder : subFolders)
054                {
055                    fileService.consolidateSubFolderIntoParentFolder(project, parentFolder, subFolder);
056                }
057                
058                // Refresh the project explorer view in the ui thread
059                Display.getDefault().syncExec(new Runnable() {
060                    @Override
061                    public void run() {
062                        final ProjectExplorerView view = (ProjectExplorerView) ViewUtils.getViewByID(PlatformUI.getWorkbench().getActiveWorkbenchWindow(), ProjectExplorerView.ID);
063                        view.refresh();
064                    }
065                });
066                
067                return Status.OK_STATUS;
068            }
069            catch(InterruptedException ie)
070            {
071                // TODO: handle this
072                return Status.CANCEL_STATUS;
073            }
074            finally
075            {
076                SemaphoreManager.getBackgroundJobSemaphore().release();
077                monitor.done();
078            }
079        }
080    
081    }
