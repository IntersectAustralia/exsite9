001    package au.org.intersect.exsite9.jobs;
002    
003    import java.util.concurrent.Semaphore;
004    
005    /**
006     * Manages the semaphore that co-ordinates the running of background jobs
007     * that affect files and folders so that only one such job can run at
008     * a time.
009     */
010    public class SemaphoreManager
011    {
012        
013        private static final Semaphore backgroundJobSemaphore = new Semaphore(1);
014        
015        public static Semaphore getBackgroundJobSemaphore()
016        {
017            return backgroundJobSemaphore;
018        }
019    }
