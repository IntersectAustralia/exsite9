007    package au.org.intersect.exsite9.jobs;
008    
009    import org.eclipse.core.runtime.jobs.Job;
010    
011    /**
012     * for jobs that need repeating
013     */
014    public abstract class AbstractRepeatingJob extends Job
015    {
016        /**
017         *  Flag to indicate that the job should continue to run to schedule
018         */
019        private boolean running = true;
020    
021        /**
022         * The frequency the job runs at
023         */
024        protected long repeatDelayMillis = 0L;
025        
026        /**
027         * A repeating job.
028         * @param name The Job Name
029         * @param repeatDelayMillis The schedule interval in milliseconds
030         */
031        public AbstractRepeatingJob(String name, long repeatDelayMillis)
032        {
033            super(name);
034            this.repeatDelayMillis = repeatDelayMillis;
035        }
036        
037        @Override
038        public boolean shouldSchedule() {
039            return running;
040        }
041        
042        public void start() {
043            running = true;
044            schedule(repeatDelayMillis);
045        }
046        
047        public void stop() {
048            running = false;
049            cancel();
050        }
051    
052        public void setDelay(final long delay)
053        {
054            this.repeatDelayMillis = delay;
055        }
056    }
