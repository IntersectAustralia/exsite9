007    package au.org.intersect.exsite9.jobs;
008    
009    import org.apache.log4j.Logger;
010    import org.eclipse.core.runtime.IProgressMonitor;
011    import org.eclipse.core.runtime.IStatus;
012    import org.eclipse.core.runtime.Status;
013    import org.eclipse.core.runtime.jobs.Job;
014    import org.eclipse.jface.util.IPropertyChangeListener;
015    import org.eclipse.jface.util.PropertyChangeEvent;
016    
017    import au.org.intersect.exsite9.Activator;
018    import au.org.intersect.exsite9.initialization.ConfigurationInitializer;
019    
020    /**
021     * For repeating {@link IdentifyAllNewFilesForProjectJob}
022     */
023    public class IdentifyAllNewFilesForProjectRepeatingJob extends AbstractRepeatingJob implements IPropertyChangeListener
024    {
025        private static final Logger LOG = Logger.getLogger(IdentifyAllNewFilesForProjectRepeatingJob.class);
026        
027        private static final String jobName = "IdentifyAllNewFilesForProjectRepeatingJob";
028        
029        public IdentifyAllNewFilesForProjectRepeatingJob(long repeatDelayMillis)
030        {
031            super(jobName, repeatDelayMillis);
032            Activator.getDefault().getPreferenceStore().addPropertyChangeListener(this);
033        }
034        
035        @Override
036        protected IStatus run(IProgressMonitor monitor)
037        {
038            try
039            {
040                LOG.debug("Starting scheduled job to find files. Current delay = " + repeatDelayMillis + "ms");
041                final Job identifyAllNewFilesForProject = new IdentifyAllNewFilesForProjectJob();
042                identifyAllNewFilesForProject.schedule();
043                return Status.OK_STATUS;
044            }
045            finally
046            {
047                if(shouldSchedule())
048                {
049                    this.schedule(this.repeatDelayMillis);
050                }
051                LOG.debug("End of scheduled job to find files.");
052            }
053        }
054    
055        @Override
056        public void propertyChange(final PropertyChangeEvent event)
057        {
058            if (event.getProperty() == ConfigurationInitializer.FOLDER_RELOAD_INTERVAL_KEY)
059            {
060                final int refreshTimeMins = Activator.getDefault().getPreferenceStore().getInt(ConfigurationInitializer.FOLDER_RELOAD_INTERVAL_KEY);
061                if (refreshTimeMins > 0)
062                {
063                    final long delayMillis = refreshTimeMins * 60 * 1000;
064                    stop();
065                    setDelay(delayMillis);
066                    start();
067                    LOG.info("Configured interval on scheduled job to find files to " + delayMillis + "ms");
068                }
069                else
070                {
071                    stop();
072                    setDelay(0);
073                    LOG.info("Disabled scheduled job to find files.");
074                }
075            }
076        }
077    }
