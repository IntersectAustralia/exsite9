007    package au.org.intersect.exsite9.initialization;
008    
009    import org.eclipse.core.runtime.preferences.AbstractPreferenceInitializer;
010    import org.eclipse.jface.preference.IPreferenceStore;
011    
012    import au.org.intersect.exsite9.Activator;
013    
014    /**
015     * Configures default values for configuration items.
016     */
017    public class ConfigurationInitializer extends AbstractPreferenceInitializer
018    {
019        /**
020         * The key used to represent the folder reload interval (in minutes).
021         */
022        public static final String FOLDER_RELOAD_INTERVAL_KEY = "exsite9.folder.refresh.interval";
023        public static final int FOLDER_RELOAD_INTERVAL_DEFAULT = 0;
024    
025        /**
026         * 
027         */
028        public ConfigurationInitializer()
029        {
030        }
031    
032        /**
033         * @{inheritDoc}
034         */
035        @Override
036        public void initializeDefaultPreferences()
037        {
038            final IPreferenceStore store = Activator.getDefault().getPreferenceStore();
039            store.setDefault(FOLDER_RELOAD_INTERVAL_KEY, FOLDER_RELOAD_INTERVAL_DEFAULT);
040        }
041    }
