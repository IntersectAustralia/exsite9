007    package au.org.intersect.exsite9.dao;
008    
009    import java.util.List;
010    
011    import javax.persistence.EntityManager;
012    import javax.persistence.TypedQuery;
013    
014    import au.org.intersect.exsite9.domain.FieldOfResearch;
015    
016    /**
017     * Data Access Object for {@link FieldOfResearch} instances
018     */
019    public final class FieldOfResearchDAO
020    {
021        private final EntityManager em;
022    
023        public FieldOfResearchDAO(final EntityManager em)
024        {
025            this.em = em;
026        }
027    
028        /**
029         * Obtains all the {@link FieldOfResearch} instances.
030         * @return All the fields of research.
031         */
032        public List<FieldOfResearch> getAll()
033        {
034            final TypedQuery<FieldOfResearch> query = em.createQuery("SELECT for FROM FieldOfResearch for ORDER BY for.code", FieldOfResearch.class);
035            return query.getResultList();
036        }
037    }
