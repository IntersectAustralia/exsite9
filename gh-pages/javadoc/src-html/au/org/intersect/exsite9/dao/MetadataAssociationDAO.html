007    package au.org.intersect.exsite9.dao;
008    
009    import javax.persistence.EntityManager;
010    
011    import au.org.intersect.exsite9.domain.MetadataAssociation;
012    
013    /**
014     * Data Access Object for {@link MetadataAssociation} instances
015     */
016    public final class MetadataAssociationDAO
017    {
018        private EntityManager em;
019    
020        public MetadataAssociationDAO(final EntityManager em)
021        {
022            this.em = em;
023        }
024    
025        public void createMetadataAssociation(final MetadataAssociation metadataAssociation)
026        {
027            em.getTransaction().begin();
028            em.persist(metadataAssociation);
029            em.getTransaction().commit();
030        }
031    
032        public void updateMetadataAssociation(final MetadataAssociation metadataAssociation)
033        {
034            em.getTransaction().begin();
035            em.merge(metadataAssociation);
036            em.getTransaction().commit();
037        }
038    
039        public void removeMetadataAssociation(final MetadataAssociation metadataAssociation)
040        {
041            final boolean transactionActive = em.getTransaction().isActive();
042            if (!transactionActive){em.getTransaction().begin();}
043            em.remove(em.merge(metadataAssociation));
044            if (!transactionActive){em.getTransaction().commit();}
045        }
046    }
