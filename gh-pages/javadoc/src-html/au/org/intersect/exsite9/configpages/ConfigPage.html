007    package au.org.intersect.exsite9.configpages;
008    
009    import org.eclipse.jface.preference.FieldEditorPreferencePage;
010    import org.eclipse.jface.preference.IntegerFieldEditor;
011    import org.eclipse.ui.IWorkbench;
012    import org.eclipse.ui.IWorkbenchPreferencePage;
013    
014    import au.org.intersect.exsite9.Activator;
015    import au.org.intersect.exsite9.initialization.ConfigurationInitializer;
016    
017    /**
018     * The configuration page.
019     * Used to get/set configuration of the application from the user.
020     */
021    public class ConfigPage extends FieldEditorPreferencePage implements IWorkbenchPreferencePage
022    {
023    
024        @Override
025        public void init(final IWorkbench workbench)
026        {
027            setPreferenceStore(Activator.getDefault().getPreferenceStore());
028            setDescription("Configure ExSite9");
029        }
030    
031        @Override
032        protected void createFieldEditors()
033        {
034            final IntegerFieldEditor refreshIntervalFieldEditor = new IntegerFieldEditor(ConfigurationInitializer.FOLDER_RELOAD_INTERVAL_KEY,
035                    "&Folder refresh interval (minutes):", getFieldEditorParent(), 3);
036            refreshIntervalFieldEditor.setEmptyStringAllowed(false);
037            refreshIntervalFieldEditor.setValidRange(0, 500);
038            addField(refreshIntervalFieldEditor);
039        }
040    }
